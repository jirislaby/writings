\begin{frame}
  \frametitle{Záměr práce -- Co a proč?}
  \begin{itemize}
  \item Automatická kontrola kódu
  \begin{itemize}
  \item Lidé se nechtějí učit složité nástroje
  \item Vstup: \emph{kód} a popř. definice kontrol (jinak možná autodetekce)
  \item Výstup: \emph{seznam chyb} (co nejméně falešných)
  \end{itemize}\pause
  \item Podpora jádra
  \begin{itemize}
  \item Chyby v jádře končí katastrofou
  \item Obsáhlý kód, rozmanitost
  \end{itemize}\pause
  \item Podpora Linuxového jádra
  \begin{itemize}
  \item Je rošířené, testované, přesto jsou v něm chyby -- výzva
  \item Ignoruje uváznutí, čeká že nenastane
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Záměr práce -- Jak?}
  \begin{enumerate}
  \item Vstup: překladový systém projektu
  \begin{itemize}
  \item Ví přesně o zdrojových kódech
  \item Automatický překlad všeho potřebného
  \end{itemize}\pause
  \item Symbolický průchod přeloženého kódu
  \begin{itemize}
  \item Nalezne (zatím jen) podezřelá místa v kódu
  \end{itemize}\pause
  \item Generace testů
  \begin{itemize}
  \item Obsahují vstup odpovídající "path-condition"
  \item Také odkaz na místo v kódu
  \end{itemize}\pause
  \item Spuštění testů a report chyb
  \begin{itemize}
  \item Úspěšné testy vypíší odkaz do kódu
  \end{itemize}
  \end{enumerate}

  \pause
  \medskip

  \heading{Konkrétněji na další obrazovce}

\end{frame}

\begin{frame}
  \frametitle{Stav práce}
  \heading{Učící fáze: \textsc{Stanse}}
  \begin{itemize}
  \item Statická analýza, počítá fix-point abstraktní interpretace
  \item Podpora jádra Linuxu, nalezeno několik desítek chyb
  \end{itemize}

  \pause
  \medskip

  \heading{Nyní nový nástroj}
  \begin{itemize}
  \item Proběhl výběr algoritmu (dle plánu tezí)
  \item Založený na:
  \begin{itemize}
  \item \textsc{Stanse} -- překladový systém + preprocesor
  \item \textsc{Sparse} -- parser pro vyhledání funkcí, struktur
  \item \textsc{Klee} -- symbolická exekuce, výstup "ktest"
  \item \textsc{Llvm} -- překladač do mezikódu, vstup pro \textsc{Klee}
  \end{itemize}\pause
  \item Nové moduly:
  \begin{itemize}
  \item Nalezení startovních funkcí (jádro/moduly nemají \texttt{main})
  \item Generace a spouštění testů z výstupu \textsc{Klee}
  \end{itemize}\pause
  \item Nový Markův algoritmus v \textsc{Klee}
  \end{itemize}

\end{frame}
