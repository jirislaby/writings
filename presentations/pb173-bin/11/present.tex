\part{Roury}
\frame{\partpage}

\begin{frame}
  \frametitle{Roury}

  \begin{itemize}
    \item Jednosměrná komunikace
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje deskriptory
    \end{itemize}

    \pause

    \item Vytvoření: \code{pipe}
    \begin{itemize}
      \item Vrací pole 2 deskriptorů (+ návratovou hodnotu)
      \item Jeden pro potomka, druhý pro rodiče

      \pause

      \item Provede se \code{fork}
      \item Zavřou se nepotřebné deskriptory
      \item Čte se z prvního deskriptoru (\code{read})
      \item Zapisuje se do druhého deskriptoru (\code{write})
    \end{itemize}

    \pause

    \item \code{popen}=\code{pipe}+\code{fork}+\code{exec}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vytvoření ,,echa'' přes roury}
  \begin{enumerate}
    \item Vytvořte si 2 roury
    \begin{itemize}
      \item Pro obousměrnou komunikaci
    \end{itemize}
    \item Proveďte \code{fork}
    \item Zavřete nepotřebné deskriptory

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Do jednoho \code{fd} zapisujte měnící se text
      \begin{itemize}
	\item Např.\ \code{char buf[10] = "Hello";} a \code{(*buf)++;}
      \end{itemize}
      \item Z druhého \code{fd} čtěte a vypisujte
    \end{itemize}

    \item V druhém procesu (echo):
    \begin{itemize}
      \item Čtěte z prvního \code{fd}
      \item A zapisujte to do druhého \code{fd}
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}

\part{Sdílená paměť}
\frame{\partpage}

\begin{frame}
  \frametitle{Sdílená paměť}

  \begin{itemize}
    \item Mapování a sdílení paměti
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje mapování
    \end{itemize}

    \pause

    \item Vytvoření: \code{mmap} s \code{MAP_ANONYMOUS | MAP_SHARED}
    \begin{itemize}
      \item Provede se \code{fork}
      \item Paměť se může libovolně měnit
      \item Ale pozor na optimalizace překladače
      \begin{itemize}
	\item \code{volatile}
      \end{itemize}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vytvoření sdílené paměti}
  \begin{enumerate}
    \item Vytvořte si mapování
    \item Proveďte \code{fork}

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Zapisujte do paměti proměnný řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item V druhém procesu:
    \begin{itemize}
      \item Čtěte a vypište řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}

\section{Semafory}

\begin{frame}[fragile]
  \frametitle{Semafory (IPC)}

  \pause

  \begin{itemize}
    \item Meziprocesové zamykání/správa prostředků

    \pause

    \item Vytvoření semaforu: \code{semget}

    \pause

    \item Změna stavu semaforu: \code{semop}

    \pause

    \item Další operace: \code{semctl}
    \begin{itemize}
      \item \emph{Nastavení iniciální hodnoty}: \code{SETVAL} (spolu s
	\code{union semun.val})
      \item Informace: \code{SEM_STAT}
      \item Zrušení: \code{IPC_RMID}
      \item \dots
    \end{itemize}
  \end{itemize}

  \heading{Parametr \code{semop}}
  \begin{lstlisting}
struct sembuf {
  ushort  sem_num;
  short   sem_op;  /*  -  0  +  */
  short   sem_flg;
};
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce se semaforem (IPC)}
  \begin{enumerate}
    \item Vytvořte si klíč (\code{ftok})
    \begin{itemize}
      \item A vypište ho
    \end{itemize}
    \item Vytvořte 2 programy

    \item Jeden tiskový server
    \begin{itemize}
      \item Vytvořte si 1 semafor (\code{semget})
      \item Obsluhuje 5 tiskáren (\code{shmctl} a \code{SETVAL})
      \item Tisk trvá 10 vteřin
    \end{itemize}

    \item Druhý je klient
    \begin{itemize}
      \item Chce tisknout
    \end{itemize}

    \item Spusťte jeden server a více než 5 klientů
    \begin{itemize}
      \item Do 10 vteřin
    \end{itemize}
  \end{enumerate}
