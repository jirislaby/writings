\begin{frame}{Anonymní sdílená paměť}

  \begin{itemize}
    \item Mapování a sdílení paměti
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje mapování
    \end{itemize}

    \pause

    \item Vytvoření: \code{mmap} s \code{MAP_ANONYMOUS | MAP_SHARED}
    \begin{itemize}
      \item Provede se \code{fork}
      \item Paměť je volně k dispozici oběma procesům
      \item Jen pozor na optimalizace překladače
      \begin{itemize}
	\item \code{volatile}
      \end{itemize}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Vytvoření sdílené paměti}
  \begin{enumerate}
    \item Vytvořte si sdílené mapování
    \item Proveďte \code{fork}

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Zapisujte do paměti proměnný řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item V druhém procesu:
    \begin{itemize}
      \item Čtěte a vypište řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}
