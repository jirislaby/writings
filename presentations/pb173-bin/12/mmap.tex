\begin{frame}[fragile]{Anonymní sdílená paměť}

  \begin{itemize}
    \item Mapování a sdílení paměti
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje mapování
    \end{itemize}

    \pause

    \item Vytvoření: \code{mmap} s \code{MAP_ANONYMOUS | MAP_SHARED}
    \begin{itemize}
      \item Provede se \code{fork}
      \item Paměť je volně k dispozici oběma procesům

      \pause

      \item Jen pozor na optimalizace překladače
      \begin{itemize}
	\item \code{volatile}
      \end{itemize}
    \end{itemize}
  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}
volatile char *map;
map = mmap(NULL, 1 << 20, PROT_READ | PROT_WRITE, MAP_ANONYMOUS | MAP_SHARED, -1, 0);
if (fork())
  map[0] = 'A';
else
  map[4096] = 'B';
  \end{lstlisting}
  \end{block}

\end{frame}

\begin{frame}{Úkol}

  \heading{Vytvoření sdílené paměti}
  \begin{enumerate}
    \item Vytvořte si sdílené mapování
    \item Proveďte \code{fork}

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Zapisujte do paměti proměnný řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item V druhém procesu:
    \begin{itemize}
      \item Čtěte a vypisujte řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}
