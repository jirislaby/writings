\section{Roura}
\frame{\sectionpage}

\begin{frame}
  \frametitle{Roury}

  \begin{itemize}
    \item Jednosměrná komunikace
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje deskriptory
    \end{itemize}

    \pause

    \item Vytvoření: \code{pipe}
    \begin{itemize}
      \item Vrací pole 2 deskriptorů (+ návratovou hodnotu)
      \item Jeden pro potomka, druhý pro rodiče

      \pause

      \item Provede se \code{fork}
      \item Zavřou se nepotřebné deskriptory
      \item Čte se z prvního deskriptoru (\code{read})
      \item Zapisuje se do druhého deskriptoru (\code{write})
    \end{itemize}

    \pause

    \item \code{popen}=\code{pipe}+\code{fork}+\code{exec}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vytvoření ,,echa'' přes roury}
  \begin{enumerate}
    \item Vytvořte si 2 roury
    \begin{itemize}
      \item Pro obousměrnou komunikaci
      \item Budete mít \code{fd1[2]} a \code{fd2[2]}
    \end{itemize}
    \item Proveďte \code{fork}
    \item Zavřete nepotřebné deskriptory

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Do \code{fd1[1]} zapisujte měnící se text
      \begin{itemize}
	\item Např.\ \code{char buf[10] = "Hello";} a \code{(*buf)++;}
      \end{itemize}
      \item Z \code{fd2[0]} čtěte a vypisujte na standardní výstup
    \end{itemize}

    \item V druhém procesu (loopback):
    \begin{itemize}
      \item Čtěte z \code{fd1[0]}
      \item A zapisujte to do \code{fd2[1]}
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}

\section{Sdílená paměť}
\frame{\sectionpage}

\begin{frame}
  \frametitle{Sdílená paměť}

  \begin{itemize}
    \item Mapování a sdílení paměti
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje mapování
    \end{itemize}

    \pause

    \item Vytvoření: \code{mmap} s \code{MAP_ANONYMOUS | MAP_SHARED}
    \begin{itemize}
      \item Provede se \code{fork}
      \item Paměť je volně k dispozici oběma procesům
      \item Jen pozor na optimalizace překladače
      \begin{itemize}
	\item \code{volatile}
      \end{itemize}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vytvoření sdílené paměti}
  \begin{enumerate}
    \item Vytvořte si sdílené mapování
    \item Proveďte \code{fork}

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Zapisujte do paměti proměnný řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item V druhém procesu:
    \begin{itemize}
      \item Čtěte a vypište řetězec
      \item Čekejte 100 ms
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}
