\begin{frame}[fragile]{Přesměrování funkcí}

  \begin{itemize}
    \item Knihovny mohou podvrhnout své implementace funkcí
    \begin{itemize}
      \item Např.\ \code{malloc} pro sledování paměti
    \end{itemize}

    \pause

    \item Podmínka: funkce musí být z dynamické knihovny

    \pause

    \item \code{LD_PRELOAD} nahraje knihovnu i se zvláštní implementací

    \pause

    \item \code{dlsym} lze použít k nalezení originálu
    \begin{itemize}
      \item \code{void *dlsym(void *handle, const char *symbol)}
      \item Speciální \code{handle} -- \code{RTLD_NEXT}
    \end{itemize}
  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}[aboveskip=2pt,belowskip=2pt]
void *malloc(size_t size)
{
  static void *(*c_malloc)(size_t);
  if (!c_malloc)
    c_malloc = dlsym(RTLD_NEXT, "malloc");

  write(1, "Somebody called malloc!\n", 24);
  return c_malloc(size);
}
  \end{lstlisting}
  \end{block}

\end{frame}
