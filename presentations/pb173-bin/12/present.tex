\section{Roura}
\frame{\sectionpage}

\begin{frame}
  \frametitle{Roury}

  \begin{itemize}
    \item Jednosměrná komunikace
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje deskriptory
    \end{itemize}

    \pause

    \item Vytvoření: \code{pipe}
    \begin{itemize}
      \item Vrací pole 2 deskriptorů (+ návratovou hodnotu)
      \item Jeden pro potomka, druhý pro rodiče

      \pause

      \item Provede se \code{fork}
      \item Zavřou se nepotřebné deskriptory
      \item Čte se z prvního deskriptoru (\code{read})
      \item Zapisuje se do druhého deskriptoru (\code{write})
    \end{itemize}

    \pause

    \item \code{popen}=\code{pipe}+\code{fork}+\code{exec}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vytvoření ,,echa'' přes roury}
  \begin{enumerate}
    \item Vytvořte si 2 roury
    \begin{itemize}
      \item Pro obousměrnou komunikaci
      \item Budete mít \code{fd1[2]} a \code{fd2[2]}
    \end{itemize}
    \item Proveďte \code{fork}
    \item Zavřete nepotřebné deskriptory

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Do \code{fd1[1]} zapisujte měnící se text
      \begin{itemize}
	\item Např.\ \code{char buf[10] = "Hello";} a \code{(*buf)++;}
      \end{itemize}
      \item Z \code{fd2[0]} čtěte a vypisujte na standardní výstup
    \end{itemize}

    \item V druhém procesu (loopback):
    \begin{itemize}
      \item Čtěte z \code{fd1[0]}
      \item A zapisujte to do \code{fd2[1]}
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}

    \end{itemize}
  \end{itemize}

\subsection{Semafory}

\begin{frame}[fragile]{IPC}{Semafory}

  \begin{itemize}
    \item \header{sys/sem.h}
    \item<+-> Meziprocesové zamykání/správa prostředků
    \item<+-> Vytvoření semaforu: \code{semget}
    \item<+-> Změna stavu semaforu: \code{semop}
    \item<+-> Další operace: \code{semctl}
    \begin{itemize}
      \item \emph{Nastavení iniciální hodnoty}: \code{SETVAL} (spolu s
	\code{union semun.val})
      \item Informace: \code{SEM_STAT}
      \item Zrušení: \code{IPC_RMID}
      \item \dots
    \end{itemize}
  \end{itemize}

  \begin{uncoverenv}<3->
  \begin{block}{Parametr \code{semop}}
  \begin{lstlisting}
struct sembuf {
  ushort  sem_num;

};
  \end{lstlisting}
  \end{block}
  \end{uncoverenv}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce se semaforem (IPC)}
  \begin{enumerate}
    \item Vytvořte si klíč (\code{ftok})
    \begin{itemize}
      \item A vypište ho (\code{\%x})
    \end{itemize}
    \item Vytvořte 2 programy

    \item Jeden program je tiskovým serverem
    \begin{itemize}
      \item Vytvořte si 1 semafor (\code{semget})
      \item Obsluhuje 5 tiskáren (\code{shmctl} a \code{SETVAL})
      \item Tisk trvá 10 vteřin
    \end{itemize}

    \item Druhý program je klientem
    \begin{itemize}
      \item Chce tisknout
    \end{itemize}

    \item Spusťte jeden server a více než 5 klientů
    \begin{itemize}
      \item Do 10 vteřin
    \end{itemize}
  \end{enumerate}

\end{frame}
