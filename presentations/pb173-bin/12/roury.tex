\begin{frame}{Roura}

  \begin{itemize}
    \item Jednosměrná komunikace
    \begin{itemize}
      \item Jádro při \code{fork} kopíruje deskriptory
    \end{itemize}

    \pause

    \item Vytvoření: \code{int pipe(int pipefd[2])} (\header{unistd.h})
    \begin{itemize}
      \item Vrací pole 2 deskriptorů (+ návratovou hodnotu)
      \item Jeden pro potomka, druhý pro rodiče
    \end{itemize}

    \pause

    \item Omezení
    \begin{itemize}
      \item Čte se z prvního deskriptoru (\code{read})
      \item Zapisuje se do druhého deskriptoru (\code{write})
    \end{itemize}

    \pause

    \item Operace
    \begin{enumerate}
      \item Provede se \code{pipe} a \code{fork}
      \item Zavřou se nepotřebné deskriptory
      \item Rodič čte a potomek zapisuje (nebo obráceně)
    \end{enumerate}

    \pause

    \item \code{popen}=\code{pipe}+\code{fork}+\code{exec}
  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Vytvoření ,,echa'' přes roury}
  \begin{enumerate}
    \item Vytvořte si 2 roury
    \begin{itemize}
      \item Pro obousměrnou komunikaci
      \item Budete mít \code{fd1[2]} a \code{fd2[2]}
    \end{itemize}
    \item Proveďte \code{fork}
    \item Zavřete nepotřebné deskriptory

    \item V jednom procesu v cyklu o několika iteracích:
    \begin{itemize}
      \item Do \code{fd1[1]} zapisujte měnící se text
      \begin{itemize}
	\item Např.\ \code{char buf[10] = "Hello";} a \code{(*buf)++;}
      \end{itemize}
      \item Z \code{fd2[0]} čtěte a vypisujte na standardní výstup
    \end{itemize}

    \item V druhém procesu (loopback):
    \begin{itemize}
      \item Čtěte z \code{fd1[0]}
      \item A zapisujte to do \code{fd2[1]}
    \end{itemize}

    \item Spusťte
  \end{enumerate}

\end{frame}
