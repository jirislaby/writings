\subsection{\cmd{gdb}}

\begin{frame}
  \frametitle{Debugger \cmd{gdb}}

  \begin{itemize}
    \item Pracuje s ELFem a DWARFem

    \item Používá systémové volání \code{ptrace}
    \begin{itemize}
      \item Sleduje proces
      \item Upravuje vykonávání
      \item Čte a mění paměť
      \item \dots
    \end{itemize}

    \pause

    \item Podpora Python skriptů
    \item Spuštění: \cmd{gdb --volby_gdb --args binarka --volby_binarky}
    \item Nápověda: \cmd{help}
    \item Manuál: \href{https://sourceware.org/gdb/current/onlinedocs/gdb/}{GDB
      User Manual}
  \end{itemize}

\end{frame}

\begin{frame}{Příkazy \cmd{gdb}}{Operace s během}

  \begin{itemize}
    \item Spustit: \cmd{run}
    \item Spustit po \code{main}: \cmd{start}
    \item Přerušit: signálem (např.\ Ctrl-c)
    \item Pokračovat po návrat z funkce: \file{finish}
    \item Pokračovat: \file{continue}

    \pause

    \item Krokovat s vnořením: \cmd{step} (po řádcích), \cmd{stepi} (po
      instrukcích)
    \item Krokovat bez vnoření: \cmd{next} (po řádcích), \cmd{nexti} (po
      instrukcích)

    \item Konec: \cmd{quit}

    \medskip
    \pause

    \item Opakování předchozího příkazu: ENTER
  \end{itemize}

  \pause
  \bigskip

  \heading{Úkol:} spusťte si \cmd{yes} v \cmd{gdb} a přerušte. Vyzkoušejte si
  \file{finish} a \file{continue}. Zkuste krokovat po instrukcích hlavní cyklus
  \cmd{yes}. Poté \cmd{quit}.

\end{frame}

\begin{frame}{Příkazy \cmd{gdb}}{Výpisy}

  \begin{itemize}
    \item Zásobník volání: \cmd{where}
    \begin{itemize}
      \item Pohyb po rámcích zásobníku: \cmd{up}, \cmd{down}
    \end{itemize}

    \pause

    \item Programu: \cmd{list} (jen s ladicími informacemi)
    \item Assembleru: \cmd{disassemble}
    \item Paměti: \cmd{print}
    \item HexDump: \cmd{x}
    \item Ostatní: \cmd{info registers}, \file{info break}, \dots
  \end{itemize}

  \bigskip

  Většina příkazů akceptuje parametry -- výpis lokací, proměnných, registrů atd.

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Výpisy v \cmd{gdb}}
  \begin{enumerate}
    \item Přeložte si \file{pb173-bin/08/}
    \item Spusťte \file{debug} (mělo by dojít k pádu)
    \item Otevřete \file{debug} v \cmd{gdb}
    \item Spusťte (\cmd{run})

    \item Prohlédněte si, kde došlo k pádu (vypište):
    \begin{itemize}
      \item Zásobník volání (\cmd{where})
      \item Místo pádu (\cmd{list} s parametrem \code{soubor:radek})
      \item Disassembly (\cmd{disassemble} s parametrem \code{adresa})
      \item Proměnnou \code{b} (\cmd{print b})
      \item Proměnnou \code{argc} v \code{main} (\cmd{up} a \cmd{print argc})
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}{Příkazy \cmd{gdb}}{Breakpointy}

  \begin{itemize}

  \end{enumerate}
\end{frame}

\subsection{Obrazy paměti}

\begin{frame}
  \frametitle{Obrazy paměti a \cmd{gdb}}

  \begin{itemize}
    \item Jádro při pádu zapíše obraz paměti (,,coredump'')
    \begin{itemize}
      \item Nutno zvýšit limit velikosti (bývá 0): \cmd{ulimit -c}
      \item Cíl, kam se ukládají: \cmd{sysctl kernel.core_pattern}
    \end{itemize}

    \pause

    \item Potom se dá analyzovat v \cmd{gdb}
    \begin{itemize}
      \item \cmd{gdb --core=core_soubor --args binarka --volby_binarky}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s obrazem paměti}
  \begin{enumerate}
    \item Nastavte limit na obrazy na \code{unlimited} (\cmd{ulimit -c})
    \item Podívejte se, kam se obraz uloží (\cmd{sysctl kernel.core_pattern})
    \item Spusťte program z předchozího příkladu
    \item Otevřete v \cmd{gdb} s obrazem paměti (\cmd{gdb --core=<core>
      ./debug})
    \item Vypište si podobné informace jako předtím
    \begin{itemize}
      \item \cmd{where}
      \item \cmd{print b}
      \item A podobně
    \end{itemize}
  \end{enumerate}

\end{frame}

\subsection{Tracery}

\begin{frame}
  \frametitle{Tracery}

  \begin{itemize}
    \item Sledují události a vypisují je
    \item \cmd{ltrace}: sleduje knihovní volání
    \item \cmd{strace}: sleduje systémová volání
    \item Používají systémové volání \code{ptrace}
    \begin{itemize}
      \item \cmd{ltrace} si nastaví breakpointy na všechny externí symboly z
	ELFu
      \item \cmd{strace} jednoduše použije \code{PTRACE_SYSCALL}
    \end{itemize}

    \pause

    \item Spuštění: \cmd{\{s,l\}trace binarka --volby}
    \item Další parametry
    \begin{itemize}
      \item \texttt{-e}: filtr (\texttt{-e write})
      \item \texttt{-f}: sleduje i potomky
      \item \texttt{-o}: výstup do souboru
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s tracery}
  \begin{enumerate}
    \item Projděte si \file{pb173-bin/08/tracer.c}
    \item Spusťte nejdříve s \cmd{ltrace}
    \begin{itemize}
      \item Kolik je volání \code{fwrite} do \file{libc}?
    \end{itemize}
    \item Spusťte s \cmd{strace}
    \begin{itemize}
      \item Kolik je volání \code{write} do jádra?
      \item $\Rightarrow$ \file{libc} bufferuje
    \end{itemize}
    \item Pomocí \cmd{ltrace} zjistěte, jak velký buffer \file{libc} používá
    \begin{itemize}
      \item V cyklu vypisujte znak pomocí \code{fwrite}
      \item Použijte volbu \texttt{-S}
    \end{itemize}
    \item Pomocí \cmd{strace} ověřte totéž
    \begin{itemize}
      \item Pozorujte třetí parametr \code{write}
    \end{itemize}
  \end{enumerate}

\end{frame}

\subsection{\cmd{valgrind}}

\begin{frame}
  \frametitle{\cmd{valgrind}}

  \begin{itemize}
    \item Sleduje (zejména) operace s pamětí (\emph{Memcheck})
    \begin{itemize}
      \item Úniky paměti
      \item Neinicializované a nezarovnané přístupy
      \item Použití paměti po \code{free}
      \item \dots
    \end{itemize}

    \pause

    \item Několik dalších modulů
    \begin{itemize}
      \item Callgrind: profiluje cache CPU
      \item Massif: profiluje použití haldy
      \item Další pro problémy se zámky atd.
    \end{itemize}

    \pause

    \item Spuštění: \cmd{valgrind --volby_valgrindu binarka --volby_binarky}
    \item Další parametry
    \begin{itemize}
      \item \texttt{-\!-tool}: výběr nástroje shora
      \item \texttt{-\!-leak-check}: sledovat úniky paměti
    \end{itemize}
  \end{itemize}

\end{frame}

\end{frame}
