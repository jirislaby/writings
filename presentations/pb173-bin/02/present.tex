\begin{frame}[fragile]
  \frametitle{Analýzy v \ant}

  \heading{\ant obsahuje obě části v jednom souboru}
  \begin{itemize}
    \item 2 typy pravidel
    \begin{itemize}
      \item Počáteční velké písmeno -- lexikální
      \item Počáteční malé písmeno -- syntaktická
    \end{itemize}
  \end{itemize}

  \pause

  \begin{lstlisting}[language=XML, columns=fullflexible]
/* syntakticka analyza */
helloWorld
  : AHOJ SVETE // 2 tokeny v jednom syntaktickem pravidle
  | SVETE AHOJ { puts("Nepise se to obracene?"); }
  ;

/* lexikalni analyza */
AHOJ
  : 'Ahoj' // vytvori token AHOJ z retezce "Ahoj"
  ;

SVETE
  : 'Svete'
  ;
  \end{lstlisting}
\end{frame}

\begin{frame}
  \frametitle{Pravá strana pravidel}

  \begin{itemize}
    \item \code{|} odděluje možnosti
    \begin{itemize}
      \item Více možností: \code{zvirata : savci | obojzivelnici ;}
      \item Kombinace podčásti: \code{zvirata : ('pe'|'ko') 's' ;}
    \end{itemize}

    \pause

    \item Opakování pravidla
    \begin{itemize}
      \item \code{pravidlo?} -- 0--1krát (\code{nicNeboX : 'x'? ;})
      \item \code{pravidlo*} -- 0--$\infty$krát (\code{kolonaAut : 'auto'* ;})
      \item \code{pravidlo+} -- 1--$\infty$krát (\code{zvukHada : 's'+ ;})
    \end{itemize}

    \item Speciální pravidlo pro konec souboru: \code{EOF}
    \begin{itemize}
      \item Donutí číst celý soubor
    \end{itemize}
  \end{itemize}

\end{frame}

    \item Vytvořte si tokeny \code{\+} a \code{\-} pro operace
    \item Vytvořte si nové pravidlo \code{expr} pro tyto 2 operace
    \begin{itemize}
      \item Bude obsahovat 2 řádky
      \item Pro sčítání např.\ \code{NUMBER PLUS NUMBER}
    \end{itemize}
    \item Přidejte do \code{translationUnit} novou pravou stranu \code{expr+}

    \item Vyzkoušejte nově podporované vstupy

      \item \code{1+2}, \code{100-10}, apod.
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Akce pravidel}

  \heading{Pravidla mohou obsahovat akce}
  \begin{itemize}
    \item Kód spouštěný při určitých událostech
    \item \code{p : ID \{ x++; \} ID \{ x++; \} ID \{ puts("neco"); \} ;}

    \pause

    \item Akce můžou referencovat předcházející části
    \begin{itemize}
      \item Implicitně jménem tokenu/pravidla (\code{p : NUMBER \{ \$NUMBER... \} ;})

      \pause

      \item Nebo pojmenováním (\code{p : n1=NUMBER PLUS n2=NUMBER \{ \$n1 ... \$n2 ... \} ;})
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Akce pravidel}
  \begin{enumerate}
    \item Dopište akce na konec jednotlivých pravidel \code{expr}
    \item Vypište nějaký text
    \item Vyzkoušejte
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Vlastnosti tokenů}

  \heading{Každý token má vlastnosti}
  \begin{itemize}
    \item \code{text} -- tokenizovaný řetězec (,,Ahoj'' pro AHOJ)
    \begin{itemize}
      \item Typ: \code{pANTLR3_STRING}, obsahuje \code{char *chars}
      \item Pravidla jej mají také
    \end{itemize}
    \item \code{line} -- zdrojový řádek
    \item \code{pos} -- zdrojový sloupec
  \end{itemize}

  \bigskip

  \begin{lstlisting}[language=XML, columns=fullflexible]
helloWorld
  : x=AHOJ y=SVETE {
    printf("line=\%d col=\%d x=\%s y=\%s whole=\%s\n",
      $x.line, $x.pos,
      $x.text->chars,
      $y.text->chars,
      $helloWorld.text->chars);
  }
  | SVETE AHOJ { puts("Nepise se to obracene?"); }
  ;
  \end{lstlisting} %stopzone

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vlastnosti pravidel}
  \begin{enumerate}
    \item Pro každou operaci sčítání/odčítání
    \begin{itemize}
      \item Vypište číslo řádku a sloupce pro obě čísla (\code{\$NUMBER.line} a
	\code{\$NUMBER.pos})
      \item Vypište obě čísla (\code{\$NUMBER.text->chars})
    \end{itemize}
    \item Vyzkoušejte roztroušením textu po testovacím vstupu
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Předávání hodnot}

  \heading{Každé pravidlo muže akceptovat/vracet hodnotu}
  \begin{itemize}
    \item \code{pravidlo[type1 param1, ...] returns [type2 name] : pravaStrana;}
    \begin{itemize}
      \item Pak lze v akcích odkazovat na \code{\$param1} typu \code{type1}
      \item Akce musí nastavovat \code{\$name} typu \code{type2}
    \end{itemize}
  \end{itemize}

  \bigskip

  \begin{lstlisting}[language=XML, columns=fullflexible]
translationUnit
        : ex=expr[5] EOF { printf("I'm done \%d\n", $expr.ret); }
;

expr[int par] returns [int ret]
  : n1=NUMBER PLUS n2=NUMBER {
    $ret = $par * atoi($n1.text->chars) + atoi($n2.text->chars);
  }
  | SVETE AHOJ { puts("Nepise se to obracene?"); $ret = 0; }
  ;
  \end{lstlisting} %stopzone

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vlastnosti pravidel}
  \begin{enumerate}
    \item Vytvořte si pravidlo pro čísla
    \item Z něj a z \code{expr} si vracejte (smysluplnou) hodnotu
    \begin{itemize}
      \item Tj.\ počítejte
    \end{itemize}

    \item V \code{translationUnit} vypište výsledek
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Inicializace/ukončení}

  \heading{Každé pravidlo muže mít inicializace/ukončení}
  \begin{itemize}
    \item Za jménem pravidla a parametry
    \item Před středníkem
    \item Inicializace: \code{@init\{ ... \}}
    \begin{itemize}
      \item Např.\ inicializovat proměnné
    \end{itemize}
    \item Ukončení: \code{@after\{ ... \}}
    \begin{itemize}
      \item Např.\ počítání
    \end{itemize}
  \end{itemize}

  \bigskip

  \begin{lstlisting}[language=XML, columns=fullflexible]
expr[int par] returns [int ret]
@init{ $ret = 0; }
@after{ $par--; }
  : n1=NUMBER PLUS n2=NUMBER {
    $ret = $par * atoi($n1.text->chars) + atoi($n2.text->chars);
  }
  ;
  \end{lstlisting} %stopzone

\end{frame}
