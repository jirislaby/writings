\section{\cmd{gdb}}
\frame{\sectionpage}

\begin{frame}
  \frametitle{Debugger \cmd{gdb}}

  \begin{itemize}
    \item Pracuje s ELFem a DWARFem

    \item Používá systémové volání \code{ptrace}
    \begin{itemize}
      \item Sleduje proces
      \item Upravuje vykonávání
      \item Čte a mění paměť
      \item \dots
    \end{itemize}

    \pause

    \item Podpora Python skriptů
    \item Spuštění: \cmd{gdb --volby_gdb --args binarka --volby_binarky}
    \item Nápověda: \cmd{help}
    \item Manuál: \href{https://sourceware.org/gdb/current/onlinedocs/gdb/}{GDB
      User Manual}
  \end{itemize}

\end{frame}

\begin{frame}{Příkazy \cmd{gdb}}{Operace s během}

  \begin{itemize}
    \item Spustit: \cmd{run}
    \item Spustit po \code{main}: \cmd{start}
    \item Přerušit: signálem (např.\ Ctrl-c)
    \item Pokračovat po návrat z funkce: \file{finish}
    \item Pokračovat: \file{continue}

    \pause

    \item Krokovat s vnořením: \cmd{step} (po řádcích), \cmd{stepi} (po
      instrukcích)
    \item Krokovat bez vnoření: \cmd{next} (po řádcích), \cmd{nexti} (po
      instrukcích)

    \item Konec: \cmd{quit}

    \medskip
    \pause

    \item Opakování předchozího příkazu: ENTER
  \end{itemize}

  \pause
  \bigskip

  \heading{Úkol:} spusťte si \cmd{yes} v \cmd{gdb} a přerušte. Vyzkoušejte si
  \file{finish} a \file{continue}. Zkuste krokovat po instrukcích hlavní cyklus
  \cmd{yes}. Poté \cmd{quit}.

\end{frame}

\begin{frame}{Příkazy \cmd{gdb}}{Výpisy}

  \begin{itemize}
    \item Zásobník volání: \cmd{where}
    \begin{itemize}
      \item Pohyb po rámcích zásobníku: \cmd{up}, \cmd{down}
    \end{itemize}

    \pause

    \item Programu: \cmd{list} (jen s ladicími informacemi)
    \item Assembleru: \cmd{disassemble}
    \item Paměti: \cmd{print}
    \item HexDump: \cmd{x}
    \item Ostatní: \cmd{info registers}, \file{info break}, \dots
  \end{itemize}

  \bigskip

  Většina příkazů akceptuje parametry -- výpis lokací, proměnných, registrů atd.

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Výpisy v \cmd{gdb}}
  \begin{enumerate}
    \item Přeložte si \file{pb173-bin/09/}
    \item Spusťte \file{debug} (mělo by dojít k pádu)
    \item Otevřete \file{debug} v \cmd{gdb}
    \item Spusťte (\cmd{run})

      \item Proměnnou \code{b} (\cmd{print b})
      \item Proměnnou \code{argc} v \code{main} (\cmd{up} a \cmd{print argc})
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}{Příkazy \cmd{gdb}}{Breakpointy}

  \begin{itemize}

    \begin{itemize}

\end{frame}

\subsection{Obrazy paměti}

\begin{frame}
  \frametitle{Obrazy paměti a \cmd{gdb}}

  \begin{itemize}
    \item Jádro při pádu zapíše obraz paměti (,,coredump'')
    \begin{itemize}
      \item Nutno zvýšit limit velikosti (bývá 0): \cmd{ulimit -c}
      \item Cíl, kam se ukládají: \cmd{sysctl kernel.core_pattern}
    \end{itemize}

    \pause

    \item Potom se dá analyzovat v \cmd{gdb}
    \begin{itemize}
      \item \cmd{gdb --core=core_soubor --args binarka --volby_binarky}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s obrazem paměti}
  \begin{enumerate}
    \item Nastavte limit na obrazy na \code{unlimited} (\cmd{ulimit -c})
    \item Podívejte se, kam se obraz uloží (\cmd{sysctl kernel.core_pattern})
    \item Spusťte program z předchozího příkladu
    \item Otevřete v \cmd{gdb} s obrazem paměti (\cmd{gdb --core=<core>
      ./debug})
    \item Vypište si podobné informace jako předtím
    \begin{itemize}
      \item \cmd{where}
      \item \cmd{print b}
      \item A podobně
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{\cmd{gcc} a \cmd{gdb}}

  \begin{itemize}
    \item \cmd{gcc} generuje různé úrovně ladicích informací
    \begin{itemize}
      \item \texttt{-g1} (méně informací)
      \item \texttt{-g2} alias \texttt{-g}
      \item \texttt{-g3} (navíc informace o makrech)
      \item \texttt{-ggdb} (i v kombinaci s \texttt{-g<cislo>})
      \begin{itemize}
	\item \cmd{gcc} použije i \cmd{gdb} rozšíření, pokud možno
      \end{itemize}
    \end{itemize}

    \pause
    \medskip

    \item S \emph{optimalizacemi} se ladí se obtížněji
    \begin{itemize}
      \item Proměnné zmizely, konstanty se propagovaly, \dots

      \item Ale bez optimalizací je pomalý kód
      \begin{itemize}
	\item Někdy i nepřeložitelný
      \end{itemize}

      \item V \cmd{gcc} 4.8 se objevilo \texttt{-Og}
      \begin{itemize}
	\item Zapíná jen optimalizace, které neztěžují ladění
	\item Nezapíná samotnou generaci ladicích informací!
	\item Používá se v kombinaci s \texttt{-g...}
      \end{itemize}
    \end{itemize}
  \end{itemize}

  \medskip
  \pause

  \heading{Demo:} makra, \cmd{gdb} a \file{pb173-bin/09/mac.c}

\end{frame}

\begin{frame}
  \frametitle{Výpisy}

  \begin{itemize}
    \item Dump (\cmd{x}) a print (\cmd{p})
    \item Lze použít s formátováním (\cmd{x/f} a \cmd{p/f}, kde \cmd{f} je
      formát)

    \pause

    \item Formát jsou tři hodnoty
    \begin{enumerate}
      \item Počet vypisovaných hodnot
      \item Formát vypisovaných hodnot
      \begin{itemize}
	\item \cmd{x}: hexa (podobně jako \code{printf})
	\item \cmd{d}: decimálně
	\item \cmd{u}: decimálně bez znaménka
	\item \cmd{s}: řetězec
	\item \cmd{a}: adresa
	\item \cmd{i}: instrukce
	\item Ostatní: viz \cmd{help x}
      \end{itemize}

      \item Velikost skupiny
      \begin{itemize}
	\item \cmd{b}: 1 bajt
	\item \cmd{h}: 2 bajty
	\item \cmd{w}: 4 bajty
	\item \cmd{g}: 8 bajtů
      \end{itemize}
    \end{enumerate}
  \end{itemize}

\end{frame}

\subsection{Profilování}

\begin{frame}
  \frametitle{Profilování}

  \heading{Co zabírá nejvíce času při vykonávání?}
  \begin{enumerate}
    \item \cmd{gprof}
    \begin{itemize}
      \item Při překladu: \cmd{gcc -pg ...}
      \item Potom se program spustí a data se uloží na disk
      \item \cmd{gprof} data načte a zobrazí
      \item Demo: dump
    \end{itemize}

    \item \cmd{perf}
    \begin{itemize}
      \item Používá čítače v procesoru (nutná podpora CPU)
      \item Zahrnuje chování jádra
      \item Umí pracovat s \cmd{gcc -pg} binárkami
      \item Spuštění: \cmd{perf record -g -- binarka --volby}
      \item Dokumentace: \url{http://perf.wiki.kernel.org}
    \end{itemize}

  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Profilování}
  \begin{enumerate}
    \item Projděte si \file{pb173-bin/09/prof.c}
    \item Spusťte \file{prof_pg}
    \item Prostudujte výstup \cmd{gprof -b prof_pg}

    \bigskip

    \item Nahrajte si výstup \cmd{perf}
    \begin{itemize}
      \item \cmd{perf record -g prof}
    \end{itemize}
    \item Zobrazte si výstup
    \begin{itemize}
      \item \cmd{perf report -g}
    \end{itemize}
    \item Zopakujte \cmd{perf} pro \file{prof_pg}
  \end{enumerate}

\end{frame}
