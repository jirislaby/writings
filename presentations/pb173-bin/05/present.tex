\begin{frame}
  \frametitle{DWARF formát}

  \begin{itemize}
    \item Ladicí informace v lese skoro-stromů
    \begin{itemize}
      \item Pro každý zdrojový soubor jeden skoro-strom
    \end{itemize}

    \item Uzly stromů jsou \emph{Debugging Information Entry} (DIE)
    \item Jeden DIE může odkazovat na kterýkoliv jiný

  \end{itemize}

\begin{frame}
  \frametitle{DWARF a ELF}

  \begin{enumerate}
    \item DWARF je v ELFu uložený v \code{.debug_*} sekcích
    \begin{itemize}
      \item \code{.debug_info}: všechny DIE z lesa stromů
      \item \code{.debug_abbrev}: formát \code{.debug_info} (Figure 48 v DWARF)
      \item \code{.debug_types}: popis datových typů
      \item \code{.debug_str}: řetězce odkazované z \code{.debug_info}
      \item \code{.debug_line}: překlad instrukce $\leftrightarrow$ řádek kódu
    \end{itemize}

    \pause

    \item Výpis DWARFu v ELFu
    \begin{itemize}
      \item \cmd{readelf --debug-dump}
      \item \cmd{objdump --dwarf}
      \item \cmd{dwarfdump} (z \file{libdwarf})
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]
  \frametitle{\file{libdw} a \file{libdwfl} -- soubory}

  \heading{Načtení ELFu s DWARFem uvnitř}
  \begin{itemize}
    \item \code{Dwfl_Module *dwfl_report_offline(Dwfl *dwfl,
      const char *name, const char *file_name, int fd)}
    \begin{itemize}
      \item \code{dwfl}: návratová hodnota z \code{dwfl_begin}
      \item \code{name}: cokoliv (např.\ \lstinline@""@)
      \item \code{file_name}: jméno souboru
      \item \code{fd}: \lstinline|-1| (nebo souborový deskriptor, pak
	\code{file_name == ""})
    \end{itemize}

  \end{itemize}

  \pause
  \medskip

  \heading{Průchod stromů (jednotlivých Compilation Unit)}
  \begin{itemize}
    \item \code{Dwarf_Die *dwfl_nextcu(Dwfl *dwfl, Dwarf_Die *lastcu,
      Dwarf_Addr *bias)}
    \begin{itemize}
      \item \code{dwfl}: návratová hodnota z \code{dwfl_begin}
      \item \code{lastcu}: poslední CU
      \item \code{bias}: přičítá se k adresám, které vrací \file{libdw}
    \end{itemize}

  \end{itemize}

    \begin{lstlisting}
mod = dwfl_report_offline(dwfl, "", argv[1], -1);
while ((die = dwfl_nextcu(dwfl, die, &bias))) { ... }
    \end{lstlisting}

\begin{frame}[fragile]
  \frametitle{DIE}

  \heading{Debugging Information Entry}
  \begin{itemize}
    \item Typ/tag \code{DW_TAG_*}
    \item Z DIE: \code{int dwarf_tag (Dwarf_Die *die)}
    \begin{itemize}
      \item \code{DW_TAG_compile_unit}: o souboru
      \item \code{DW_TAG_subprogram}: o funkci
      \item \code{DW_TAG_base_type}: o typu
      \item \dots
    \end{itemize}

    \pause

    \item Seznam dvojic
    \begin{itemize}
      \item Typ hodnoty: \code{DW_AT_*}
      \begin{itemize}
	\item \code{DW_AT_name}: jméno (proměnné, souboru, \dots)
	\item \code{DW_AT_decl_file}: soubor s deklarací
	\item \code{DW_AT_decl_line}: řádek s deklarací
	\item \code{DW_AT_type}: typ proměnné
	\item \dots
      \end{itemize}
      \item Hodnota
    \end{itemize}
  \end{itemize}

\begin{frame}[fragile]
  \frametitle{Atributy DIE}

  \heading{Iterace přes všechny atributy v jednom DIE}
  \begin{itemize}
    \item \code{ptrdiff_t dwarf_getattrs(Dwarf_Die *die, int
      (*callback)(Dwarf_Attribute *, void *), void *arg, ptrdiff_t offset)}
    \begin{itemize}
      \item \code{die}: z \code{dwfl_nextcu}
      \item \code{callback}: vaše funkce, která se zavolá pro každý atribut v
	DIE
      \item \code{arg}: něco, co se předá do \code{callback} jako 2.\ parametr
      \item \code{offset}: \code{0}
    \end{itemize}

  \end{itemize}

  \pause
  \medskip

  \heading{Operace nad atributem}
  \begin{itemize}
    \item Typ: \code{dwarf_whatattr}
    \begin{itemize}
      \item Vrací \code{DW_AT_*} (\code{DW_AT_name} apod.)
    \end{itemize}

    \item Typ obsahu: \code{dwarf_whatform}
    \begin{itemize}
      \item Vrací \code{DW_FORM_*} (\code{DW_FORM_string} apod.)
    \end{itemize}

    \item Obsah: \code{dwarf_form*}
    \begin{itemize}
      \item Např.\ \code{dwarf_formstring}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Výpis atributů}
  \begin{enumerate}
    \item Definujte funkci (háček) pro \code{dwarf_getattrs}
    \item V ní vypište informace
    \begin{itemize}
      \item Typ atributu (\code{dwarf_whatattr(attr)})
      \item Typ obsahu atributu (\code{dwarf_whatform(attr)})
      \item Adresu, pokud se jedná o typ \code{DW_AT_low_pc}
      \item Řetězec a typ atributu \code{DW_AT_*}, pokud se jedná o typ hodnoty
      \code{DW_FORM_string} nebo \code{DW_FORM_strp}
    \end{itemize}
    \item V těle cyklu \code{dwfl_nextcu} zavolejte \code{dwarf_getattrs}
    \item Přeložte a spusťte
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Průchod skoro-stromem}

  \heading{Iterace přes všechny atributy v jednom DIE}
  \begin{itemize}
    \item \code{int dwarf_child(Dwarf_Die *die, Dwarf_Die *result)}
    \begin{itemize}
      \item Do \code{result} vloží potomka \code{die}
      \item Vrací 0 (OK), 1 (žádný potomek), -1 (chyba)
      \item \code{result} musí být ukazatel na lokální proměnnou
    \end{itemize}

    \item \code{int dwarf_siblingof(Dwarf_Die *die, Dwarf_Die *result)}
    \begin{itemize}
      \item Do \code{result} vloží sourozence \code{die}
      \item Vrací 0 (OK), 1 (žádný sourozenec), -1 (chyba)
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Výpis potomků CU}
  \begin{enumerate}
    \item Vezměte 1. potomka od CU (\code{dwarf_child})
    \item Iterujte přes sourozence potomka (\code{dwarf_siblingof})
    \item Pro všechny potomky vypište typ DIE (\code{dwarf_tag})
    \begin{itemize}
      \item Nejprve zjistěte, které tam jsou
      \item Potom vypisujte v textové podobě
    \end{itemize}

    \item Rozšiřte na celý podstrom, nejen  první úroveň
    \item Přeložte a spusťte
  \end{enumerate}

\end{frame}
