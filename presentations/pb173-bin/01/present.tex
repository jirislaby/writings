\begin{frame}
  \frametitle{GIT a commity}

  \begin{itemize}
    \item Každý commit je pojmenován SHA hashem
    \begin{itemize}
      \item Např.\ \code{d3323c1503d54d83b0eae6c7927dede2d2973059}, lze i
	zkráceně \code{d3323c1503}
    \end{itemize}

    \item \cmd{HEAD} je alias pro horní/poslední commit

    \item Lze odkazovat předchůdce pomocí \textasciitilde\ nebo \textasciicircum
    \begin{itemize}
      \item Např.\ \code{HEAD}\textasciicircum\textasciicircum\textasciicircum,
	\code{HEAD~1}, \code{d3323c1503~5}
    \end{itemize}

    \item Lze vytvořit jmenný alias, tzv.\ tag

    \item Více: \doc{man gitrevisions}

  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{\textsc{gcc}}

  \begin{itemize}
    \item Pro překlad C, assembleru a hlavičkových souborů

    \pause

    \item Důležité volby
    \begin{itemize}
      \item \texttt{-E} -- spusť jen preprocesor a skonči
      \item \texttt{-S} -- generuj assembleru
      \item \texttt{-c} -- generuj objektový soubor
      \item \texttt{-x} -- definujeme jazyk souboru, když nelze určit dle
	přípony (\texttt{-x c soubor\_v\_C -x assembler soubor\_v\_asm})
      \item \texttt{-O} -- optimalizace (\texttt{-O0}, \texttt{-O2}, \dots)
      \item \texttt{-W} -- varování (\texttt{-Wall}, \dots)
    \end{itemize}

    \pause

    \item Dokumentace
    \begin{itemize}
      \item Offline: \textsc{info}/\textsc{pinfo}
      \item Na webu: \href{https://gcc.gnu.org/onlinedocs/}{\textit{Using the
	GNU Compiler Collection}}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Volání \textsc{gcc}}
  \begin{enumerate}
    \item Pomocí roury přeložte \code{main}, který něco výpíše, do objektu
    \begin{itemize}
      \item \cmd{echo -e '}\code{\#include <stdio.h>\\n int main() \{ puts("Hello"); return 0; \}}\cmd{' | ...}
    \end{itemize}

    \item Spusťte všechny tři fáze (\texttt{-E}, \texttt{-S} a \texttt{-c})
      postupně jako samostatné příkazy propojené rourami
    \begin{itemize}
      \item \cmd{echo ... | gcc -E ...}
      \item \cmd{echo ... | gcc -E ... | gcc -S ... }
      \item \cmd{echo ... | gcc -E ... | gcc -S ... | gcc -c ...}
    \end{itemize}

    \item Pozorujte \texttt{-E} a \texttt{-S} výstupy
    \item Zapněte optimalizace \texttt{-O2} a pozorujte oba výstupy znovu
    \begin{itemize}
      \item Mění optimalizace oba výstupy?
    \end{itemize}
  \end{enumerate}
\end{frame}

\begin{frame}
  \frametitle{\textsc{objdump}/\textsc{objcopy}}

  \begin{itemize}
    \item Slouží k práci s objekty

    \pause

    \item \textsc{objdump} vypisuje informace
    \begin{itemize}
      \item \texttt{-D} disassembler vstupu
      \item \texttt{-h} výpis sekcí vstupu
      \item \texttt{-b} formát vstupu
      \item \texttt{-m} stroj vstupu
    \end{itemize}

    \pause

    \item \textsc{objcopy} transformuje
    \begin{itemize}
      \item \texttt{-I} formát vstupu
      \item \texttt{-O} formát výstupu
    \end{itemize}

    \pause

    \item Společný důležitý argument
    \begin{itemize}
      \item \texttt{-j} specifikace sekce
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}

  \heading{Práce s objekty}
  \begin{enumerate}
    \item Vypište si informace o svém objektu z předchozího příkladu pomocí
      \cmd{objdump}
    \begin{itemize}
      \item Porovnejte výstupy \cmd{objdump -D} a \cmd{gcc -S -fverbose-asm}
      \item Použijte \texttt{-h}
    \end{itemize}

    \pause

    \item Extrahujte \code{.text} sekci pomocí
      \cmd{objcopy}
    \begin{itemize}
%      \item Z předchozího výpisu \cmd{objdump -h} zjistěte, která má nenulovou
%	velikost
      \item Vložte ji do nového souboru v binární podobě
      \item Tj.\ použijete \texttt{-j\ .sekce} a \texttt{-O binary}
    \end{itemize}

    \pause

    \item Disassemblujte tento binární výstup pomocí \cmd{objdump}
    \begin{itemize}
      \item Musíte specifikovat formát (\texttt{-b binary}) a stroj
	(\texttt{-m i386:x86-64})
    \end{itemize}

  \end{enumerate}
