\begin{frame}
  \frametitle{\textsc{gcc}}

  \begin{itemize}
  \item Pro překlad C i assembleru
  \item C budeme psát ve stylu jádra (\emph{CodingStyle})
  \begin{itemize}
  \item Kontrola: \cmd{scripts/checkpatch.pl} (není 100\%)
  \end{itemize}
  \item Důležité volby
  \begin{itemize}
  \item \texttt{-E} -- jen preprocesor
  \item \texttt{-S} -- generování assembleru
  \item \texttt{-c} -- generování objektových souborů
  \item \texttt{-x} -- definuje jazyk souboru (\texttt{-x c soubor1 -x assembler soubor2})
  \item \texttt{-O} -- optimalizace (\texttt{-O0}, \texttt{-O2}, \dots)
  \end{itemize}
  \item \textsc{info}/\textsc{pinfo}
  \item Dokumentace: \textit{Using the GNU Compiler Collection}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Volání \textsc{gcc}}
  \begin{enumerate}
  \item Pomocí roury přeložte \code{main}, který něco výpíše, do objektu
  \item \cmd{echo -e '}\code{\#include <stdio.h>\\n int main() \{ puts("Hello"); return 0; \}}\cmd{' | ...}
  \item Spusttě všechny tři fáze (\texttt{-E}, \texttt{-S} a \texttt{-c})
    jako samostatné příkazy propojené rourami
  \item Pozorujte \texttt{-E} a \texttt{-S} výstupy
  \item Zapněte optimalizace \texttt{-O2} a pozorujte znovu
  \end{enumerate}
\end{frame}

\begin{frame}
  \frametitle{\textsc{objdump}/\textsc{objcopy}}

  \begin{itemize}
  \item Slouží k práci s objekty
  \item \textsc{objdump} vypisuje informace
  \begin{itemize}
  \item \texttt{-D} disassembler
  \item \texttt{-h} výpis sekcí souboru
  \item \texttt{-b} formát vstupu
  \item \texttt{-m} stroj vstupu
  \end{itemize}

  \item \textsc{objcopy} transformuje
  \begin{itemize}
  \item \texttt{-I} formát vstupu
  \item \texttt{-O} formát výstupu
  \end{itemize}

  \item Společný důležitý argument
  \begin{itemize}
  \item \texttt{-j} specifikace sekce
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s objekty}
  \begin{enumerate}
  \item Vypište si informace o vašem objektu z předchozího příkladu pomocí
    \cmd{objdump}
  \begin{itemize}
  \item Za použití \texttt{-h}
  \end{itemize}

  \item Extrahujte \code{.text} sekci pomocí \cmd{objcopy}
  \begin{itemize}
  \item Z předchozího výpisu zjistěte, která má nenulovou velikost
  \item Vložte ji do souboru v binární podobě
  \item Tj. použijete \texttt{-j .sekce} a \texttt{-O binary}
  \end{itemize}

  \item Disassemblujte tento binární výstup pomocí \cmd{objdump}
  \begin{itemize}
  \item Musíte specifikovat formát (\texttt{-b binary}) a stroj
    (\texttt{-m i386:x86-64})
  \end{itemize}

  \end{enumerate}
\end{frame}
