\begin{frame}{Základní informace o procesu}

  \begin{itemize}
    \item PID: \code{pid_t getpid(void)} (\header{unistd.h})
    \item PPID: \code{pid_t getppid(void)} (\header{unistd.h})

    \pause

    \item ID: \code{getuid}, \code{getgid}, \code{geteuid}, \code{getegid},
      \dots \ifpcdirlinux (opakování) \fi

    \pause

    \item Časy běhu: \code{clock_t times(struct tms *buf)} (\header{sys/times.h})
    \item Pracovní adresář: \code{char *getcwd(char *buf, size_t size)}
      (\header{unistd.h})

    \item \dots
  \end{itemize}

\end{frame}

\begin{frame}[fragile]{Vytvoření a zrušení procesu}

  \begin{itemize}
    \item \code{pid_t fork(void)} (\header{unistd.h})
    \begin{itemize}
      \item Vznikne nový proces (potomek)
      \item Návratová hodnota
      \begin{itemize}
        \item \code{\-1} -- chyba
        \item \code{0} -- \code{fork} se vrátil do potomka
        \item \code{> 0} -- rodič a \code{fork} vrací PID potomka
      \end{itemize}

      \pause

      \item Často následuje \code{exec} -- spuštění nové binárky
    \end{itemize}

    \pause

    \item \code{void exit(int status)} (\header{stdlib.h})
    \begin{itemize}
      \item \code{status}: dostane rodič (v shellu \cmd{\$?})
      \item Zánik procesu
      \item Nikdy se nevrátí
    \end{itemize}
  \end{itemize}

  \pause

  \begin{block}{Příklad}
    \begin{lstlisting}[aboveskip=2pt,belowskip=2pt]
if (fork())
  exit(0); // parent
else
  exit(1); // child
    \end{lstlisting}
  \end{block}

\end{frame}

\begin{frame}{Úkol}

  \heading{Vytváření procesů}
  \begin{enumerate}
    \item Vytvořte si 10 potomků
    \item V každém potomkovi si vypište jeho PID a PID rodiče
    \item Potomka ukončete

    \item V rodiči čekejte na vstup (\code{getchar})
    \begin{itemize}
      \item Jaký stav mají po tuto dobu potomci?
    \end{itemize}
    \item Vyzkoušejte
  \end{enumerate}

\end{frame}

\begin{frame}{Čekání na potomky}

  \begin{itemize}
    \item Správný rodič by měl na své potomky počkat
    \begin{itemize}
      \item Než skončí
      \item A pochovat je
      \item Jinak vznikají zombie
    \end{itemize}

    \pause

    \item { \footnotesize \code{pid_t waitpid(pid_t pid, int *status,
      int options)} (\header{sys/wait.h}) }
    \begin{itemize}
      \item \code{pid}: na koho čekat, \code{\-1} = na kohokoliv

      \pause

      \item \code{status}: co vrátil \code{main} nebo parametr \code{exit}
      \begin{itemize}
	\item Kromě návratové hodnoty i stav
	\item Makra \code{WIFEXITED}, \code{WEXITSTATUS}, \code{WIFSIGNALED},
	  \dots
      \end{itemize}

      \pause

      \item \code{options}: blokující čekání apod.
      \item Návratová hodnota: PID potomka

      \pause

      \item \code{wait(status)} je \code{waitpid(-1, status, 0)}
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Čekání na potomky}
  \begin{enumerate}
    \item Odstraňte spaní v předchozím příkladě
    \item Přidejte spaní 1 vteřinu v \emph{potomkovi} po výpise
    \item V rodiči čekejte na potomky
    \item Vypište jejich PID

    \item Zjistěte, jak byly ukončeny
    \begin{itemize}
      \item \code{WIFEXITED}, \code{WIFSIGNALED}
    \end{itemize}

    \item Tuto informaci vypište
    \item Vyzkoušejte
  \end{enumerate}

\end{frame}
