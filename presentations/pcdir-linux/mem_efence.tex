\begin{frame}[fragile]{Ladění paměti}

  \begin{block}{Špatná správa paměti}
  \begin{lstlisting}
int main(void) {
  char *mem1 = malloc(100);
  char *mem2 = malloc(100);

  free(mem2);
  mem2[3] = 10;

  return mem1[5];
}
  \end{lstlisting}
  \end{block}

  \pause
  \medskip

  \heading{Pomocníci}
  \begin{enumerate}
    \item \textsc{ElectricFence}
    \item \textsc{Valgrind}
  \end{enumerate}

\end{frame}

\begin{frame}{\textsc{ElectricFence}}

  \begin{itemize}
    \item Zachytává přístupy paměti
    \begin{itemize}
      \item Mimo alokovanou oblast
      \item Po provedeném \code{free} na paměti
    \end{itemize}

    \pause

    \item Knihovna, která se musí přilinkovat
    \begin{itemize}
      \item Buď \cmd{\-lefence} při linkování
      \item Anebo \cmd{export LD_PRELOAD=libefence.so.0.0}

      \pause

      \item Nahradí \code{malloc}, \code{free}, \dots svými implementacemi
    \end{itemize}

    \pause

    \item Odmapuje stránky
    \begin{itemize}
      \item Program začne padat
      \item Na selhání stránkování
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{\textsc{ElectricFence}}
  \begin{enumerate}
    \item Otevřete si \ifpcdirlinux \file{10/memcheck.pro} \else
      \file{06/memcheck.c} \fi
    \item Přeložte a spustťe
    \begin{itemize}
      \item Nemělo by dojít k pádu
    \end{itemize}

    \item Nyní zapněte \textsc{ElectricFence}
    \begin{itemize}
      \item Nastavte \cmd{LD_PRELOAD=libefence.so.0.0}
\ifpcdirlinux
      \item Nebo \code{QMAKE_LIBS=-lefence} v \file{.pro}
\else
      \item Nebo \code{LDLIBS=-lefence} v \file{Makefile}
\fi
    \end{itemize}

    \item Spusťte
    \item Opravte 2 chyby, na kterých \textsc{ElectricFence} způsobuje pád
    \item Vyzkoušejte
    \item Vraťte kód do původní podoby
    \begin{itemize}
      \item Pro \textsc{Valgrind}
      \item Např.\ \cmd{git checkout HEAD memcheck.c}
    \end{itemize}
  \end{enumerate}

\end{frame}
