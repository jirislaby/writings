\begin{frame}[fragile]{Signály}

  \begin{itemize}
    \item Signály předávané mezi procesy (\doc{man 7 signal}, \header{signal.h})
    \begin{itemize}
      \item Čísla: \code{SIGHUP}, \code{SIGINT}, \code{SIGKILL},
	\code{SIGTERM}, \dots

      \pause

      \item Některé lze ignorovat, popř.\ jim nastavit akci
    \end{itemize}

    \pause

    \item Rozesílání: \code{int kill(pid_t pid, int sig)}
    \begin{itemize}
      \item \code{pid}: koho zabít, \code{\-1} znamená všechny
      \item \code{sig}: číslo signálu
    \end{itemize}

    \pause

    \item Obsluha: \code{sighandler_t signal(int signum, sighandler_t handler)}
    \begin{itemize}
      \item \code{typedef void (*sighandler_t)(int)}
      \item \pozor: některé funkce nejsou dostupné v obsluze!
    \end{itemize}
  \end{itemize}

  \begin{block}{Příklad (bez ošetření návratových hodnot)}
  \begin{lstlisting}
static void my_sig(int sig) { printf("Signal %d raised\n", sig); }
int main(void) {
  signal(SIGINT, my_sig);
  raise(SIGINT); /* alias pro kill(getpid(), SIGINT); */
}
  \end{lstlisting}
  \end{block}

\end{frame}
