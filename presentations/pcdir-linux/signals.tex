\begin{frame}{Signály}{Účel a použití}

  \begin{itemize}
    \item Podobné přerušením procesoru
    \begin{enumerate}
      \item Jádro doručí signál procesu
      \item Proces přestává vykonávat kód
      \item Obsluží signál
      \item Pokračuje, kde skončil
    \end{enumerate}

    \pause

    \item Signál lze
    \begin{itemize}
      \item Ignorovat (až na výjimky)

      \pause

      \item Dočasně blokovat

      \pause

      \item Poslat (sobě, ostatním procesům)

      \pause

      \item Na něj čekat
    \end{itemize}

    \pause

    \item Signály jsou čísla
    \begin{itemize}
      \item \code{SIGHUP} (1), \code{SIGINT} (2), \code{SIGKILL} (9),
        \code{SIGTERM} (15), \dots
    \end{itemize}

    \pause

    \item \doc{man 7 signal}, \header{signal.h}
  \end{itemize}

\end{frame}

\ifpcdirlinuxb

\begin{frame}[fragile]{Masky signálů}

  \begin{itemize}
    \item Některé signálové funkce pracují s maskami
    \begin{itemize}
      \item Typ \code{sigset_t}
    \end{itemize}

    \pause

    \item Inicializace
    \begin{itemize}
      \item Na 0: \code{int sigemptyset(sigset_t *set)}
      \item Na 1: \code{int sigfillset(sigset_t *set)}
    \end{itemize}

    \pause

    \item Operace
    \begin{itemize}
      \item Přidat: \code{int sigaddset(sigset_t *set, int sig)}
      \item Odebrat: \code{int sigdelset(sigset_t *set, int sig)}
    \end{itemize}

    \pause

    \item Je nastaven?
    \begin{itemize}
      \item \code{int sigismember(const sigset_t *set, int sig)}
    \end{itemize}
  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}
sigset_t mask;
sigfillset(&mask);
sigdelset(&mask, SIGINT);
  \end{lstlisting}
  \end{block}

\end{frame}

\fi

\begin{frame}[fragile]{Rozesílání}

  \begin{itemize}
    \item \code{int kill(pid_t pid, int sig)}
    \begin{itemize}
      \item \code{pid}: koho zabít
      \ifpcdirlinuxb
      \begin{itemize}
	\item \code{0} -- všechny ve svojí skupině
	\item \code{\-1} -- všechny
	\item \code{\-pid} -- skupinu \code{pid}
      \end{itemize}
      \fi

      \pause

      \item \code{sig}: číslo signálu
    \end{itemize}

    \pause

    \item \code{int raise(int sig)}
    \begin{itemize}
      \item Alias pro \code{kill(getpid(), sig)}
    \end{itemize}

  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}
int main(void) {
  raise(SIGINT);
}
  \end{lstlisting}
  \end{block}

\end{frame}

\ifpcdirlinuxb

\begin{frame}[fragile]{Čekání na signál}

  \begin{itemize}
    \item Lze čekat na signál a pokračovat až po jeho obsluze

    \pause

    \item \code{int pause(void)}

    \pause

    \item \code{int sigsuspend(const sigset_t *mask)}
    \begin{itemize}
      \item Blokuje signály specifikované v \code{mask}
      \item Čeká na ostatní
    \end{itemize}

  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}
int main(void) {
  sigset_t s;
  sigfillset(&s);
  sigdelset(&s, SIGTERM);
  sigsuspend(&s);
  return 0;
}
  \end{lstlisting}
  \end{block}

\end{frame}

\fi

  void (*sa_handler)(int);

  \end{lstlisting}~

  \end{tabular}

\begin{frame}{Úkol}

  \heading{Signály}
  \begin{enumerate}
    \ifpcdirlinux
      \item Upravujte předchozí příklad
    \else
      \item Vytvořte si nový \code{main}
    \fi

    \item Přidejte obsluhu pro signál TERM
    \begin{itemize}
      \item Vypište v ní PID
    \end{itemize}

    \ifpcdirlinuxb {
      \item Vytvořte si opět 10 potomků
      \begin{itemize}
	\item Budou spát 5 vteřin
      \end{itemize}
    } \fi

    \item Prvním 5 potomkům pošlete signál TERM
    \begin{itemize}
      \item Zatímco spí
    \end{itemize}

    \ifpcdirlinux
      \item Důvod ukončení (výpis) by se pro těch 5 měl změnit
    \else
      \item V rodiči si na potomky počkejte (\code{wait})
      \item Vypište, zda byl potomek zabit (\code{WIFSIGNALED(status)})
    \fi

    \item Vyzkoušejte
  \end{enumerate}

\end{frame}
