\begin{frame}{Adresace v IP}

  \heading{IPv4}
  \begin{itemize}
    \item 32bitová, udává se jako 4 desítková čísla oddělená tečkou
    \item Za lomítkem může být maska sítě

    \pause

    \item Např.\ \cmd{10.0.0.1/24}
    \begin{itemize}
      \item Adresa \cmd{10.0.0.1}
      \item Patří do sítě \cmd{10.0.0.255}
      \begin{itemize}
	\item Všichni v okolí mají adresu \cmd{10.0.0.X}
	\item Nemůže přímo komunikovat s nikým mimo svoji síť
      \end{itemize}
    \end{itemize}
  \end{itemize}

  \heading{IPv6}
  \begin{itemize}
    \item 128bitová, udává se jako dvojice šestnáctkových čísel
      oddělených dvojtečkou
    \begin{itemize}
      \item Nuly lze vynechat
    \end{itemize}

    \pause

    \item Např.\ \cmd{2001:718:801:230::1/64}
  \end{itemize}

  \pause

  \begin{center}
  \heading{Kdo si má adresu pamatovat?} \\
  \pause
  \heading{DNS}
  \end{center}

\end{frame}

\begin{frame}{Protokoly nad IP}

  \heading{Nad IP nejčastěji}
  \begin{itemize}
    \item UDP -- nespojované
    \begin{itemize}
      \item Málo garancí -- nespolehlivá
      \item Bez latencí

      \pause

      \item Výměna jednotlivých datagramů

      \pause

      \item Vhodná na zvuk, video, hry, kde výpadky nevadí
    \end{itemize}

    \pause

    \item TCP -- spojované
    \begin{itemize}
      \item Garantuje pořadí paketů, ověřuje doručení -- spolehlivá
      \item Ale vyšší latence

      \pause

      \item Výměna proudu bajtů

      \pause

      \item Vhodná na data
    \end{itemize}

    \pause

    \item Jedna IP může obsluhovat více UDP i TCP služeb
    \begin{itemize}
      \item Obě podporují tzv.\ porty
      \item 16bitové číslo -- 64k služeb
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Protokol UDP/IP}

  \heading{UDP komunikace}
  \begin{itemize}
    \item Server poslouchá
    \begin{itemize}
      \item Předem daný UDP port (\code{bind} dále)
      \item Nebo náhodný, volný

      \pause

      \item Provede se \code{read}, \code{recvfrom}, \dots
    \end{itemize}

    \pause

    \item Klient pošle datagram
    \begin{itemize}
      \item Musí vědět/umět zjistit IP adresu i UDP port

      \pause

      \item Provede se \code{write}, \code{sendto}, \dots
    \end{itemize}

    \pause

    \item Samostatný datagram ne/proletí sítí
    \begin{itemize}
      \item Další může ne/projít jinudy, rychleji, před prvním apod.
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{UDP komunikace}
  \begin{enumerate}
    \item Otevřete UDP socket na IPv4
    \begin{itemize}
      \item \code{socket}, \code{AF_INET} a \code{SOCK_DGRAM}
    \end{itemize}

    \item Proveďte do něj \code{write} prázdného řetězce a délky 0
    \begin{itemize}
      \item Selže, ale socketu přidělí UDP port
    \end{itemize}

    \item Udělejte \code{recv} do bufferu
    \item Buffer vypište
    \item Zavřete socket
    \item Přeložte a spusťte

    \item Zjistěte port
    \begin{itemize}
      \item Pomocí \cmd{ss -ulpn}
    \end{itemize}

    \item Pošlete něco na UDP port
    \begin{itemize}
      \item Pomocí \cmd{netcat -u 127.0.0.1 port}
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}{Úkol}

  \heading{Získání adres}
  \begin{enumerate}
    \item Doplňte do klienta z předchozího příkladu za vytvoření socketu
    \item Použijte \code{getaddrinfo}

    \item Zjistěte, kolik je k dispozici adres pro server \cmd{google.com}
    \begin{itemize}
      \item Parametr \code{node}
    \end{itemize}

    \item Omezte se jen na TCP a port HTTP
    \begin{itemize}
      \item \code{hints.ai_socktype} bude \code{SOCK_STREAM}
      \item \code{hints.ai_protocol} může být \code{IPPROTO_TCP}
      \item \code{service} bude \code{\"http\"} nebo \code{\"80\"}
    \end{itemize}

    \item Projděte seznam
    \begin{itemize}
      \item \code{for (struct addrinfo *i = ai; i; i = i->ai_next)}
      \item Pro každou položku vypište \code{ai_family} a \code{ai_socktype}
    \end{itemize}

    \item Vypište počet iterací

    \item Přeložte, vyzkoušejte
    \begin{itemize}
      \item Kolik je adres?
    \end{itemize}
  \end{enumerate}

      \item \code{sockfd}: návratová hodnota funkce \code{socket}
      \item \code{addr}: adresa, kam se připojit
      \item \code{addrlen}: velikost struktury adresy
    \end{itemize}

\ifpcdirlinux
    \pause

    \item Příjem: \code{read} nebo \code{ssize_t recv(int sockfd, void *buf,

    \item Vysílání: \code{write} nebo \code{ssize_t send(int sockfd, const void

  \pause

  \begin{block}{Příklad (bez ošetření chyb)}

\begin{frame}[fragile]{Server}

  \begin{block}{Prototypy}
  \begin{lstlisting}[aboveskip=2pt,belowskip=0pt]
int bind(int sockfd, const struct sockaddr *addr, socklen_t addrlen)
  \end{lstlisting}
  \onslide<2->
  \begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
int listen(int sockfd, int backlog)
  \end{lstlisting}
  \onslide<4->
  \begin{lstlisting}[aboveskip=0pt,belowskip=2pt]
int accept(int sockfd, struct sockaddr *addr, socklen_t *addrlen)
  \end{lstlisting}
  \end{block}

  \onslide<1->

  \heading{Tři kroky}
  \begin{enumerate}
    \item Nastavení adresy a portu, kde poslouchat (\code{bind})
    \begin{itemize}
      \item \code{addr}: IP adresa se nastavuje často na \code{INADDR_ANY}
	(IPv4) nebo \code{in6addr_any} (IPv6), port je
	\code{sin_port}/\code{sin6_port} (ale \code{htons}!)
    \end{itemize}

    \pause

    \item Nastavení socketu jako pasivního (\code{listen})
    \begin{itemize}
      \item \code{backlog}: kolik spojení může čekat na obsluhu

      \pause

      \item Socket bude čekat na spojení
      \item Od tohoto okamžiku bude fungovat \code{connect} na tento port
    \end{itemize}

    \pause

    \item Čekání na příchozí spojení (\code{accept})
    \begin{itemize}
      \item \code{addr}: návratová hodnota -- adresa toho, kdo se připojil
      \item Vrací socket, nad kterým se provádějí operace

      \pause

      \item Volá se opakovaně v cyklu pro obsluhu dalších spojení
    \end{itemize}

  \end{enumerate}

\end{frame}

\begin{frame}{Úkol}

  \heading{Server -- echo}
  \begin{enumerate}
    \item Vytvořte server
    \item Pro každé příchozí spojení vytvořte potomka
    \begin{itemize}
      \item \code{bind}, \code{listen}, opakuj (\code{accept}, \code{fork})
    \end{itemize}

    \item Rodič bude čekat na další spojení
    \item Potomek obslouží spojení
    \item Přečte ze socketu až 100 znaků
    \item Pošle je zpět
    \item Upravte klienta, aby po připojení něco poslal
    \begin{itemize}
      \item A to na váš port z \code{bind}
    \end{itemize}

    \item Potom přečte ze socketu až 100 znaků
    \item Vypíše je
    \item Vyzkoušejte
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]{\code{poll}}

  \begin{block}{Prototyp a příklad}
  \begin{lstlisting}[aboveskip=2pt]
#include <poll.h>
struct pollfd {  // from poll.h
  int   fd;      // file descriptor
  short events;  // requested events: POLLIN, POLLOUT, ...
  short revents; // returned events: POLLIN, POLLOUT, POLLERR, POLLHUP, ...
};
int poll(struct pollfd *fds, nfds_t nfds, int timeout)
  \end{lstlisting}
  \onslide<4->
  \begin{lstlisting}[belowskip=2pt]
struct pollfd pfd[] = { { 0, POLLIN }, { 2, POLLIN }, { fd, POLLIN | POLLOUT } };
if (poll(pfd, 3, -1) > 0)
  puts("some event!");
  \end{lstlisting}
  \end{block}

  \onslide<1->

  \begin{itemize}
    \item \code{fds}: pole popisující, na které \code{fd} čekat (a jaké
      události)
    \item \code{nfds}: velikost pole \code{fds}

    \pause

    \item \code{timeout}: timeout v ms, nebo \code{\-1}

    \pause

    \item \code{fds} je po návratu patřičně \emph{upraveno}
    \begin{itemize}
      \item Nový \code{poll} $\Rightarrow$ nová inicializace pole
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]{\code{select}}

  \begin{block}{Prototypy}
  \begin{lstlisting}[aboveskip=2pt]
#include <sys/select.h>
int select(int nfds, fd_set *rdfds, fd_set *wrfds, fd_set *exceptfds, struct timeval *timeout)
void FD_ZERO(fd_set *set);                void FD_CLR(int fd, fd_set *set);
void FD_SET(int fd, fd_set *set);         int  FD_ISSET(int fd, fd_set *set);
  \end{lstlisting}
  \onslide<5->
  \begin{lstlisting}[belowskip=2pt]
fd_set rd, wr;
FD_ZERO(&rd);        FD_ZERO(&wr);
FD_SET(0, &rd);      FD_SET(2, &rd);      FD_SET(fd, &rd);      FD_SET(fd, &wr);
if (select(fd + 1, &rd, &wr, NULL, NULL) > 0)
  puts("some event!");
  \end{lstlisting}
  \end{block}

  \onslide<1->

  \begin{itemize}
    \item \code{nfds}: nejvyšší číslo \code{fd} v množinách + 1

    \pause

    \item \code{rdfds}: čekat, až je bude možno číst, nebo \code{NULL}
    \item \code{wrfds}: čekat, až do nich bude možno zapisovat, nebo
      \code{NULL}
    \item \code{exceptfds}: čekat, až nastane chyba, nebo \code{NULL}

    \pause

    \item \code{timeout}: timeout v \code{timeval}, nebo \code{NULL}

    \pause

    \item Návratová hodnota: počet bitů 1 v \code{rdfds}, \code{wrfds} a
      \code{exceptfds}
  \end{itemize}
