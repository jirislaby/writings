\begin{frame}{Úkol}

  \heading{Práce s ID}
  \begin{enumerate}
    \item Prostudujte \code{print_uids} v \ifpcdirlinux \file{06/main.c} \else
      \file{02/uids.c} \fi
    \item Přeložte
    \item Spusťte jako uživatel i jako root

    \item Nastavte programu set-uid bit
    \begin{itemize}
      \item Jako root spusťte: \cmd{chmod u+s main}
    \end{itemize}

    \item Spusťte jako uživatel i jako root

    \item Nastavte vlastníka na 0 (root)
    \begin{itemize}
      \item Jako root spusťte: \cmd{chown 0 main}
      \item Musíte teď znovu: \cmd{chmod u+s main}
    \end{itemize}

    \item Spusťte jako uživatel i jako root
    \item Jak se mění výstup?
  \end{enumerate}

\end{frame}

\begin{frame}{ID procesu}{Změna}

  \begin{block}{Funkce pro změnu ID (\header{unistd.h}, \header{sys/types.h},
    \header{sys/fsuid.h})}
  \begin{center}
  \begin{tabular}{|c|c|c|c|c|} \hline
		      & \textbf{Reálné} & \textbf{Efektivní} & \textbf{Uložené}  & \textbf{FS} \\ \hline
    \textbf{Uživatel} & \code{setuid}   & \code{seteuid} & \code{setresuid} & \code{setfsuid} \\ \hline
    \textbf{Skupina}  & \code{setgid}   & \code{setegid} & \code{setresgid} & \code{setfsgid} \\ \hline
  \end{tabular} \\[1ex]
  \end{center}
  \end{block}

  \begin{itemize}
    \item Obyčejný uživatel: \code{setuid} nastavuje \emph{efektivní} ID!
    \begin{itemize}
      \item Jenom root může měnit reálné ID
    \end{itemize}

    \pause

    \item root: \code{setuid} nastavuje všechny tři!
    \begin{itemize}
      \item Nelze \code{setuid(100); setuid(0);}

      \pause

      \item Tedy: nutno použít \code{seteuid}
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Změna ID}
  \begin{enumerate}
    \item Upravujte předchozí program
    \item Po výpisu ID změňte reálné ID na ID uživatele vyvoj (\code{setuid})
    \begin{itemize}
      \item ID uživatele si zjistěte pomocí \cmd{id -u}
    \end{itemize}

    \item Vypište opět ID
    \item Změňte reálné ID na 0 (\code{setuid})
    \item Vypište opět ID
    \item Spusťte jako uživatel i jako root
    \item Nastavte programu set-uid bit a vlastníka na 0 (tj.\ root)
    \item Spusťte jako uživatel
    \item Nyní měňte efektivní ID (změna \code{setuid} na \code{seteuid})
    \item Spusťte jako uživatel i jako root
    \item Nastavte programu set-uid bit a vlastníka na 0 (tj.\ root)
    \item Spusťte jako uživatel

  \end{enumerate}

\end{frame}
