\begin{frame}[fragile]{Makefile}{Pravidla}

\begin{frame}{Terminál}

  \begin{itemize}

      \item Ale stále terminál

    \end{itemize}

    \pause

    \item Jak zjistit, zda je souborový deskriptor terminál?

    \pause

    \begin{itemize}
      \item \code{int isatty(int fd)} (POSIX, \header{unistd.h})
      \begin{itemize}
	\item \code{fd}: souborový deskriptor (např.\ \code{STDIN_FILENO})
	\item Návratová hodnota: \code{1} -- \code{fd} je terminál, jinak
	  \code{0} (\code{errno})
      \end{itemize}
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Rozpoznání terminálu}
  \begin{enumerate}
    \item Zjistěte, zda je váš stdin a stdout terminál
    \item Pokud ne, skončete s chybou
    \item Vyzkoušejte
    \begin{itemize}
      \item Zkuste i chybu přesměrováním
      \item Např.\ \cmd{./main < /dev/null >/dev/null}
    \end{itemize}

  \end{enumerate}

\end{frame}

\begin{frame}[fragile]{Vlastnosti terminálu}

  \begin{itemize}
    \item \code{struct termios} (\header{termios.h}, \doc{man 3 termios})

    \pause

    \item Obsahuje téměř všechny vlastnosti terminálu
    \begin{itemize}
      \item Chybí např.\ velikost okna (viz dále)
    \end{itemize}

    \pause

    \item V shellu: \code{stty} (Demo)

    \pause

    \item \code{int tcgetattr(int fd, struct termios *termios_p)} a \\
      \code{int tcsetattr(int fd, int optional_actions,}\\
      \code{const struct termios *termios_p)}
    \begin{itemize}
      \item \code{fd}: otevřený terminál, např.\ \code{STDOUT_FILENO}
      \item \code{termios_p}: kam přečíst/odkud nastavit
      \item \code{optional_actions}: kdy -- \code{TCSANOW} (teď),
	\code{TCSADRAIN} (až je prázdný výstup), \code{TCSAFLUSH}
	(\code{TCSADRAIN} + zahodí se vstup)
    \end{itemize}

  \end{itemize}

  \begin{block}{\code{struct termios}}
  \begin{lstlisting}
tcflag_t c_iflag;  /* input modes */            cc_t     c_cc[NCCS];  /* special characters */
tcflag_t c_oflag;  /* output modes */
tcflag_t c_cflag;  /* control modes (HW) */
tcflag_t c_lflag;  /* local modes (user) */
  \end{lstlisting}
  \end{block}

\end{frame}

\begin{frame}{Rychlost terminálu}

  \begin{itemize}
    \item Baudy (bit/s)
    \item 50, 75, \dots, 9600, 19200, 38400, 57600, 115200, \dots

    \pause

    \item Je součástí \code{struct termios}

    \item Čtení: \code{speed_t cfgetispeed(const struct termios *termios)} a
      \code{cfgetospeed}
    \begin{itemize}
      \item Zvlášť pro vstup a výstup
      \item \code{termios}: struktura naplněná pomocí \code{tcgetattr}
      \item Návratová hodnota: \emph{konstanta} pro rychlost -- \code{B50},
        \code{B9600}, \dots
      \item Typicky: \code{tcgetattr} + \code{cfgetispeed}
    \end{itemize}

    \pause

    \item Zápis: \code{int cfsetispeed(struct termios *termios,} \\
      \code{speed_t speed)}, \code{cfsetospeed}, \code{cfsetspeed} (jen BSD)
    \begin{itemize}
      \item \code{termios}: struktura naplněná pomocí \code{tcgetattr}
      \item \code{speed}: \emph{konstanta} pro rychlost
      \item Typicky: \code{tcgetattr} + \code{cfsetospeed} + \code{tcsetattr}
    \end{itemize}
  \end{itemize}

\end{frame}

    \item Vyzkoušejte

\begin{frame}[fragile]{Další vlastnosti}

  \begin{itemize}
    \item Nastavení speciálních znaků
    \begin{itemize}
      \item \code{intr} (Ctrl-C), \code{eof} (Ctrl-D), \code{stop} (Ctrl-S),
        \code{start} (Ctrl-Q), \dots

      \pause

      \item \code{termios.c_cc[znak]}, kde \code{znak} je \code{VINTR},

    \end{itemize}

    \pause

    \begin{itemize}
      \item Pro vstup (\code{termios.c_iflag}) -- klávesnice, např.:

      \begin{itemize}
	\item \code{ECHO} -- má jít vstup hned na výstup?
      \end{itemize}

    \end{itemize}

  \end{itemize}
