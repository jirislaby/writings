\begin{frame}[fragile]{Direktivy preprocesoru}

  \begin{itemize}[<+->]
    \item \code{#include} -- vložení souboru (\code{\"x.h\"} vs.\ \code{<x.h>})
    \item \code{#define}, \code{#undef} -- definice/zrušení makra
    \item \code{#if}, \code{#else}, \code{#elif}, \code{#endif} -- podmíněný
      překlad
    \item \code{#ifdef}, \code{#ifndef} -- ekvivalent
      \code{#if defined}/\code{#if !defined}
    \item \code{#error}, \code{#warning} -- vypiš chybu a ukonči překlad/vypiš
      varování
  \end{itemize}

  \begin{block}{Příklad}
    \begin{lstlisting}
#include "x.h"

#ifdef X
# define Y printf
#elif defined(Y) && defined(Z)
# warning Both Y and Z defined
#else
# error X undefined
#endif
    \end{lstlisting}
  \end{block}

\end{frame}

\begin{frame}[fragile]{Direktivy preprocesoru}

  \begin{itemize}[<+->]
    \item \code{__FILE__}, \code{__LINE__} -- soubor, řádek
    \item \code{__DATE__}, \code{__TIME__} -- datum, čas
    \item \code{__TIMESTAMP__} -- časová značka
    \item \code{#pragma} -- \textsc{Gcc}: ,,we do not recommend the use of
      pragmas''

    \pause

    \item \code{# macro operator} -- vytvoření řetězce z parametru makra
    \item \code{## macro operator} -- zřetězení dvou parametrů makra
  \end{itemize}

  \begin{block}{Příklad}
    \begin{lstlisting}
#define concat_underscore(x, y) x ## _ ## y

void fun() {
  write(concat_underscore(STDOUT, FILENO), "test", 4);
}
    \end{lstlisting}
  \end{block}

\end{frame}

\begin{frame}[fragile]{Přepínače preprocesoru}
