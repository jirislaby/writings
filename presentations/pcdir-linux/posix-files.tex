\begin{frame}[fragile]{Procházení adresářů}

  \begin{itemize}
    \item Hlavička: \header{dirent.h}
    \item Otevření: \code{DIR *opendir(const char *dirname)}
    \item Čtení: \code{struct dirent *readdir(DIR *dp)}
    \begin{itemize}
      \item Opakované volání postupně prochází adresář
    \end{itemize}
    \item Zavření: \code{int closedir(DIR *dp)}
  \end{itemize}

  \begin{block}{\code{struct dirent}}
  \begin{lstlisting}
struct dirent {
  ino_t d_ino;       /* inode number */
  char  d_name[256]; /* filename */
};
  \end{lstlisting}
  Může obsahovat i další položky (mimo standard POSIX).
  \end{block}

\end{frame}

\begin{frame}{Úkol}

  \heading{Průchod adresáře}
  \begin{enumerate}
    \item Otevřete si nějaký adresář
    \item Vypište položky adresáře
    \item Vyzkoušejte
    \item Jak jsou položky seřazeny?
    \item Jsou vráceny vždy ve stejném pořadí?
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]{Typy položek}

  \begin{itemize}
    \item \code{struct dirent} může mít položku \code{d_type}
    \begin{itemize}
      \item Ale není v žádném standardu
    \end{itemize}

    \pause

    \item \code{int stat(const char *pathname, struct stat *buf)}
    \begin{itemize}
      \item Hlavičky: \header{sys/stat.h}, \header{sys/types.h},
	\header{unistd.h}
    \end{itemize}
  \end{itemize}

  \begin{block}{struct stat}
  \begin{lstlisting}
mode_t    st_mode;        /* protection */
nlink_t   st_nlink;       /* number of hard links */
uid_t     st_uid;         /* user ID of owner */
gid_t     st_gid;         /* group ID of owner */
off_t     st_size;        /* total size, in bytes */
struct timespec st_atim;  /* time of last access */
struct timespec st_mtim;  /* time of last modification */
struct timespec st_ctim;  /* time of last status change */
...
  \end{lstlisting}
  \end{block}

  \begin{itemize}
    \item \code{st_mode} kromě práv (\code{& 07777}) obsahuje i typ
      (\code{& S_IFMT})
    \begin{itemize}
      \item Makra \code{S_ISREG}, \code{S_ISDIR}, \code{S_ISLNK}, \dots
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Typy položek}
  \begin{enumerate}
    \item Upravujte předchozí úkol
    \item Pro každou vypsanou položku navíc vypište
    \begin{itemize}
      \item Práva -- v osmičkové soustavě
      \item Typ položky (r, l, d)
      \item Velikost položky
      \item Vlastníka a skupinu
      \item Čas poslední změny
    \end{itemize}

    \item Vyzkoušejte
  \end{enumerate}

\end{frame}
