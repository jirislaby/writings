\begin{frame}
  \frametitle{Paměť}

  \begin{itemize}
  \item Omezený zásobník (4k-8k)
  \begin{itemize}
  \item Žádná nebo malá rekurze
  \item Jen pro malá data (\texttt{int}, malé \texttt{struct}, krátká pole,
    \ldots)
  \end{itemize}

  \pause

  \item Pracuje se se stránkami (na x86 4K-1G)
  \begin{itemize}
  \item Omezená velikost alokace (fragmentace)
  \item Podrobnosti v dalších cvičeních
  \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Jednoduché alokace}

  \heading{Malé alokace (sta bajtů, max. až cca. 4M)}
  \begin{itemize}
  \item Horní mez závislá na architektuře
  \item \texttt{kmalloc}, \texttt{kfree}
  \end{itemize}

  \pause
  \bigskip

  Alokace mají (většinou) GFP parametr. Ten určuje, co si alokátor může dovolit
  (spát, swapovat, použít HIGHMEM, \ldots). Prozatím nám stačí
  \texttt{GFP\_KERNEL}.

  \pause
  \bigskip

  \begin{lstlisting}[language=C]
void *mem = kmalloc(100, GFP_KERNEL);
if (mem) {
  ...
}
kfree(mem);
  \end{lstlisting}

\end{frame}
