\begin{frame}[fragile]
  \frametitle{Práce s pamětí}

  \begin{enumerate}
  \item Po stránkách (makro \texttt{PAGE\_SIZE})
  \begin{itemize}
  \item \texttt{\_\_get\_free\_page}, \texttt{free\_page}
  \item \texttt{get\_free\_pages}, \texttt{free\_pages} (řád)
  \item Vracejí adresu jako \texttt{unsigned long}
  \end{itemize}

  \pause

  \item Malé alokace (sta bajtů, max. až cca. 4M)
  \begin{itemize}
  \item \texttt{kmalloc}, \texttt{kfree}
  \end{itemize}
  \end{enumerate}

  \pause

  Alokace mají (většinou) GFP parametr. Ten určuje, co si alokátor může dovolit
  (spát, swapovat, použít HIGHMEM, \ldots). Prozatím nám stačí
  \texttt{GFP\_KERNEL}.

  \pause

  \begin{lstlisting}[language=C]
void *mem1 = (void *)__get_free_page(GFP_KERNEL);
void *mem2 = kmalloc(PAGE_SIZE, GFP_KERNEL);
if (mem1 && mem2) {
  ...
}
kfree(mem2);
free_page(mem1);
  \end{lstlisting}

\end{frame}
