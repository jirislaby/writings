\begin{frame}[fragile]
  \frametitle{Jednoduché alokace}

  \heading{Malé alokace (sta bajtů, max. až cca. 4M)}
  \begin{itemize}
  \item Horní mez závislá na architektuře
  \item \texttt{kmalloc}, \texttt{kfree}
  \end{itemize}

  \pause
  \bigskip

  Alokace mají (většinou) GFP parametr. Ten určuje, co si alokátor může dovolit
  (spát, swapovat, použít HIGHMEM, \ldots). Prozatím nám stačí
  \texttt{GFP\_KERNEL}.

  \pause
  \bigskip

  \begin{lstlisting}[language=C]
void *mem = kmalloc(100, GFP_KERNEL);
if (mem) {
  ...
}
kfree(mem);
  \end{lstlisting}

\end{frame}
