\begin{frame}
  \frametitle{Úkol}
  
  \heading{Práce s fyzickou adresou -- přečtení signatury a délky nějaké ACPI
    tabulky}
  \begin{enumerate}
  \item Najít fyzickou adresu nějaké ACPI tabulky v \texttt{dmesg}
  \begin{itemize}
  \item Řádky formátu \texttt{ACPI: RSDP 00000000000f8550 00014}
  \item Jeden si zvolit a zapamatovat adresu (\texttt{0x00000000000f8550})
  \end{itemize}
  \item Získat virtuální adresu (\texttt{u32 *virt = ioremap(phys, 8)})
  \item Vypsat signaturu (text, 4B -- \texttt{virt[0]})
  \item Vypsat délku tabulky (4B -- \texttt{virt[1]})
  \item Odmapovat (\texttt{iounmap(virt)})
  \item Porovnejte délku s výpisem (\texttt{0x00014})
  \item Můžete si přemapovat a vypsat celou tabulku a porovnat položky se
    specifikací ACPI
  \end{enumerate}

  \bigskip

  Pozn.: na starších jádrech se výpis v dmesg liší

\end{frame}

\begin{frame}
  \frametitle{Virtuální prostor}

  \begin{itemize}
  \item To, co vidí CPU (a překládá MMU)
  \begin{itemize}
  \item I 1:1 -- start systému, noMMU, \ldots
  \end{itemize}
  \end{itemize}

  \pause

  \begin{itemize}
  \item OS ,,rozdělí'' FP na stránky
  \begin{itemize}
  \item I různě velké
    \begin{itemize}
    \item Jiné virtuální adresy
    \item Např. přístupy programů po 4K, přístupy jádra po 1G (méně tabulek)
    \end{itemize}
  \item Index stránky je tzv. page frame number (pfn)
  \end{itemize}
  \item OS namapuje stránky do VP
  \begin{itemize}
  \item A může přidat ,,virtuální'' stránky např. na disku (swap)
    \begin{itemize}
    \item Programy vidí více ,,paměti''
    \end{itemize}
  \item Toto mapování čte MMU (ví odkud)
  \end{itemize}
  \end{itemize}

  \pause

  \begin{itemize}
  \item Každá stránka má svoji \texttt{struct page}
  \begin{itemize}
  \item Informace o stránce (nikdy nevyswapovat, špinavá, počet uživatelů)
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}
  
  \heading{Práce s virtuální adresou}
  \begin{enumerate}
  \item \header{linux/mm.h}
  \item Naalokujte stránku (\texttt{virt=\_\_get\_free\_page(GFP\_KERNEL)})
  \item Nakopírujte do ní řetězec
  \item Zjistěte fyzickou adresu (\texttt{phys=virt\_to\_phys(virt)})
  \item Zjistěte adresu {struct page} (\texttt{page=virt\_to\_page(virt)})
  \item Zarezervujte stránku (\texttt{SetPageReserved(page)})
  \item Přemapujte fyzickou stránku (\texttt{map=ioremap(phys, \ldots)})
  \item Vypište \texttt{virt}, \texttt{phys}, \texttt{page}, \texttt{map} a
    \texttt{page\_to\_pfn(page)}
  \item Vypište obsah (\texttt{\%s}) \texttt{virt} a \texttt{map}
  \item Změňte obsah \texttt{map} a vypište obsah \texttt{virt}
  \item \texttt{iounmap}, \texttt{ClearPageReserved} a \texttt{free\_page}
  \item Zkonzultujte s \doc{Documentation/x86/x86\_64/mm.txt}
  \end{enumerate}
\end{frame}

\begin{frame}
  \frametitle{Alokace paměti}

  \heading{Typy alokací}
  \begin{itemize}
  \item Malé bloky až souvisle po stránkách (\texttt{kmalloc})
  \item Souvisle po stránkách (\texttt{\_\_get\_free\_pages})
  \item Nesouvisle po stránkách a namapovat (\texttt{vmalloc})
  \end{itemize}

  \pause
  \medskip

  \heading{GFP\_*}
  \begin{itemize}
  \item Parametr pro alokátory
  \item \texttt{GFP\_ATOMIC} -- nespi (např. uvnitř spinlocků)
  \item \texttt{GFP\_TEMPORARY} -- uvolním během pár příštích instrukcí
  \item \texttt{GFP\_KERNEL} -- obyčejné alokace všude jinde
  \item a další: \texttt{GFP\_NOFS}, \texttt{GFP\_NOIO}, \ldots
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{\texttt{kmalloc}}

  \begin{itemize}
  \item \header{linux/slab.h}
  \item Až cca. 8M (různí se pro architektury)
  \item Využívá 2. typ alokátorů, ale rozděluje stránky na menší kusy
  \item \texttt{kmalloc}, \texttt{kzalloc}, \texttt{kfree}
  \item Různé implementace: SLAB, \emph{SLUB}, SLOB, SLQB
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{\texttt{\_\_get\_free\_pages}}

  \begin{itemize}
  \item \header{linux/mm.h}
  \item Stránka má velikost \texttt{PAGE\_SIZE} (x86: 4K)
  \item Velikost alokace se udává řádem, \texttt{order} ($2^{order}$ = počet
    stránek)
  \begin{itemize}
  \item \texttt{get\_order}, \texttt{MAX\_ORDER} (na x86 = 11 $=$ 8M)
  \end{itemize}
  \item Musí najít souvislý blok volných stránek
  \begin{itemize}
  \item Může být problém najít větší bloky (řádu $>=$ 2) kvůli fragmentaci
  \end{itemize}
  \item Používá tzv. \emph{buddy} systém
  \begin{itemize}
  \item Snižuje fragmentaci
  \item Podrobnosti viz wiki
  \end{itemize}
  \item \texttt{\_\_get\_free\_pages}, \texttt{free\_pages}
  \item \texttt{\_\_get\_free\_page}, \texttt{\_\_free\_page}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{\texttt{vmalloc}}

  \begin{itemize}
  \item \header{linux/vmalloc.h}
  \item Podobný \texttt{kmalloc}
  \item Neatomický (nemá \texttt{GFP\_*})
  \item Maximální velikost daná omezením architektury
  \begin{itemize}
  \item x86: 128M (lze zvýšit parametrem, ne o moc)
  \item x86\_64: 30T
  \end{itemize}
  \item Alokuje po stránkách a mapuje je souvisle
  \begin{itemize}
  \item Tj. používá 2. alokátor
  \item Fyzické stránky jsou umístěné různě po RAM
  \end{itemize}
  \item Dražší
  \item \texttt{vmalloc}, \texttt{vfree}
  \item Odlišná práce s adresami
  \begin{itemize}
  \item \texttt{vmalloc\_to\_page}, \texttt{vmalloc\_to\_pfn}
  \end{itemize}
  \end{itemize}

\end{frame}
