  \end{itemize}

\begin{frame}
  \frametitle{Úkol}

  \heading{Atomické čtení/zápis bufferu o velikosti 128 bytů}
  \begin{itemize}
  \item Globální buffer
  \item 2 znaková (misc) zařízení
  \begin{itemize}
  \item 1 implementuje \code{.read}
  \item 1 implementuje \code{.write}
  \end{itemize}
  \item Zápis
  \begin{itemize}
  \item Umožněn max. po 5 znacích (\code{.write} vrací max. 5)
  \item Spí 20 ms po každém zápisu \emph{znaku} do bufferu (\code{msleep} z
    \header{linux/delay.h})
  \end{itemize}
  \item Čtení
  \begin{itemize}
  \item Vrátí naráz celých 128 B (je-li \code{count} dostatečně velký)
  \item Musí vidět změny pouze po 5 znacích (až na poslední pětici)
  \end{itemize}
  \item Vyzkoušejte
  \end{itemize}

  \medskip

  Pozn. 1: práce s \code{offp} v pb173/04\\
  Pozn. 2: odevzdat s domácím

\end{frame}
