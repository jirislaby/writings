\begin{frame}[fragile]
  \frametitle{Zámky v jádře -- spinlocky}

  \begin{itemize}
  \item \header{linux/spinlock.h}, \doc{Documentation/spinlocks.txt}
  \item \texttt{DEFINE\_SPINLOCK(lock)}, \texttt{spinlock\_t lock}
  \item \texttt{spin\_lock}, \texttt{spin\_unlock}
  \item Podobné \texttt{pthread} spinlockům
  \end{itemize}

  \pause

  \begin{center}
  \begin{tabular}{ccc}
  \begin{lstlisting}[language=C]
int *addr;
...

  \end{lstlisting} & $\Rightarrow$ &
  \begin{lstlisting}[language=C]
DEFINE_SPINLOCK(addr_lock);

spin_unlock(&addr_lock);
  \end{lstlisting}

\begin{frame}
  \frametitle{Úkol}
  
  \heading{Atomické čtení/zápis bufferu o velikosti 128 bytů}
  \begin{itemize}
  \item Globální buffer
  \item 2 znaková (i misc) zařízení
  \begin{itemize}
  \item 1 implementuje \texttt{.read}
  \item 1 implementuje \texttt{.write}
  \end{itemize}
  \item Zápis
  \begin{itemize}
  \item Umožněn max. po 5 znacích (\texttt{.write} vrací max. 5)
  \item Spí 10 ms po každém zápisu znaku do bufferu (\texttt{msleep} z
    \header{linux/delay.h})
  \end{itemize}
  \item Čtení
  \begin{itemize}
  \item Vrátí naráz celých 128 B (je-li \texttt{count} dostatečně velký)
  \item Musí vidět změny pouze po 5 znacích (až na poslední pětici)
  \end{itemize}
  \item Vyzkoušejte
  \end{itemize}
