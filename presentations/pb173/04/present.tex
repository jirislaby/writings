\begin{frame}[fragile]
  \frametitle{Atomické operace}

  \begin{itemize}
  \item \header{asm/atomic.h}, \doc{Documentation/atomic\_ops.txt}
  \item \texttt{atomic\_t a = ATOMIC\_INIT(5)}
  \item Pojme 32 bitů se znaménkem (\texttt{int}) (historicky jen 24)
  \item \texttt{atomic\_read}, \texttt{atomic\_set}
  \item \texttt{atomic\_add}, \texttt{atomic\_inc}, \texttt{atomic\_sub},
    \texttt{atomic\_dec}, \texttt{atomic\_*\_return} a další (LXR)
  \end{itemize}

  \pause

  \begin{center}
  \begin{tabular}{ccc}
  \begin{lstlisting}[language=C]

...

a = a + 1;
store(a, addr);
  \end{lstlisting} & $\Rightarrow$ &
  \begin{lstlisting}[language=C]
atomic_t a;
...
atomic_inc(&a);
/* nebo atomic_add(1, &a); */
  \end{lstlisting}
  \end{tabular}\\[2mm]
  \emph{\footnotesize Řešení pomocí atomických operací}
  \end{center}

  \item \texttt{atomic64\_t} (drahý na 32-bitu)
  \end{itemize}

\begin{frame}[fragile]

  \begin{center}

\begin{frame}
  \frametitle{Úkol}

  \heading{Atomické čtení/zápis bufferu o velikosti 128 bytů}
  \begin{itemize}
  \item Globální buffer
  \item 2 znaková (misc) zařízení
  \begin{itemize}
  \item 1 implementuje \texttt{.read}
  \item 1 implementuje \texttt{.write}
  \end{itemize}
  \item Zápis
  \begin{itemize}
  \item Umožněn max. po 5 znacích (\texttt{.write} vrací max. 5)
  \item Spí 20 ms po každém zápisu znaku do bufferu (\texttt{msleep} z
    \header{linux/delay.h})
  \end{itemize}
  \item Čtení
  \begin{itemize}
  \item Vrátí naráz celých 128 B (je-li \texttt{count} dostatečně velký)
  \item Musí vidět změny pouze po 5 znacích (až na poslední pětici)
  \end{itemize}
  \item Vyzkoušejte
  \end{itemize}

  \medskip

  Pozn. 1: práce s \texttt{offp} v pb173/04\\
  Pozn. 2: odevzdat s domácím
