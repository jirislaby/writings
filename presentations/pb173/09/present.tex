  \end{enumerate}

\begin{frame}[fragile]
  \frametitle{Parametry \code{mmap} v jádře}

  \begin{itemize}
    \item V uživatelském prostoru volání \code{mmap}
    \begin{itemize}
      \item \footnotesize{\code{void *mmap(void *addr, size_t len,
	int prot, int flags, int fd, off_t off)}}
    \end{itemize}

    \item V jádře jedna položka v \code{struct file_operations}
    \begin{itemize}
      \item \footnotesize{\code{int mmap(struct file *filp,
	struct vm_area_struct *vma)}}

      \pause

      \item Parametry jsou v \code{struct vm_area_struct}
    \end{itemize}
  \end{itemize}

  \medskip
  \pause

  \begin{lstlisting}
struct vm_area_struct {
  unsigned long vm_start; /* addr or random when addr is NULL */
  unsigned long vm_end;   /* vm_end = vm_start+len */
  unsigned long vm_pgoff; /* vm_pgoff = off/PAGE_SIZE */
  unsigned long vm_flags; /* vm_flags = encoded(flags|prot) */
  pgprot_t vm_page_prot;  /* only for remap_* functions */
  ...
  const struct vm_operations_struct *vm_ops; /* later ... */
  void *vm_private_data;
}
  \end{lstlisting}

  \pause

  Je třeba namapovat stránky mezi \code{vm_start} a \code{vm_end}. \\
  Ale také ověřit privilegia (čtení, zápis, spuštění). \emph{POZOR}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Základní \code{mmap} funkce}

  \heading{API}
  \begin{itemize}
  \item \header{linux/mm.h}, \code{struct vm_area_struct}
  \item \code{__get_free_page/pages} $\Rightarrow$
  \code{remap_pfn_range}
  \item \code{vmalloc_user} $\Rightarrow$ \code{remap_vmalloc_range}
  \end{itemize}

  \begin{lstlisting}[language=C]
int my_init(void)
{
  mem = __get_free_pages(GFP_KERNEL, 2);
  /* mem = vmalloc_user(PAGE_SIZE); */
}
...
int my_mmap(struct file *filp, struct vm_area_struct *vma)
{
  if ((vma->vm_flags & (VM_WRITE | VM_READ)) != VM_READ)
    return -EINVAL;
  return remap_pfn_range(vma, vma->vm_start, page_to_pfn(virt_to_page(mem)), 4 * PAGE_SIZE, vma->vm_page_prot);
  /* return remap_vmalloc_range(vma, mem, 0); */
}
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Přemapování prostorů}
  \begin{enumerate}
    \item Alokovat 2+2 stránky
    \begin{itemize}
      \item Dvoustránku pomocí \code{vmalloc}
      \item Dvoustránku pomocí stránkového alokátoru
    \end{itemize}

    \item Zapsat na všechny 4 stránky libovolný, ale různý řetězec
    \item Vystavit (mapovat) stránky v \code{mmap}
    \begin{itemize}
      \item První dvojice RO, druhá R/W (ověření \code{prot})
      \item $0 \leq \text{\code{pgoff}} < 2$ $\Rightarrow$ jedno mapování
      \item $2 \leq \text{\code{pgoff}} < 4$ $\Rightarrow$ druhé mapování
    \end{itemize}

    \item Z userspace 2$\times$ \code{mmap} s \code{off} \code{0} a
      \code{2*4096}
    \begin{itemize}
      \item Již hotovo v \file{pb173/09}
    \end{itemize}
  \end{enumerate}

  \medskip

  \heading{API (zopakováno)}
  \begin{itemize}
  \item \header{linux/mm.h}, \code{struct vm_area_struct}
  \item \footnotesize{\code{int mmap(struct file *filp,
    struct vm_area_struct *vma)}}
  \item \code{__get_free_page*} $\Rightarrow$ \code{remap_pfn_range}
  \item \code{vmalloc_user} $\Rightarrow$ \code{remap_vmalloc_range}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{\code{mmap} po stránkách}

  \heading{Přemapování roztroušených stránek}
  \begin{itemize}
    \item Přes \code{remap_pfn_range} obtížně
    \item Při výpadcích stránek se mapují takové stránky jednotlivě
    \begin{itemize}
      \item Každý ovladač má ,,page fault handler''
      \item Stará jádra: \code{nopage}
      \item Nová jádra: \code{fault}
    \end{itemize}

    \pause

    \item V \code{mmap}/\code{nopage}/\code{fault} je třeba zkontrolovat
      rozsahy a velikosti
    \begin{itemize}
      \item Předtím to dělaly \code{remap_*_range} funkce
    \end{itemize}

    \pause

    \item \code{vma->vm_ops}
    \begin{itemize}
      \item Struktura ukazující na háčky (včetně \code{nopage}/\code{fault})
    \end{itemize}

    \pause

    \item \code{vma->vm_private_data}
    \begin{itemize}
      \item Pro naše potřeby
      \item K předání informací z \code{file_operations->mmap} do
        \code{vm_ops->*}
    \end{itemize}
  \end{itemize}

\end{frame}

  \frametitle{Úkol}

  \heading{Mapování roztroušených stránek} (součást domácího)
  \begin{enumerate}
    \item Předchozí příklad rozšiřte
    \item Místo alokací dvoustránek alokujte 4 samostatné stránky
    \begin{itemize}
      \item 2$\times$ \code{vmalloc} a 2$\times$ \code{__get_free_pages}
    \end{itemize}

    \item Přemapovat stránky v \code{mmap}
    \begin{itemize}
      \item Změna \code{remap_pfn_range} na \code{vma->vm_ops->fault}
    \end{itemize}

    \item Userspace program stejný (funkčnost navenek stejná)
  \end{enumerate}

  \medskip

  \heading{Potřebné podkroky}
  \begin{itemize}
    \item Definice \footnotesize{\code{int fault(struct vm_area_struct *vma,
