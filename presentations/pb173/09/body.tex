\begin{frame}
  \frametitle{Časovače}

  \begin{itemize}
    \item \doc{Documentation/timers/*}
    \item Činnost v nastavených časech (i periodicky)

    \pause

    \item Kód je volán v kontextu přerušení (nelze spát)
    \item \file{/proc/timer_list} (mimo historická jádra)
  \end{itemize}

  \pause

  \heading{Obyčejné časovače}
  \begin{itemize}
    \item Fungují na každé architektuře

    \pause

    \item Založené na \code{jiffies}
    \begin{itemize}
      \item Proměnná zvyšující se o 1 s frekvencí \code{HZ}
    \end{itemize}

    \pause

    \item Rozlišení: 1--10 ms (pro \code{HZ}: 1000--100)
  \end{itemize}

  \pause

  \heading{High-res časovače}
  \begin{itemize}
    \item \code{usleep_range} (\header{linux/hrtimer.h})
    \item Rozlišení: $\sim \mu$s
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Obyčejné časovače}

  \heading{API}
  \begin{itemize}
    \item \header{linux/timer.h}, \code{struct timer_list}
    \item \code{DEFINE_TIMER}$_S$, \code{setup_timer}$_D$

    \pause

    \item Načasování: \code{mod_timer} (=\code{del_timer}+\code{add_timer})
    \item Zrušení: \code{del_timer_sync} (\emph{POZOR})
  \end{itemize}

  \onslide<1->

  \begin{block}{Příklad}
    \begin{lstlisting}[aboveskip=2pt,belowskip=0pt]
static void my_fun(unsigned long data);
static DEFINE_TIMER(my_timer, my_fun, 0, 30);
    \end{lstlisting}
    \onslide<2->
    \begin{lstlisting}[aboveskip=0pt,belowskip=2pt]
static void my_fun(unsigned long data)
{
  my_timer.data *= 2;
  mod_timer(&my_timer, jiffies + msecs_to_jiffies(data));
}
...
  mod_timer(&my_timer, jiffies + msecs_to_jiffies(100));
...
  del_timer_sync(&my_timer);
    \end{lstlisting}
  \end{block}

  \begin{textblock*}{4.1cm}(\textwidth - 4.3cm,\textheight - 3cm)
    \begin{onlyenv}<3->
      \heading{Úkol:} Každou vteřinu, kdy je modul v systému, vypište \code{HZ}
	(\code{\%u}) a \code{jiffies} (\code{\%lu})
    \end{onlyenv}
  \end{textblock*}

\end{frame}

\begin{frame}

  \end{itemize}

  \begin{enumerate}

  \end{enumerate}

  \heading{Rozdíly}
  \begin{itemize}
    \item Rychlost zavolání (tasklet dřív)
    \item Kontext zavolání (tasklet z přerušení, workqueue z procesu)
    \item Množství kódu (tj.\ rychlost vykonání)
  \end{itemize}

\begin{frame}
  \frametitle{Workqueue}

  \begin{itemize}
    \item \doc{Documentation/workqueue.txt}

    \item Speciální \emph{procesy} volající funkce řazené do fronty
    \begin{itemize}
      \item Globální, společné pro všechny -- většinou stačí

      \pause

      \item Vlastní -- pro speciální případy
    \end{itemize}

    \pause

    \item Lze v nich spát

    \item Spustí se, až se plánovač rozhodne
    \begin{itemize}
      \item Lze specifikovat minimální prodlevu
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Workqueue}{API}

  \begin{itemize}
    \item \header{linux/workqueue.h}, \code{struct work_struct}
    \item Definice práce: \code{DECLARE_WORK}$_S$, \code{INIT_WORK}$_D$

    \pause

    \item Globální proces:
    \begin{itemize}
      \item Zařazení do fronty: \code{schedule_work}
      \item Vyřazení z fronty (předčasně): \code{cancel_work_sync}
    \end{itemize}

    \pause

    \item Vlastní proces:
    \begin{itemize}
      \item Vytvoření procesu: \code{create_workqueue},
	\code{destroy_workqueue}
      \item Zařazení do fronty: \code{queue_work}
      \item Vyřazení z fronty: \code{cancel_work_sync} (tj.\ stejně)
    \end{itemize}

    \pause

    \item Delayed (s garantovanou prodlevou): \code{*_delayed_work}
  \end{itemize}

  \pause
  \medskip

  \heading{Úkol:} v \code{module_init} nahrajte modul \code{ieee80211} nebo
    \code{mac80211} (\code{request_module}). \code{request_module} nelze volat
    přímo v \code{module_init} a je nutný i vlastní workqueue proces.

\end{frame}

\begin{frame}
  \frametitle{Tasklet}

  \begin{itemize}
    \item Běží v kontextu přerušení (jako časovače)
    \item Nelze spát
    \item Musí být rychlý
    \item Spustí se \emph{po} příštím přerušení
  \end{itemize}

  \pause
  \smallskip

  \heading{API}
  \begin{itemize}
    \item \header{linux/interrupt.h}, \code{struct tasklet_struct}
    \item Definice taskletu: \code{DECLARE_TASKLET}$_S$,
      \code{tasklet_init}$_D$

    \pause

    \item Zařazení do fronty: \code{tasklet_schedule}
    \item Vyřazení z fronty: \code{tasklet_kill}
  \end{itemize}

  \pause
  \bigskip

  \heading{Úkol:} vytvořte tasklet, spusťte a vypište posloupnost volání pomocí
    \code{dump_stack}

\end{frame}

\begin{frame}
  \frametitle{Přerušení HW}

  \begin{itemize}
    \item HW informuje o změně stavu
    \begin{itemize}
      \item Časovač tiknul, přišel paket, přečten blok z disku, \dots
    \end{itemize}

    \pause

    \item CPU přeruší chod programu/jádra a zavolá jádro
    \begin{itemize}
      \item Ví, koho volat, má tabulku (IDT)
      \item Cyklus? Priority přerušení!
    \end{itemize}

    \pause

    \item OS musí obsloužit přerušení
    \begin{itemize}
      \item Zjistí zdroj a zavolá odpovídající ovladač (jeho obsluhu přerušení)
      \begin{itemize}
	\item Přerušení je identifikováno číslem (IRQ)
      \end{itemize}

      \item Vynuluje přerušení na řadiči přerušení
    \end{itemize}
  \end{itemize}

  \pause
  \bigskip

  \heading{Nutná podpora HW (CPU, řadiče, sběrnice, zařízení)}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Přerušení v Linuxu}

  \begin{itemize}
  \item \header{linux/interrupt.h}
  \item \code{request_irq}, \code{free_irq}
  \item Flags: hlavně \code{IRQF_SHARED}
  \item Návratová hodnota z obsluhy: \code{IRQ_NONE}, \code{IRQ_HANDLED}
  \item \file{/proc/interrupts}
  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}
static irqreturn_t my_handler(int irq, void *data)
{ /* data == my_data */
  return my_device_raised_interrupt ? IRQ_HANDLED : IRQ_NONE;
}
static int my_probe(struct pci_dev *pdev, ...)
{
  /* here: enable device etc. */
  my_data = kmalloc(...);
  ret = request_irq(pdev->irq, my_handler, IRQF_SHARED, "my", my_data);
}
  \end{lstlisting}
  \end{block}

  \pause

  \heading{Úkol:} vytvořte obsluhu pro EDU (zatím vracejte \code{IRQ_HANDLED})

\end{frame}

\begin{frame}{Úkol}

  \heading{Práce s přerušením} \iffimuni (součást domácího) \fi
  \begin{enumerate}
    \item Vytvořte pojmenování (makra) pro registry (viz tabulku dole)

    \item Spusťte 100ms periodický časovač (pro každé EDU zařízení)
    \begin{itemize}
      \item Nechte počítat faktoriál a po výpočtu generovat přerušení (r.\ 0x20)
      \item Můžete občas nechat EDU vyvolat i jiné přerušení (r.\ 0x60)
    \end{itemize}

    \item Navažte se v probe na přerušení (\code{request_irq})

    \item Implementujte obsluhu
    \begin{itemize}
      \item Přečtěte stav (r.\ 0x24), vypište stav -- limitovaně
      \item Nenula: odsouhlaste přerušení (r.\ 0x64), vraťte \code{IRQ_HANDLED}
      \item Jinak: vráťte \code{IRQ_NONE}
    \end{itemize}
  \end{enumerate}
