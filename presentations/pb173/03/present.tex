\begin{frame}
  \frametitle{Výpisy}
  \begin{itemize}
  \item \code{printk} již známe
  \item Před formátovací řetězec se vkládá \code{KERN_*}
  \begin{itemize}
  \item Řetězec ve formě \texttt{<číslo>} nebo \texttt{\textbackslash 001číslo}
  \item Sděluje úroveň (důležitost) zprávy
  \item Definované v \header{linux/kernel.h}
  \end{itemize}
  \end{itemize}

  \begin{block}{Obvyklé nastavení}
  \begin{center}
  \begin{tabular}{|l|l|l|l|} \hline
  Důležitost & Konzole & \file{/var/log/*}   & \cmd{dmesg} \\ \hline\hline
  vysoká     & ANO     & ANO                 & ANO         \\ \hline
  normální   & NE      & ANO                 & ANO         \\ \hline
  ladicí     & NE      & NE                  & ANO         \\ \hline\hline
  zapisuje   & jádro   & syslog*             & jádro       \\ \hline
  \end{tabular}
  \end{center}
  \end{block}

  \begin{center}
  \heading{Úroveň ovlivňuje, co se kde zobrazí}
  \end{center}

\end{frame}

\begin{frame}
  \frametitle{Nastavení výpisů}

  \begin{itemize}
  \item \cmd{dmesg} nastavit nelze
  \begin{itemize}
    \item Obsahuje vždy všechno
  \end{itemize}
  \item Nastavení syslogu závislé na systému
  \end{itemize}

  \pause
  \medskip

  \heading{Nastavení konzole}
  \begin{itemize}
  \item \texttt{dmesg -n číslo}
  \begin{itemize}
    \item \emph{Zprávy s vyšší úrovní se objeví na konzoli}
    \item Hodnota 8 $\Rightarrow$ všechno dění
    \begin{itemize}
      \item Např.\ při ,,hard-locku''
    \end{itemize}
  \end{itemize}
  \pause
  \item Demo: \file{pb173/03} + \cmd{dmesg -n}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

    \item Přidejte si do kódu modulu \code{WARN_ON(1)}
    \item Zjistěte s jakou úrovní se varování vypisuje
    \begin{itemize}
      \item Pomocí postupného nastavování \cmd{dmesg -n}
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Úrověň výpisů}

  \heading{Volba úrovně}
  \begin{itemize}
    \item Nechceme 10G \file{/var/log/messages}
    \begin{itemize}
      \item Ladicí výpisy: \code{DEBUG}
    \end{itemize}
    \item Nechceme spoustu hlášek na konzoli
    \begin{itemize}
      \item Obvyklé chyby: \code{WARNING}, \code{ERR}
    \end{itemize}
  \end{itemize}

  \pause
  \bigskip

  \heading{Příklady}
  \begin{lstlisting}

printk(KERN_ERR "button.c: Not enough memory\n");
printk(KERN_WARNING "Only using first memory bank\n");
printk(KERN_NOTICE "PCI: Starting initialization.\n");
printk(KERN_INFO "SMP mode deactivated.\n");
printk(KERN_DEBUG "cosa%d: microcode started\n", ...
  \end{lstlisting} %stopzone

\end{frame}

\begin{frame}[fragile]
  \frametitle{Nevýhody výpisů}

  \begin{itemize}

    \begin{itemize}

      \item Nestíhá se posílat na konzoli (ztráta informací)
      \item \emph{Částečné řešení} (\header{linux/printk.h}):
      \begin{lstlisting}
if (printk_ratelimit())
  printk(...);
      \end{lstlisting}
    \end{itemize}

    \pause

    \item Binární data
    \begin{itemize}
      \item \cmd{dmesg} zobrazuje jen textová data
      \item \emph{Částečné řešení} (\header{linux/kernel.h}):
      \begin{lstlisting}
print_hex_dump_bytes(...) // Vyzaduje DEBUG
print_hex_dump(...)
      \end{lstlisting}
    \end{itemize}
  \end{itemize}

  \pause

  \begin{center}
  \heading{Mnoho binárních dat se předává pomocí speciálních souborů}
  \end{center}

\end{frame}

\section{debugfs}
\frame{\sectionpage}

\begin{frame}
  \frametitle{debugfs}

  \begin{itemize}
    \item Předávání ladicích dat přes soubor
    \item Většinou v \file{/sys/kernel/debug/}
    \begin{itemize}
      \item Pokud ne: \cmd{mount -t debugfs none /sys/kernel/debug/}
    \end{itemize}
    \item \doc{Documentation/filesystems/debugfs.txt}
    \item \header{linux/debugfs.h}
    \item \code{debugfs_create_dir}, \\
      \code{debugfs_create_file} (\code{file_operations}), \\
      \code{debugfs_create_symlink},\\
      \code{debugfs_create_u\{8,16,32,bool\}},\\
      \code{debugfs_remove} a \code{debugfs_remove_recursive}
    \item Lze si předat \code{data}, jsou pak v \code{inode->i_private} (v
      \code{open})
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vytvoření položek v debugfs}
  \begin{enumerate}
  \item Vytvořit adresář (\code{debugfs_create_dir})

  \item Vytvořit v něm soubor (\code{debugfs_create_u16})
  \begin{itemize}
    \item Vrací nějakou hodnotu (např.\ 11320)
  \end{itemize}

  \item Přeložit, vložit do systému (\cmd{make}, \cmd{insmod})

  \item Přečíst soubor
  \begin{itemize}
    \item \cmd{cat /sys/kernel/debug/<dir>/<file>}
  \end{itemize}

  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{SysRq}

  \begin{itemize}
  \item System-request (\doc{Documentation/sysrq.txt})
  \begin{itemize}
    \item Klávesnicí: Alt-PrintScreen
    \item Vzdáleně: \file{/proc/sysrq-trigger}
  \end{itemize}
  \item Funguje (většinou) i po pádu systému
  \item Zapnutí: \cmd{sysctl kernel.sysrq=1}
  \end{itemize}

  \pause

  \begin{center}
  \small
  \begin{tabular}{|l|l|} \hline
  h & zkrácená nápověda \\ \hline
  r & ,,reset klávesnice'' \\ \hline
  p & výpis tras procesů na CPU \\ \hline
  t & výpis tras všech procesů \\ \hline
  w & výpis blokujících procesů \\ \hline
  s-u-s-b & sync + unmount + reboot \\ \hline
  \end{tabular} \\[2mm]
  \emph{Příklady SysRq}
  \end{center}

\end{frame}


\begin{frame}
  \frametitle{Úkol}

  \heading{Sysrq}
  \begin{enumerate}
    \item Vypište si nápovědu (\cmd{sysrq-h})
    \item Vypište si trasy procesů na CPU (\cmd{sysrq-p})
    \item Vypište si informace o paměti (\cmd{sysrq-m})
  \end{enumerate}

  \bigskip

  Pozn.: v Qemu lze vyzkoušet v ovládací konzoli (Ctrl-Alt-2) příkazem
  \texttt{sendkey alt-sysrq-písmeno}

\end{frame}
