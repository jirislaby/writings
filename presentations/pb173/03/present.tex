\begin{frame}
  \frametitle{Výpisy}
  \begin{itemize}
  \item \texttt{printk} již známe
  \item Před formátovací řetězec se vkládá \texttt{KERN\_*}
  \begin{itemize}
  \item Řetězec ve formě \texttt{<číslo>}
  \item Sděluje úroveň (důležitost) zprávy
  \item Definované v \header{linux/kernel.h}
  \end{itemize}
  \end{itemize}

  \begin{center}
  \begin{tabular}{|l|l|l|l|} \hline
  Důležitost & Konzole & \texttt{/var/log/*} & \texttt{dmesg} \\ \hline\hline
  vysoká     & ANO     & ANO                 & ANO            \\ \hline
  normální   & NE      & ANO                 & ANO            \\ \hline
  ladicí     & NE      & NE                  & ANO            \\ \hline\hline
  zapisuje   & jádro   & syslog*             & jádro          \\ \hline
  \end{tabular} \\[2mm]
  \emph{Obvyklé nastavení}

  \medskip

  \heading{Úroveň ovlivňuje, co se kde zobrazí}
  \end{center}

\end{frame}

\begin{frame}
  \frametitle{Nastavení výpisů}

  \begin{itemize}
  \item \texttt{dmesg} nastavit nelze
  \item Nastavení syslogu závislé na systému

  \pause

  \item Nastavení konzole
  \begin{itemize}
  \item \texttt{/proc/sys/kernel/printk}
  \item 4 čísla:
    \begin{enumerate}
    \item \emph{Zprávy s vyšší úrovní se objeví na konzoli} (\texttt{dmesg -n})
    \item Zprávy bez úrovně dostanou tuto
    \item Minimum pro bod 1
    \item Nevyužito
    \end{enumerate}
  \item Hodnota 8 $\Rightarrow$ všechno dění
  \begin{itemize}
  \item Např. při ,,hard-locku''
  \end{itemize}
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}

  \end{itemize}

  \medskip

  \doc{Documentation/console/console.txt}\\
  \doc{Documentation/serial-console.txt}

\end{frame}

\begin{frame}
  \frametitle{Úkol}
  \heading{Použití netconsole}
  \begin{enumerate}
  \item Na hostovi: \texttt{nc -ul 0.0.0.0 60000}
  \item \texttt{dhclient}, je-li třeba
  \item \texttt{dmesg -n 8} (vypisovat vše)
  \item \texttt{modinfo netconsole}
  \item \texttt{modprobe netconsole netconsole=@/,60000@10.0.2.2/}
  \item Např. \texttt{echo h >/proc/sysrq-trigger}
  \end{enumerate}

  Více informací v \doc{Documentation/networking/netconsole.txt}

\end{frame}

\begin{frame}
  \frametitle{Omezení výpisů}

  \begin{itemize}
  \item Mnoho dat
  \begin{itemize}
  \item Znepřehlednění (typické pro ACPI, iwlwifi)
  \item Nestíhá se posílat na konzoli (ztráta informací)
  \item \emph{Řešení} (částečné): \texttt{if (printk\_ratelimit()) printk();}
  \end{itemize}
  \item Binární data
  \begin{itemize}
  \item \texttt{dmesg} zobrazuje jen textová data
  \item \emph{Řešení} (částečné): \texttt{print\_hex\_dump\_bytes}
    (\header{linux/kernel.h})
  \end{itemize}
  \end{itemize}

  \medskip
  \pause

  \begin{center}
  \heading{Mnoho binárních dat se předává pomocí speciálních souborů}
  \end{center}

\end{frame}

\end{frame}

\begin{frame}
  \frametitle{SysRq}

  \heading{Vyžádání akce jádra}
  \begin{itemize}
  \item System-request (Alt-PrintScreen)
  \item Funguje (většinou) i po pádu systému
  \item Zapnutí: \texttt{sysctl kernel.sysrq=1}
  \item \doc{Documentation/sysrq.txt}
  \end{itemize}

  \pause

  \begin{center}
  \small
  \begin{tabular}{|l|l|} \hline
  h & zkrácená nápověda \\ \hline
  p & výpis tras procesů na CPU \\ \hline
  t & výpis tras všech procesů \\ \hline
  w & výpis blokujících procesů \\ \hline
  s-u-s-b & sync + unmount + reboot \\ \hline
  \end{tabular} \\[2mm]
  \emph{Příklady SysRq}
  \end{center}

  Pozn.: v Qemu lze vyzkoušet v ovládací konzoli (Ctrl-Alt-2) příkazem
  \texttt{sendkey alt-sysrq-p}

\end{frame}
