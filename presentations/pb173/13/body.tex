\begin{frame}[fragile]
  \frametitle{Firmware}

  \begin{itemize}
  \item Binární kód v zařízení
  \item Nahrává jej (většinou) ovladač
  \end{itemize}

  \bigskip
  \pause

  \heading{API}
  \begin{itemize}
    \item \header{linux/firmware.h}, \code{struct firmware}
    \item \code{request_firmware(const struct firmware **fw, const char *name,
      struct device *device)}
    \begin{itemize}
      \item \code{fw} -- návratová hodnota
      \item \code{name} -- jméno souboru
    \end{itemize}

    \item \code{release_firmware}
  \end{itemize}

  \bigskip

  \begin{lstlisting}
struct firmware {
  size_t size;
  const u8 *data;
  ...
};
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Nahrání firmware}
  \begin{enumerate}
    \item Zkopírujte \file{pb173/13/fw/firmware_*.bin} do \file{/lib/firmware/}
    \item Nahrajte firmware v ovladači
    \begin{itemize}
      \item V závislosti na architektuře (např.\ makro \code{CONFIG_64BIT})
    \end{itemize}

    \item Zkopírujte data do spustitelné paměti
    \begin{itemize}
      \item \code{__vmalloc + PAGE_KERNEL_EXEC}
    \end{itemize}

    \item V cyklu pro 0 \dots 100
    \begin{itemize}
      \item Zavolejte 0.\ offset alokované paměti
      \item Prototyp: \code{unsigned int (*)(unsigned int)}
      \item Vypište návratovou hodnotu
    \end{itemize}

    \item Zjistěte, co firmware dělá
  \end{enumerate}

\end{frame}
