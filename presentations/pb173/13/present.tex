\begin{frame}[fragile]
  \frametitle{Firmware}

  \begin{itemize}
  \item Binární kód v zařízení
  \item Nahrává jej (většinou) ovladač
  \end{itemize}

  \bigskip
  \pause

  \heading{API}
  \begin{itemize}
  \item \header{linux/firmware.h}, \texttt{struct firmware}
  \item \texttt{request\_firmware(const struct firmware **fw, const char *name,
    struct device *device)}
  \begin{itemize}
  \item \texttt{fw} -- návratová hodnota
  \item \texttt{name} -- jméno souboru
  \end{itemize}
  \item \texttt{release\_firmware}
  \end{itemize}

  \bigskip

  \begin{lstlisting}[language=C]
struct firmware {
  size_t size;
  const u8 *data;
  ...
};
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Nahrátí firmware}
  \begin{enumerate}
  \item Zkopírujte pb173/13/fw/firmware\_*.bin do /lib/firmware
  \item Nahrajte firmware v ovladači
  \begin{itemize}
  \item V závislosti na architektuře (např. makro \texttt{CONFIG\_64BIT})
  \end{itemize}
  \item Zkopírujte data do spustitelné paměti
  \begin{itemize}
  \item \texttt{\_\_vmalloc + PAGE\_KERNEL\_EXEC}
  \end{itemize}
  \item V cyklu pro 0 \ldots 100
  \begin{itemize}
  \item Zavolejte 0. offset alokované paměti
  \item Prototyp: \texttt{unsigned int (*)(unsigned int)}
  \item Vypište návratovou hodnotu
  \end{itemize}
  \item Zjistěte, co firmware dělá
  \end{enumerate}

\end{frame}
