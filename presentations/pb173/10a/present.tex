\begin{frame}
  \frametitle{DMA}

  \heading{Přímý přístup do paměti}
  \begin{itemize}
  \item Prozatím jsme ze/do zařízení četli/zapisovali přes CPU
  \begin{itemize}
  \item Tj. standardními operacemi s (přemapovanou) pamětí
  \end{itemize}
  \item Velké přenosy = velká zátěž CPU
  \begin{itemize}
  \item Cykly, čekání na pomalou sběrnici atd.
  \end{itemize}
  \item Místo toho naprogramujeme HW, aby přenášel data sám
  \begin{itemize}
  \item Nutná podpora HW (sběrnice -- arbitrace, zařízení -- přenosy)
  \end{itemize}
  \end{itemize}

  \heading{Princip}
  \begin{enumerate}
  \item Alokace (speciální) paměti
  \item (TX: vyplnění daty)
  \item Předání ukazatele zařízení
  \item Start přenosu
  \item Přerušení či jiná signalizace konce přenosu
  \item (RX: přečtení dat)
  \end{enumerate}

\end{frame}

\begin{frame}[fragile]
  \frametitle{DMA v Linuxu}

  \heading{API}
  \begin{itemize}
  \item \header{linux/dma-mapping.h}, \doc{Documentation/DMA-*}
  \item Alokace: \texttt{dma\_alloc\_coherent}, \texttt{dma\_free\_coherent}
  \end{itemize}

  \begin{lstlisting}[language=C]
int my_setup_DMA(struct device *dev)
{
  dma_addr_t phys;
  void *virt;

  virt = dma_alloc_coherent(dev, 100, &phys, GFP_KERNEL);
  if (!virt)
    return -ENOMEM;
  memset(virt, 0xfe, 100);
  my_store_to_HW(dev, phys);
  my_start_HW_transfer(dev);
  return 0;
}
  \end{lstlisting}

\end{frame}

\begin{frame}[fragile]
  \frametitle{DMA pro PCI}

  \begin{itemize}
  \item Nastavení masky adres
  \begin{itemize}
  \item Ne všechna zařízení zvládnou adresovat celý fyzický rozsah
  \item \texttt{pci\_set\_dma\_mask(pdev, DMA\_BIT\_MASK(27))}
  \end{itemize}
  \pause
  \item Zapnutí ,,spravování sběrnice''
  \begin{itemize}
  \item Tj. zařízení umí samo iniciovat přenosy apod.
  \item \texttt{pci\_set\_master(pdev)}
  \end{itemize}
  \end{itemize}

  \begin{lstlisting}[language=C]
int my_probe(struct pci_dev *pdev, ...)
{
  dma_addr_t phys;
  void *virt;
  ...
  ret = pci_set_dma_mask(pdev, DMA_BIT_MASK(27));
  if (ret) { /* bail out */ }

  pci_set_master(pdev);

  virt = dma_alloc_coherent(&pdev->dev, 100, &phys, GFP_KERNEL);
  ...
}
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Combo a DMA}

  \begin{itemize}
  \item Zvládá 32-bitové adresy
  \item DMA řadič napojený na LocalBus (1), PCI (2) a PowerPC (4)
  \item Generuje přerušení 8 na kartě
  \item \texttt{cmd} registr \texttt{|A               |        NDstSrcR|}
  \begin{itemize}
  \item Bit 0: \texttt{RUN} (transfer now)
  \item Bits 1-3: \texttt{SOURCE} (BUS)
  \item Bits 4-6: \texttt{DESTINATION} (BUS)
  \item Bit 7: \texttt{NOINT} (do not generate interrupt after transaction)
  \item Bit 31: \texttt{ACKINT} (acknowledge interrupt)
  \end{itemize}
  \end{itemize}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s DMA}
  \begin{enumerate}
  \item Naalokovat stránku DMA prostoru
  \item 10B nainicializovat řetězcem
  \item Přenést 10B na PowerPC sběrnici na adresu 0x40000
  \begin{itemize}
  \item Nastavit všechny 4 registry
  \item Bez přerušení (nastavit \texttt{NOINT})
  \item Počkat na nulovou hodnotu bitu \texttt{RUN} (na dokončení přenosu)
  \end{itemize}
  \item Přenést 10B z PowerPC na DMA stránku + 10
  \begin{itemize}
  \item Vznikne za sebou 2$\times$ stejný vzor
  \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{DMA s přerušením} (domácí)
  \begin{enumerate}
  \item Rozšířit předchozí o přerušení
  \begin{itemize}
  \item Přidat ještě jeden přenos z PowerPC na DMA + 20
  \item Přenos bude bez \texttt{NOINT}
  \item Je třeba povolit přerušení 8 na kartě
  \end{itemize}
  \item Z přerušení iniciovat tasklet
  \item V taskletu vypsat obsah DMA paměti + 20
  \item DMA stránku vystavit přes mmap
  \end{enumerate}

  \bigskip

  Pozn.: pokud se zařízení zasekne (např. špatně nastaveným DMA), je nutné
  stroj vypnou a zapnout, reboot nepomůže

\end{frame}
