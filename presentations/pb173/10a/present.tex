\begin{frame}
  \frametitle{DMA}

  \heading{Přímý přístup do paměti}
  \begin{itemize}
  \item Prozatím jsme ze/do zařízení četli/zapisovali přes CPU
  \begin{itemize}
  \item Tj. standardními operacemi s (přemapovanou) pamětí
  \end{itemize}
  \item Velké přenosy = velká zátěž CPU
  \begin{itemize}
  \item Cykly, čekání na pomalou sběrnici atd.
  \end{itemize}
  \item Místo toho naprogramujeme HW, aby přenášel data sám
  \begin{itemize}
  \item Nutná podpora HW (sběrnice -- arbitrace, zařízení -- přenosy)
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Princip DMA}

  \heading{Princip}
  \begin{enumerate}
  \item Alokace (speciální) paměti
  \item \emph{Pro TX}: vyplnění daty (paket, zvuk, data na disk, \dots)
  \item Předání ukazatele do zařízení
  \item Odstartování přenosu v zařízení
  \item Přerušení či jiná signalizace konce přenosu od zařízení
  \item \emph{Pro RX}: práce s příchozími daty (paket, data z disku, \dots)
  \end{enumerate}

\end{frame}

\begin{frame}

  \end{itemize}

  \pause
  \bigskip

  \footnotesize{\code{void *dma_alloc_coherent(struct device *dev, size_t size, dma_addr_t *dma_handle, gfp_t gfp)}}\\[-0.3em]
  \begin{itemize}
  \item \code{dev} -- zařízení, které bude k paměti přistupovat
    (\code{NULL} pokud neznáme)
  \item \code{size} -- velikost, jakou požadujeme
  \item \code{dma_handle} -- fyzická adresa (návratová hodnota) -- pro
  zařízení
  \item návratová hodnota -- virtuální adresa -- pro nás
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{DMA v Linuxu -- příklad}

  \begin{lstlisting}[language=C]
int my_do_DMA(struct device *dev)
{
  dma_addr_t phys;
  char *virt;

  virt = dma_alloc_coherent(dev, 100, &phys, GFP_KERNEL);
  if (!virt)
    return -ENOMEM;
  memset(virt, 0, 100);
  my_HW_set_addr(dev, phys);
  my_HW_start_transfer(dev);
  while (my_HW_working(dev)) /* polling */
    msleep(100);
  printk(KERN_INFO "%s\n", virt);
  return 0;
}
  \end{lstlisting}

  \pause
  \bigskip

  Úkol: alokujte a uvolněte pomocí \code{dma_*} 1 stránku paměti (v ovladači
  EDU z předminula)

\end{frame}

\begin{frame}[fragile]
  \frametitle{DMA pro PCI}

  \heading{Navíc}
  \begin{itemize}
  \item Nastavení masky adres
  \begin{itemize}
  \item Ne všechna zařízení zvládnou adresovat celý fyzický prostor
  \item \code{pci_set_dma_mask(pdev, DMA_BIT_MASK(27))}
  \end{itemize}
  \pause
  \item Zapnutí ,,spravování sběrnice''
  \begin{itemize}
  \item Tj. zařízení umí samo iniciovat přenosy apod.
  \item \code{pci_set_master(pdev)}
  \end{itemize}
  \end{itemize}

  \pause

  \begin{lstlisting}[language=C]
int my_probe(struct pci_dev *pdev, ...)
{
  dma_addr_t phys;
  void *virt;
  ... /* enable etc. here */
  ret = pci_set_dma_mask(pdev, DMA_BIT_MASK(27));

  pci_set_master(pdev);

  virt = dma_alloc_coherent(&pdev->dev, 100, &phys, GFP_KERNEL);
  ...
}
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{EDU a DMA}

  \begin{itemize}
    \item Zvládá 32bitové adresy
    \item DMA řadič napojený na PCI a lokální paměť
    \begin{itemize}
      \item Na adrese 0x40000
      \item Velikost 4096~B
    \end{itemize}

    \item Generuje přerušení 8 (\code{0x100}) na kartě
    \begin{itemize}
      \item Po dokončení přenosu
      \item (Jen je-li vyžádáno před přenosem)
    \end{itemize}
  \end{itemize}

  \pause
  \bigskip

  Úkol: doplňte předešlou alokaci (o \code{set_dma_mask} a \code{set_master})

\end{frame}

\begin{frame}
  \frametitle{EDU DMA registry}

\begin{table}
\footnotesize
\begin{tabular}{|l|l|l|l|l|} \hline
Offset & Len & R/W & Contents & Meaning \\ \hline\hline
0x0080 & 4B  & R/W & src      & Source address \\ \hline
0x0084 & 4B  & R/W & dst      & Destination address \\ \hline
0x0088 & 4B  & R/W & count    & Transfer count \\ \hline
0x008c & 4B  & R/W & cmd      & Command register \\ \hline
\end{tabular}
\caption{Specifikace baru 0 (pokračování z minula)}
\end{table}

  \pause

  \begin{itemize}
    \item \code{cmd} registr
    \begin{itemize}
      \item Zapisuje se po nastavení ostatních registrů
      \item Zapisuje se jednou s kombinací bitů
      \item Bit 0: \code{RUN} (transfer now)
      \item Bit 1: \code{DIRECTION} (0: from PCI, 1: to PCI)
      \item Bit 2: \code{RQINT} (generate an interrupt after the transaction)
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s DMA na EDU}
  \begin{enumerate}
    \item Naalokovat stránku DMA prostoru
    \item 10~B inicializovat textovým řetězcem
    \item Přenést 10~B do EDU na adresu 0x40000
    \begin{itemize}
      \item Nastavit všechny 4 registry
      \item Bez přerušení (bez \code{RQINT})
      \item Počkat na nulovou hodnotu bitu \code{RUN} (na dokončení přenosu)
    \end{itemize}

    \item Přenést 10~B z EDU na DMA stránku + 10
    \begin{itemize}
      \item Vznikne za sebou 2$\times$ stejný řetězec
    \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{DMA s přerušením} (domácí)
  \begin{enumerate}
    \item Rozšířit předchozí o přerušení
    \begin{itemize}
      \item Přidat ještě jeden přenos z EDU na DMA + 20
      \item Přenos bude s \code{RQINT}
    \end{itemize}

    \item Obsloužit přerušení
    \begin{itemize}
      \item ACK přerušení
      \item Iniciovat tasklet
    \end{itemize}
    \item V taskletu vypsat obsah DMA paměti + 20
    \item DMA stránku vystavit přes \code{mmap}
  \end{enumerate}

  \bigskip

  \emph{POZOR} na prezenci a správnou posloupnost zabíjení taskletu, zákazu
  přerušení, dealokace atd.

\end{frame}
