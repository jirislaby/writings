\begin{frame}
  \frametitle{DMA}

  \heading{Přímý přístup do paměti}
  \begin{itemize}
  \item Prozatím jsme ze/do zařízení četli/zapisovali přes CPU
  \begin{itemize}
  \item Tj. standardními operacemi s (přemapovanou) pamětí
  \end{itemize}
  \item Velké přenosy = velká zátěž CPU
  \begin{itemize}
  \item Cykly, čekání na pomalou sběrnici atd.
  \end{itemize}
  \item Místo toho naprogramujeme HW, aby přenášel data sám
  \begin{itemize}
  \item Nutná podpora HW (sběrnice -- arbitrace, zařízení -- přenosy)
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Princip DMA}

  \heading{Princip}
  \begin{enumerate}
  \item Alokace (speciální) paměti
  \item Pro TX: vyplnění daty (paket, zvuk, data na disk, \ldots)
  \item Předání ukazatele do zařízení
  \item Odstartování přenosu v zařízení
  \item Přerušení či jiná signalizace konce přenosu od zařízení
  \item Pro RX: práce s příchozími daty (paket, data z disku, \ldots)
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{DMA v Linuxu}

  \heading{API alokace}
  \begin{itemize}
  \item \header{linux/dma-mapping.h}, \doc{Documentation/DMA-*}
  \item \texttt{dma\_alloc\_coherent}, \texttt{dma\_free\_coherent}
  \end{itemize}

  \pause
  \bigskip

  \texttt{\footnotesize void *dma\_alloc\_coherent(struct device *dev, size\_t size, dma\_addr\_t *dma\_handle, gfp\_t gfp)}
  \begin{itemize}
  \item \texttt{dev} -- zařízení, které bude k paměti přistupovat
    (\texttt{NULL} pokud neznáme)
  \item \texttt{size} -- velikost, jakou požadujeme
  \item \texttt{dma\_handle} -- fyzická adresa (návratová hodnota) -- pro
  zařízení
  \item návratová hodnota -- virtuální adresa -- pro nás
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{DMA v Linuxu -- příklad}

  \begin{lstlisting}[language=C]
int my_do_DMA(struct device *dev)
{
  dma_addr_t phys;
  void *virt;

  virt = dma_alloc_coherent(dev, 100, &phys, GFP_KERNEL);
  if (!virt)
    return -ENOMEM;
  memset(virt, 0, 100);
  my_HW_set_addr(dev, phys);
  my_HW_start_transfer(dev);
  while (my_HW_working(dev)) /* polling */
    msleep(100);
  printk(KERN_INFO "%s", virt);
  return 0;
}
  \end{lstlisting}

  \pause
  \bigskip

  Úkol: alokujte pomocí \texttt{dma\_*} 1 stránku paměti (pro dev =
  \texttt{NULL})

\end{frame}

\begin{frame}[fragile]
  \frametitle{DMA pro PCI}

  \heading{Navíc}
  \begin{itemize}
  \item Nastavení masky adres
  \begin{itemize}
  \item Ne všechna zařízení zvládnou adresovat celý fyzický prostor
  \item \texttt{pci\_set\_dma\_mask(pdev, DMA\_BIT\_MASK(27))}
  \end{itemize}
  \pause
  \item Zapnutí ,,spravování sběrnice''
  \begin{itemize}
  \item Tj. zařízení umí samo iniciovat přenosy apod.
  \item \texttt{pci\_set\_master(pdev)}
  \end{itemize}
  \end{itemize}

  \pause

  \begin{lstlisting}[language=C]
int my_probe(struct pci_dev *pdev, ...)
{
  dma_addr_t phys;
  void *virt;
  ... /* enable etc. here */
  ret = pci_set_dma_mask(pdev, DMA_BIT_MASK(27));

  pci_set_master(pdev);

  virt = dma_alloc_coherent(&pdev->dev, 100, &phys, GFP_KERNEL);
  ...
}
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Combo a DMA}

  \begin{itemize}
  \item Zvládá 32-bitové adresy
  \item DMA řadič napojený na LocalBus (1), PCI (2) a PowerPC (4)
  \item Generuje přerušení 8 na kartě
  \end{itemize}

\begin{frame}
  \frametitle{Combo DMA registry}

\begin{table}
\footnotesize
\begin{tabular}{|l|l|l|l|l|} \hline
Offset & Len & R/W & Contents & Meaning \\ \hline\hline
0x0080 & 4B  & R/W & src      & Source address \\ \hline
0x0084 & 4B  & R/W & dst      & Destination address \\ \hline
0x0088 & 4B  & R/W & count    & Transfer count \\ \hline
0x008c & 4B  & R/W & cmd      & Command register \\ \hline
\end{tabular}
\caption{Specifikace baru 0 (pokračování z minula)}
\end{table}

  \begin{itemize}
  \item \texttt{cmd} registr
  \begin{itemize}
  \item Bit 0: \texttt{RUN} (transfer now)
  \item Bits 1-3: \texttt{SOURCE BUS}
  \item Bits 4-6: \texttt{DESTINATION BUS}
  \item Bit 7: \texttt{NOINT} (do not generate interrupt after transaction)
  \item Bit 31: \texttt{ACKINT} (acknowledge interrupt)
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s DMA}
  \begin{enumerate}
  \item Naalokovat stránku DMA prostoru
  \item 10B inicializovat textovým řetězcem
  \item Přenést 10B na PowerPC sběrnici na adresu 0x40000
  \begin{itemize}
  \item Nastavit všechny 4 registry
  \item Bez přerušení (nastavit \texttt{NOINT})
  \item Počkat na nulovou hodnotu bitu \texttt{RUN} (na dokončení přenosu)
  \end{itemize}
  \item Přenést 10B z PowerPC na DMA stránku + 10
  \begin{itemize}
  \item Vznikne za sebou 2$\times$ stejný řetězec
  \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{DMA s přerušením} (domácí)
  \begin{enumerate}
  \item Rozšířit předchozí o přerušení
  \begin{itemize}
  \item Přidat ještě jeden přenos z PowerPC na DMA + 20
  \item Přenos bude bez \texttt{NOINT}
  \item Je třeba povolit 8. (0x100) přerušení na kartě
  \end{itemize}
  \item Obsloužit přerušení
  \begin{itemize}
  \item ACK přerušení a DMA
  \item Iniciovat tasklet
  \end{itemize}
  \item V taskletu vypsat obsah DMA paměti + 20
  \item DMA stránku vystavit přes mmap
  \end{enumerate}

  \bigskip

  Pozn.: pokud se zařízení zasekne (např. špatně nastaveným DMA), je nutné
  stroj vypnout a zapnout, reboot nepomůže

\end{frame}
