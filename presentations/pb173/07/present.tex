\begin{frame}
  \frametitle{I/O -- porty}

  \heading{Porty}
  \begin{itemize}
  \item Speciální adresový prostor
  \begin{itemize}
  \item Závislé na architektuře
  \item Speciální instrukce (\code{in}, \code{out} na x86)
  \end{itemize}
  \item Samostatná (malá) sběrnice
  \begin{itemize}
  \item Na x86: řadič klávesnice, PC spkr, staré časovače a ovladače přerušení,
    ladicí port (0x80 $\Rightarrow$ segmentový displej na desce), \ldots
  \end{itemize}
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
  \item \header{linux/io.h}, \header{linux/ioport.h}
  \item Vytvoření: \code{request_region}, \code{release_region}
  \item R/W: \code{inX(port)}, \code{outX(co, port)}, kde
    $\text{\code{X}}\in\left\{\text{\code{b}},\text{\code{w}},\text{\code{l}}\right\}$
  \item Vícenásobné R/W: \code{insX}, \code{outsX}
  \end{itemize}

 \begin{itemize}

  \heading{Hledání zařízení}
  \begin{enumerate}
  \item Iterátory (starší)
  \begin{itemize}
  \item Vícenásobný přístup k zařízení
  \item Nepodporuje hotplug
  \item \code{pci_get_device} (vendor, device)
  \item Reference: \code{pci_dev_get}, \code{pci_dev_put}
  \end{itemize}
  \end{enumerate}

  \begin{lstlisting}[language=C]
struct pci_dev *pdev = NULL;
while ((pdev = pci_get_device(VENDOR, DEVICE, pdev))) {
  printk("%2.x:%.2x.%.2x\n", pdev->bus->number, PCI_SLOT(pdev->devfn), PCI_FUNC(pdev->devfn));
}
  \end{lstlisting}

\begin{frame}[fragile]
  \frametitle{PCI zařízení v jádře II.}

  \begin{enumerate}
  \setcounter{enumi}{1}
  \item Událostmi
  \begin{itemize}
  \item Registrace seznamu chtěných zařízení a háčků
  \item Seznam: \code{struct pci_device_id} (vendor, device, atd.)
  \item Háčky: \code{struct pci_driver} (probe, remove, suspend, atd.)
  \item \code{pci_register_driver}, \code{pci_unregister_driver}
  \end{itemize}
  \end{enumerate}

  \begin{lstlisting}[language=C]
struct pci_device_id my_table[] = {
  { PCI_DEVICE(VENDOR1, DEVICE1) }, { PCI_DEVICE(VENDOR1, DEVICE2) },
  { PCI_DEVICE(0x8086, PCI_ANY_ID), .driver_data = 1 },
  { 0, }
};
MODULE_DEVICE_TABLE(pci, my_table);

struct pci_driver my_pci_driver = {
  .name = "my_driver",     .id_table = my_table,
  .probe = my_probe,       .remove = my_remove,
};

int my_probe(struct pci_dev *pdev, const struct pci_device_id *id)
{
  printk("%2.x %lu\n", pdev->bus->number, id->driver_data);
  return 0;
}
  \end{lstlisting}

\end{frame}
