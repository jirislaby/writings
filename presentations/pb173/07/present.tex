  \heading{Memory Mapped I/O}
  \begin{itemize}
  \item Součástí fyzického adresového prostoru
  \item Přístup standardním čtením/zápisem
  \begin{itemize}
  \item Nutnost přemapovat na virtuální adresy
  \end{itemize}
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
  \item \header{linux/io.h}, \header{linux/ioport.h}
  \item Vyloučení: \texttt{request\_mem\_region}, \texttt{release\_mem\_region}
  \begin{itemize}
  \item \texttt{/proc/iomem}
  \end{itemize}
  \item Mapování: \texttt{virt=ioremap(phys)}, \texttt{iounmap(virt)}
  \begin{itemize}
  \item \texttt{/proc/vmallocinfo} (novější jádra)
  \end{itemize}
  \item R/W: \texttt{readX(odkud)}, \texttt{writeX(co, kam)}, kde
    $\text{\texttt{X}}\in\left\{\text{\texttt{b}},\text{\texttt{w}},\text{\texttt{l}},\text{\texttt{q}}\right\}$
  \item Vícenásobné R/W: \texttt{memcpy\_fromio}, \texttt{memcpy\_toio}

\begin{frame}

  \heading{Porty}
  \begin{itemize}
  \item Speciální adresový prostor
  \begin{itemize}
  \item Závislé na architektuře

    $\text{\texttt{X}}\in\left\{\text{\texttt{b}},\text{\texttt{w}},\text{\texttt{l}}\right\}$
  \item Vícenásobné R/W: \texttt{insX}, \texttt{outsX}

\begin{frame}[fragile]
  \frametitle{PCI zařízení v jádře I.}

  \begin{itemize}
  \item \header{linux/pci.h}, \doc{Documentation/pci/*}
  \item \emph{\texttt{struct pci\_dev}}, \texttt{struct pci\_bus}
  \item \texttt{PCI\_ANY\_ID} značí jakékoliv ID
  \end{itemize}

  \medskip

  \heading{Hledání zařízení}
  \begin{enumerate}
  \item Iterátory (starší)
  \begin{itemize}
  \item Vícenásobný přístup k zařízení
  \item Nepodporuje hotplug
  \item \texttt{pci\_get\_device} (vendor, device)
  \item Reference: \texttt{pci\_dev\_get}, \texttt{pci\_dev\_put}
  \end{itemize}
  \end{enumerate}

  \begin{lstlisting}[language=C]
struct pci_dev *pdev = NULL;
while ((pdev = pci_get_device(VENDOR, DEVICE, pdev))) {
  printk("%2.x:%.2x.%.2x\n", pdev->bus->number, PCI_SLOT(pdev->devfn), PCI_FUNC(pdev->devfn));
}
  \end{lstlisting}

  Úkol: vypsat všechna zařízení v systému (jejich ID)

\end{frame}

\begin{frame}[fragile]
  \frametitle{PCI zařízení v jádře II.}

  \begin{enumerate}
  \setcounter{enumi}{1}
  \item Událostmi
  \begin{itemize}
  \item Registrace háčků a seznamu chtěných zařízení
  \item Seznam: \texttt{struct pci\_device\_id} (vendor, device, atd.)
  \item Háčky: \texttt{struct pci\_driver} (probe, remove, suspend, atd.)
  \item \texttt{pci\_register\_driver}, \texttt{pci\_unregister\_driver}
  \end{itemize}
  \end{enumerate}

  \begin{lstlisting}[language=C]
struct pci_device_id my_table[] = {
  { PCI_DEVICE(VENDOR1, DEVICE1) }, { PCI_DEVICE(VENDOR1, DEVICE2) },
  { PCI_DEVICE(0x8086, PCI_ANY_ID), .driver_data = 1 },
  { 0, }
};
MODULE_DEVICE_TABLE(pci, my_table);

struct pci_driver my_pci_driver = {
  .name = "my_driver",     .id_table = my_table,
  .probe = my_probe,       .remove = my_remove,
};

int my_probe(struct pci_dev *pdev, const struct pci_device_id *id)
{
  printk("%2.x %lu\n", pdev->bus->number, id->driver_data);
  return 0;
}
  \end{lstlisting}

\end{frame}
