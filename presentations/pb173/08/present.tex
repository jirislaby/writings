\begin{frame}[fragile]
  \frametitle{Obyčejné časovače}

  \heading{API}
  \begin{itemize}
  \item \header{linux/timer.h}, \texttt{struct timer\_list}
  \item \texttt{DEFINE\_TIMER}$_S$, \texttt{setup\_timer}$_D$
  \item \texttt{mod\_timer} (=\texttt{del\_timer}+\texttt{add\_timer})
  \item \texttt{del\_timer\_sync}
  \end{itemize}

  \begin{lstlisting}[language=C]
static void my_fun(unsigned long data);
static DEFINE_TIMER(my_timer, my_fun, 0, 30);
static void my_fun(unsigned long data)
{
  my_timer.data *= 2;
  mod_timer(&my_timer, jiffies + msecs_to_jiffies(data));
}
...
  mod_timer(&my_timer, jiffies + msecs_to_jiffies(100));
...
  del_timer_sync(&my_timer);
  \end{lstlisting}

\begin{frame}
  \frametitle{Workqueue}

  \begin{itemize}
  \item Speciální \emph{proces} volající funkce ve frontě
  \begin{itemize}
    \item Globální, společný pro všechny -- většinou stačí
    \item Vlastní -- pro speciální případy
  \end{itemize}
  \item Lze v něm spát
  \item Spustí se, až se plánovač rozhodne
  \begin{itemize}
    \item Lze specifikovat minimální prodlevu
  \end{itemize}
  \item \doc{Documentation/workqueue.txt}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Workqueue API}

  \begin{itemize}
  \item \header{linux/workqueue.h}, \texttt{struct work\_struct}
  \item Definice funkce (práce): \texttt{DECLARE\_WORK}$_S$,
    \texttt{INIT\_WORK}$_D$
  \item Globální proces:
  \begin{itemize}
  \item Zařazení do fronty: \texttt{schedule\_work}
  \item Vyřazení z fronty (předčasně): \texttt{cancel\_work\_sync}
  \end{itemize}
  \item Vlastní proces:
  \begin{itemize}
  \item Vytvoření procesu: \texttt{create\_workqueue},
    \texttt{destroy\_workqueue}
  \item Zařazení do fronty: \texttt{queue\_work}
  \item Vyřazení z fronty: \texttt{cancel\_work\_sync} (stejně)
  \end{itemize}
  \item Delayed (s min. prodlevou): \texttt{*\_delayed\_work}
  \end{itemize}

  \pause
  \medskip

  Úkol: nahrát modul \texttt{ieee80211} (\texttt{request\_module}). Nelze
  volat přímo v \texttt{module\_init} a je nutný i vlastní proces

\end{frame}

\begin{frame}
  \frametitle{Tasklet}

  \begin{itemize}
  \item Běží v kontextu přerušení (jako časovače)
  \item Nelze spát
  \item Musí být rychlý
  \item Spustí se \emph{po} příštím přerušení
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
  \item \header{linux/interrupt.h}, \texttt{struct tasklet\_struct}
  \item Definice funkce (taskletu): \texttt{DECLARE\_TASKLET}$_S$,
    \texttt{tasklet\_init}$_D$
  \item Zařazení do fronty: \texttt{tasklet\_schedule}
  \item Vyřazení z fronty: \texttt{tasklet\_kill}
  \end{itemize}

  \pause
  \medskip

  Úkol: vytvořit tasklet, spustit a vypsat posloupnost volání pomocí
  \texttt{dump\_stack}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Přerušení v Linuxu}

  \begin{itemize}
  \item \header{linux/interrupt.h}
  \item \texttt{request\_irq}, \texttt{free\_irq}
  \item Flags: hlavně \texttt{IRQF\_SHARED}
  \item Návratová hodnota z obsluhy: \texttt{IRQ\_NONE}, \texttt{IRQ\_HANDLED}
  \item \texttt{/proc/interrupts}
  \end{itemize}

  \begin{lstlisting}[language=C]
static irqreturn_t my_handler(int irq, void *data)
{ /* data == my_data */
  return my_device_raised_interrupt ? IRQ_HANDLED : IRQ_NONE;
}
static int my_probe(struct pci_dev *pdev, ...)
{
  /* here: enable device etc. */
  my_data = kmalloc(...);
  ret = request_irq(pdev->irq, my_handler, IRQF_SHARED, "my", my_data);
}
  \end{lstlisting}

  \pause

  Úkol: vytvořte obsluhu pro Combo (zatím vracejte \texttt{IRQ\_HANDLED})

\end{frame}

\begin{frame}
  \frametitle{Kontext přerušení}

  \heading{Změny}
  \begin{itemize}
  \item Zámky
  \begin{itemize}
  \item Třeba používat \texttt{*\_irqflags} varianty
  \item Jinak by mohlo dojít k deadlocku
  \end{itemize}
  \item Alokace
  \begin{itemize}
  \item \texttt{GFP\_ATOMIC}
  \item A tedy nevelké alokace
  \end{itemize}
  \item Kód
  \begin{itemize}
  \item Rychlý, krátký
  \item (Víc kódu později v taskletu, ještě víc ve workqueue)
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s přerušením} (součást domácího)
  \begin{enumerate}
  \item Vytvořit pojmenování (makra) pro registry (v tabulce)
  \item Spustit 100\,ms periodický časovač (pro každé Combo zařízení)
  \begin{itemize}
  \item Vyvolat z Comba přerušení 0x1000 (registr 0x60)
  \end{itemize}
  \item Navázat se v probe na prerušení (\texttt{request\_irq})
  \item Povolit v Combu přerušení 0x1000 (registr 0x44)
  \item Implementovat obsluhu
  \begin{itemize}
  \item Přečíst stav (registr 0x40), vypsat stav -- limitovaně
  \item Nenula: odsouhlasit přerušení (registr 0x64), vrátit
    \texttt{IRQ\_HANDLED}
  \item Jinak: vrátit \texttt{IRQ\_NONE}
  \end{itemize}
  \end{enumerate}
