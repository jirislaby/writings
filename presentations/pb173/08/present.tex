\begin{frame}
  \frametitle{Časovače}

  \begin{itemize}
  \item \doc{Documentation/timers/*}
  \item Činnost v nastavených časech (i periodicky)
  \item Kód je volán v kontextu přerušení (nelze spát)
  \item \file{/proc/timer_list} (jen nová jádra)
  \end{itemize}

  \pause

  \heading{Obyčejné časovače}
  \begin{itemize}
    \item Fungují na každé architektuře
    \item Založené na \code{jiffies}
    \begin{itemize}
      \item Proměnná zvyšující se o 1 s frekvencí \code{HZ}
    \end{itemize}
    \item Rozlišení: 1--10 ms (pro \code{HZ}: 1000--100)
  \end{itemize}

  \pause

  \heading{High-res časovače}
  \begin{itemize}
    \item \header{linux/hrtimer.h}
    \item \code{usleep_range}
    \item Rozlišení: $\sim \mu$s
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Obyčejné časovače}

  \heading{API}
  \begin{itemize}
  \item \header{linux/timer.h}, \code{struct timer_list}
  \item \code{DEFINE_TIMER}$_S$, \code{setup_timer}$_D$
  \item \code{mod_timer} (=\code{del_timer}+\code{add_timer})
  \item \code{del_timer_sync} (\emph{POZOR})
  \end{itemize}

  \begin{lstlisting}[language=C]
static void my_fun(unsigned long data);
static DEFINE_TIMER(my_timer, my_fun, 0, 30);
static void my_fun(unsigned long data)
{
  my_timer.data *= 2;
  mod_timer(&my_timer, jiffies + msecs_to_jiffies(data));
}
...
  mod_timer(&my_timer, jiffies + msecs_to_jiffies(100));
...
  del_timer_sync(&my_timer);
  \end{lstlisting}

  \pause
  \medskip

  Úkol: Každou vteřinu, kdy je modul v systému, vypsat \code{HZ} a
  \code{jiffies}

\end{frame}

\begin{frame}
  \frametitle{Workqueue}

  \begin{itemize}
  \item Speciální \emph{proces} volající funkce ve frontě
  \begin{itemize}
    \item Globální, společný pro všechny -- většinou stačí
    \item Vlastní -- pro speciální případy
  \end{itemize}
  \item Lze v něm spát
  \item Spustí se, až se plánovač rozhodne
  \begin{itemize}
    \item Lze specifikovat minimální prodlevu
  \end{itemize}
  \item \doc{Documentation/workqueue.txt}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Workqueue API}

  \begin{itemize}
  \item \header{linux/workqueue.h}, \code{struct work_struct}
  \item Definice funkce (práce): \code{DECLARE_WORK}$_S$, \code{INIT_WORK}$_D$
  \item Globální proces:
  \begin{itemize}
  \item Zařazení do fronty: \code{schedule_work}
  \item Vyřazení z fronty (předčasně): \code{cancel_work_sync}
  \end{itemize}
  \item Vlastní proces:
  \begin{itemize}
  \item Vytvoření procesu: \code{create_workqueue},
    \code{destroy_workqueue}
  \item Zařazení do fronty: \code{queue_work}
  \item Vyřazení z fronty: \code{cancel_work_sync} (stejně)
  \end{itemize}
  \item Delayed (s min. prodlevou): \code{*_delayed_work}
  \end{itemize}

  \pause
  \medskip

  Úkol: nahrát modul \code{ieee80211} (\code{request_module}). Nelze volat
  přímo v \code{module_init} a je nutný i vlastní proces

\end{frame}

\begin{frame}
  \frametitle{Tasklet}

  \begin{itemize}
  \item Běží v kontextu přerušení (jako časovače)
  \item Nelze spát
  \item Musí být rychlý
  \item Spustí se \emph{po} příštím přerušení
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
  \item \header{linux/interrupt.h}, \code{struct tasklet_struct}
  \item Definice funkce (taskletu): \code{DECLARE_TASKLET}$_S$,
  \code{tasklet_init}$_D$
  \item Zařazení do fronty: \code{tasklet_schedule}
  \item Vyřazení z fronty: \code{tasklet_kill}
  \end{itemize}

  \pause
  \medskip

  Úkol: vytvořit tasklet, spustit a vypsat posloupnost volání pomocí
  \code{dump_stack}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Přerušení v Linuxu}

  \begin{itemize}
  \item \header{linux/interrupt.h}
  \item \code{request_irq}, \code{free_irq}
  \item Flags: hlavně \code{IRQF_SHARED}
  \item Návratová hodnota z obsluhy: \code{IRQ_NONE}, \code{IRQ_HANDLED}
  \item \file{/proc/interrupts}
  \end{itemize}

  \begin{lstlisting}
static irqreturn_t my_handler(int irq, void *data)
{ /* data == my_data */
  return my_device_raised_interrupt ? IRQ_HANDLED : IRQ_NONE;
}
static int my_probe(struct pci_dev *pdev, ...)
{
  /* here: enable device etc. */
  my_data = kmalloc(...);
  ret = request_irq(pdev->irq, my_handler, IRQF_SHARED, "my", my_data);
}
  \end{lstlisting}

  \pause

  Úkol: vytvořte obsluhu pro EDU (zatím vracejte \code{IRQ_HANDLED})

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Práce s přerušením} (součást domácího)
  \begin{enumerate}
    \item Vytvořit pojmenování (makra) pro registry (v tabulce)
    \item Spustit 100ms periodický časovač (pro každé EDU zařízení)
    \begin{itemize}
      \item Vyvolat z EDU přerušení 0x1000 (registr 0x60)
    \end{itemize}
    \item Navázat se v probe na prerušení (\code{request_irq})
    \item Implementovat obsluhu
    \begin{itemize}
      \item Přečíst stav (registr 0x24), vypsat stav -- limitovaně
      \item Nenula: odsouhlasit přerušení (registr 0x64), vrátit
	\code{IRQ_HANDLED}
      \item Jinak: vrátit \code{IRQ_NONE}
    \end{itemize}
  \end{enumerate}
