\begin{frame}[fragile]
  \frametitle{Příklad operací}
  \begin{lstlisting}[language=C]
struct file_operations my_fops = {
  .owner = THIS_MODULE,
  .open = my_open,
  .write = my_write,
  .release = my_release,
};

int my_open(struct inode *inode, struct file *filp) {
  filp->private_data = __something;
  return 0;
}

ssize_t my_write(struct file *filp, const char __user *buf, size_t count, loff_t *offp) {
  /* filp->private_data is __something here */
  if (count == 0)
    return -EINVAL;
  return 0;
}
  \end{lstlisting}

\end{frame}

  \end{itemize}

  \begin{center}
  \begin{tabular}{|c|c|c|} \hline
  Uživatel (32-bit) & & Jádro (64-bit) \\ \hline

struct my {
  void *data;
};
  \end{lstlisting} &

\begin{frame}[fragile]
  \frametitle{Pevné typy v jádře}
  \framesubtitle{z hlediska počtu bitů}

  \begin{itemize}
  \item \header{linux/types.h}
  \item \texttt{\_\_u8}, \texttt{\_\_u16}, \texttt{\_\_u32}, \texttt{\_\_u64}
  \item \texttt{\_\_s8}, \texttt{\_\_s16}, \texttt{\_\_s32}, \texttt{\_\_s64}
  \item Ukazatele musí být v union s \texttt{\_\_u64}
  \end{itemize}

  \begin{center}
  \begin{tabular}{ccc}
  \begin{lstlisting}[language=C]
struct my {
  unsigned long flags;
  short index;
  void *data;
} my;
  \end{lstlisting} & $\Rightarrow$ &
  \begin{lstlisting}[language=C]
struct my {
  __u64 flags;
  __s16 index;
  union {
    void *data;
    __u64 filler;
  };
} my;
  \end{lstlisting}
  \end{tabular}
  \end{center}
  \begin{lstlisting}[language=C]
read(fd, &my, sizeof(my));
ioctl(fd, DO_SOMETHING, &my);
  \end{lstlisting}
