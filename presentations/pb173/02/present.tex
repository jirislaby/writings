\begin{frame}

  \begin{enumerate}
  \item Voláním funkce: system call (syscall)
  \begin{itemize}
  \item V jádře: tabulka číslo-funkce (\texttt{unistd.h})
  \item \texttt{syscall(\_\_NR\_write, STDOUT\_FILENO, "Test\textbackslash n",
    5);}
  \item Skok do jádra speciální instrukcí (x86\_64: \texttt{syscall})
    \begin{itemize}
    \item O skok se stará libc (\texttt{fwrite}$\rightarrow$\texttt{write}$\rightarrow$\texttt{syscall}$\rightarrow$instrukce)
    \item Drahá operace (přepnutí kontextu)
    \end{itemize}
  \item Demo: lxr $\rightarrow$ \texttt{\_\_NR\_write}
  \end{itemize}

  \pause

  \item Speciální syscall
  \begin{itemize}
  \item Bez skoku do jádra, v knihovně linkované jádrem (\texttt{vdso.so})
  \item Několik málo jednoduchých funkcí (\texttt{gettimeofday}, \ldots)
  \item Demo: \texttt{ldd /bin/cat}
  \end{itemize}
  \end{enumerate}

  \pause
  \medskip

  \begin{center}
  \heading{Každá nová funkcionalita = nový syscall}
  \end{center}

\end{frame}

\begin{frame}
  \frametitle{Úkol}
  Pomocí volání \texttt{syscall} spustit \texttt{/bin/date} a počkat na
  skončení. Předtím a potom vypsat nějakou informaci.

  \medskip

  \begin{enumerate}
  \item \texttt{syscall(\_\_NR\_write, \ldots, "Running date", \ldots)}
  \item \texttt{syscall(\_\_NR\_fork)}
    \begin{itemize}
    \item Potomek: \texttt{syscall(\_\_NR\_execve, "/bin/date", \ldots)}
    \end{itemize}
  \item \texttt{syscall(\_\_NR\_wait4, \ldots)}
  \item \texttt{syscall(\_\_NR\_write, \ldots, "date finished", \ldots)}
  \end{enumerate}

  \medskip

  Je třeba znát prototypy funkcí (\texttt{man write}, \texttt{man fork},
  \ldots).

\end{frame}

\begin{frame}[fragile]
  \frametitle{Datové typy}
  \begin{itemize}
  \item Jádro a proces může běžet s různými bitovými šířkami (32, 64-bit)
  \item Problém s ukazateli a long proměnnými
  \begin{itemize}
  \item Jiná délka dat
  \item Jiné zarovnání struktur
  \end{itemize}

  \pause

  \item Jádro definuje pevné typy (z hlediska počtu bitů)
  \begin{itemize}
  \item \texttt{linux/types.h}
  \item \texttt{\_\_u8}, \texttt{\_\_u16}, \texttt{\_\_u32}, \texttt{\_\_u64}
  \item \texttt{\_\_s8}, \texttt{\_\_s16}, \texttt{\_\_s32}, \texttt{\_\_s64}
  \item Ukazatele musí být v union s \texttt{\_\_u64}
  \end{itemize}
  \end{itemize}

  \begin{lstlisting}[language=C]
struct my {
  __u64 flags;              unsigned long flags;
  __s16 index; <instead of> short index;
  union {                   void *data;
    void *data;
    __u64 filler;
  };
} my;
read(fd, &my, sizeof(my));
ioctl(fd, DO_SOMETHING, &my);
  \end{lstlisting}

\end{frame}
