\part{Komunikace mezi procesy}

\frame{\partpage}

\begin{frame}
  \frametitle{Komunikace mezi procesy}

  \heading{LDD3 část kap. 6 (zastaralá)}

  \bigskip

  \begin{itemize}
  \item Čekání na událost
  \item Buzení procesů
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Jednoduchá komunikace}

  \begin{itemize}
  \item 2 procesy (producent-konzument)
  \begin{itemize}
  \item A: čeká na nějakou hotovou práci
  \item B: udělá nějakou práci a oznámí A dokončení
  \item A: pokračuje
  \end{itemize}
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
  \item \header{linux/completion.h}, \texttt{struct completion}
  \item \texttt{DECLARE\_COMPLETION}$_S$, \texttt{init\_completion}$_D$
  \item Čekání: \texttt{wait\_for\_completion},
    \texttt{wait\_for\_completion\_interruptible} (retval)
  \item Dokončení: \texttt{complete}, \texttt{complete\_all}
  \end{itemize}

  \begin{lstlisting}[language=C]
static DECLARE_COMPLETION(my_comp);
static char *str;
...
A                                        B
wait_for_completion(&my_comp);           strcpy(str, "Ahoj");
printk(KERN_DEBUG "%s\n", str);          complete(&my_comp);
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Použití \texttt{completion}}
  \begin{enumerate}
  \item Doplnit do pb173/11/events \texttt{completion}
  \item Číst se bude, až někdo dokončí zápis
  \begin{itemize}
  \item Čekání v \texttt{read} je možné přerušit signálem
    (\texttt{wait\_for\_completion\_interruptible})
  \end{itemize}
  \item Spustit 2 instance \texttt{cat /dev/my\_name}
  \item Pozorovat, co se děje
  \item Spustit nekolikrát \texttt{echo XXX > /dev/my\_name}
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Složitější komunikace}

  \begin{itemize}
  \item Čekání na více místech na jinou událost
  \item 3 procesy
  \begin{itemize}
  \item A: plní buffer
  \item B: čeká na 100B
  \item C: čeká na 200B
  \end{itemize}
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
  \item \header{linux/wait.h}, \texttt{wait\_queue\_head\_t}
  \item \texttt{DECLARE\_WAIT\_QUEUE\_HEAD}$_S$,
    \texttt{init\_waitqueue\_head}$_D$
  \item Čekání: \texttt{wait\_event}, \texttt{wait\_event\_interruptible}
  \item Dokončení: \texttt{wake\_up}, \texttt{wake\_up\_all}
  \item \texttt{completion} je obalená \texttt{wait\_queue}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Složitější komunikace -- příklad}

  \begin{lstlisting}[language=C]
static DECLARE_WAIT_QUEUE_HEAD(my_wait);
static atomic_t my_cnt = ATOMIC_INIT(0);
static char *str;
...
A
  while (1) {
    str[atomic_inc_return(&my_cnt) - 1] = 'A';
    wake_up_all(&my_wait);
    msleep(10);
  }

B
  wait_event(my_wait, atomic_read(&my_cnt) > 100);
  printk(KERN_DEBUG "%s\n", str);

C
  if (wait_event_interruptible(my_wait, atomic_read(&my_cnt) > 200)) {
    printk(KERN_WARNING "interrupted\n");
    return;
  }
  printk(KERN_DEBUG "%s\n", str);
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Použití \texttt{wait\_queue}}
  \begin{enumerate}
  \item Místo \texttt{completion}, použít v pb173/11/events \texttt{wait\_queue}
  \item Číst se bude, až někdo zápíše více než 5 znaků
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Scheduler}

  \begin{itemize}
  \item Celé je to instruování plánovače
  \item Ve skutečnosti je \texttt{wait\_event} a \texttt{wake\_up}:
  \begin{itemize}
  \item Nastavení stavu procesu: \texttt{set\_current\_state}
  \item Uspání procesu: \texttt{schedule}
  \item Probuzení procesu: \texttt{wake\_up\_process}
  \item Popř. s obsluhou signálů: \texttt{signal\_pending}
  \item \header{linux/sched.h}
  \end{itemize}
  \pause
  \item \texttt{completion} a \texttt{wait\_queue} ulehčuje práci
  \begin{itemize}
  \item Pamatováním si, koho vzbudit
  \item Případnou kontrolou signálů
  \item Případnou kontrolou vypršení timeoutu
  \item \texttt{completion} navíc řeší předání čítače (kolik procesů vzbudit)
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Hackovací techniky}

  \heading{Z pohledu jádra}
  \begin{itemize}

  \item Zneužít stroj
  \end{itemize}

\begin{frame}
  \frametitle{Zabít stroj}

  \begin{itemize}
  \item Jednodušší
  \begin{itemize}
  \item Stačí provést něco zlého
  \begin{itemize}
  \item Předat špatný ukazatel
  \item Vytvořit příliš procesů
  \item Naalokovat hodně paměti
  \end{itemize}
  \item Jádro udělá dereferenci a spadne
  \end{itemize}
  \item Nejčastější druhy
  \begin{itemize}
  \item DDoS
  \item Paket smrti
  \item Neočekávaný vstup od uživatele
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Zneužít stroj}

  \begin{itemize}
  \item K získání administrátorských oprávnění
  \item Složitější
  \begin{itemize}
  \item Musí se předat ,,divný'' ukazatel
  \item Jádro udělá dereferenci a spustí náš kód
  \end{itemize}
  \item Obtížná část:
  \begin{itemize}
  \item Zjistit, jak se dá díra zneužít
  \item Vyrobit správná data, která podstrčit
  \item Nepřepsat toho příliš mnoho, aby nedošlo k pádu
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Co s root účtem}

  \begin{itemize}
  \item Vyrobit si e-mail relay pro spamy
  \item Upravit jakoukoliv službu
  \item Ukrást peníze/vědomosti (vláda, armáda)
  \pause
  \item Skrýt se v systému (rootkit) a sbírat data
  \begin{itemize}
  \item Změnit tabulku systémových volání
    \begin{itemize}
    \item \texttt{ls /proc} a \texttt{ps} neuvidí proces
    \item Napsat vlastní readdir s patřičným filtrem
    \item Najít tabulku např. přes /proc/kallsyms
    \item Změnit tabulku a volat původní readdir
    \end{itemize}
  \pause
  \item Podobně pro přerušení
    \begin{itemize}
    \item Stisky klávesnice (hesla) apod.
    \end{itemize}
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Buffer overflow příklad}
  \begin{enumerate}
  \item Prostudujte pb173/11/hack
  \begin{itemize}
  \item Uživatelský proces: user
  \item Jaderný modul: kern
  \end{itemize}
  \item Přeložit a vložit modul
  \item Nastavit práva /dev/my\_name na 666
  \item Pod běžným uživatelem spustit pb173:
  \begin{itemize}
  \item \texttt{pb173 0 /dev/my\_name} (nezneužít díru)
  \item \texttt{pb173 1 /dev/my\_name} (zneužít díru)
  \end{itemize}
  \item Opravit díru ve \texttt{write}
  \begin{itemize}
  \item \texttt{pb173 1 /dev/my\_name} nesmí skončit s úspěchem
  \end{itemize}
  \end{enumerate}

\end{frame}
