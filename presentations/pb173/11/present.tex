\subsection{Completion}

\begin{frame}[fragile]
  \frametitle{Jednoduchá komunikace}

  \begin{itemize}
    \item 2 procesy (producent-konzument)
    \begin{itemize}
      \item A: čeká na nějakou hotovou práci
      \item B: udělá nějakou práci a oznámí A dokončení
      \item A: pokračuje
    \end{itemize}
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
    \item \header{linux/completion.h}, \code{struct completion}
    \item \code{DECLARE_COMPLETION}$_S$, \code{init_completion}$_D$
    \item Čekání: \code{wait_for_completion},
      \code{wait_for_completion_interruptible} (retval)
    \item Dokončení: \code{complete}, \code{complete_all}
  \end{itemize}

  \begin{lstlisting}
static DECLARE_COMPLETION(my_comp);
static char *str;
...
A                                        B
wait_for_completion(&my_comp);           strcpy(str, "Ahoj");
pr_info("%s\n", str);                    complete(&my_comp);
  \end{lstlisting} %stop_zone

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Použití \code{completion}}
  \begin{enumerate}
    \item Doplnit do \file{pb173/11/events} \code{completion}
    \item Číst se bude, až někdo dokončí zápis
    \begin{itemize}
      \item Čekání v \code{read} bude možné přerušit signálem
	(\code{wait_for_completion_interruptible})
    \end{itemize}

    \item Spustit 2 instance \cmd{cat /dev/my_name}
    \item Spustit nekolikrát \cmd{echo XXX > /dev/my_name}

    \item Pozorovat, co se děje při použití
    \begin{itemize}
      \item \code{complete}
      \item \code{complete_all}
    \end{itemize}
  \end{enumerate}

\end{frame}

\subsection{Wait event a ostatní}

\begin{frame}
  \frametitle{Složitější komunikace}

  \begin{itemize}
    \item Čekání na více místech na jinou událost
    \item 3 procesy
    \begin{itemize}
      \item A: plní buffer
      \item B: čeká na 100\,B
      \item C: čeká na 200\,B
    \end{itemize}
  \end{itemize}

  \pause

  \heading{API}
  \begin{itemize}
    \item \header{linux/wait.h}, \code{wait_queue_head_t}
    \item \code{DECLARE_WAIT_QUEUE_HEAD}$_S$,
      \code{init_waitqueue_head}$_D$
    \item Čekání: \code{wait_event}, \code{wait_event_interruptible}
    \item Dokončení: \code{wake_up}, \code{wake_up_all}

    \pause

    \item \code{completion} je obalená \code{wait_queue}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]{Složitější komunikace}{Příklad}

  \begin{lstlisting}
static DECLARE_WAIT_QUEUE_HEAD(my_wait);
static atomic_t my_cnt = ATOMIC_INIT(0);
static char *str;
...
A
  while (1) {
    str[atomic_inc_return(&my_cnt) - 1] = 'A';
    wake_up_all(&my_wait);
    msleep(10);
  }
  \end{lstlisting}

  \pause

  \begin{lstlisting}
B
  wait_event(my_wait, atomic_read(&my_cnt) > 100);
  pr_info("%s\n", str);
  \end{lstlisting} %stop_zone

  \pause

  \begin{lstlisting}
C
  if (wait_event_interruptible(my_wait, atomic_read(&my_cnt) > 200)) {
    pr_info("interrupted\n");
    return;
  }
  pr_info("%s\n", str);
  \end{lstlisting} %stop_zone

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Použití \code{wait_queue}}
  \begin{enumerate}
    \item Namísto \code{completion}, použít v \file{pb173/11/events}
      \code{wait_queue}
    \item Číst se bude, až bude do bufferu zapsáno více než 5 znaků
  \end{enumerate}

\end{frame}

\subsection{Scheduler a probouzení procesů}

\begin{frame}
  \frametitle{Scheduler}

  \begin{itemize}
    \item Celé je to instruování plánovače
    \item Ve skutečnosti je \code{wait_event} a \code{wake_up}:
    \begin{itemize}
      \item Nastavení stavu procesu: \code{set_current_state}
      \item Uspání procesu: \code{schedule}
      \item Probuzení procesu: \code{wake_up_process}
      \item Popř.\ s obsluhou signálů: \code{signal_pending}
      \item \header{linux/sched.h}
    \end{itemize}

    \pause

    \item \code{completion} a \code{wait_queue} ulehčuje práci
    \begin{itemize}
      \item Pamatováním si, koho vzbudit
      \item Případnou kontrolou signálů
      \item Případnou kontrolou vypršení timeoutu
      \item \code{completion} navíc řeší předání čítače (kolik procesů vzbudit)
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Explicitní čekání}
  \begin{enumerate}
    \item Místo \code{wait_event}, použijte funkce z předchozího slidu
    \begin{itemize}
      \item Prostudujte (a použijte) tělo makra
	\code{__wait_event_interruptible}
    \end{itemize}
    \item Pozměňte čekací podmínku
    \begin{itemize}
      \item Bude se čekat alespoň na 2~,,\code{a}'' a 1~,,\code{b}'' v poli
    \end{itemize}
  \end{enumerate}

\end{frame}

\section{Vlákna}
\frame{\sectionpage}

\begin{frame}[fragile]
  \frametitle{\code{kthread}}

  \begin{itemize}
    \item Vytvoření vlákna (procesu) pro výpočty
    \begin{itemize}
      \item Běží v jádře
      \item Chování jako uživatelský proces
    \end{itemize}
    \item \header{linux/kthread.h}
    \item Vytvoření, zrušení: \code{kthread_run}, \code{kthread_stop}
    \item Ve vlákně: \code{kthread_should_stop}
  \end{itemize}

  \pause

  \begin{block}{Příklad}
  \begin{center}
  \begin{tabular}{c|c}
  \begin{lstlisting}
static int my_proc(void *data)
{ /* data = my_data */
  while (1) {
    wait_event_interruptible(...);
    if (kthread_should_stop())
      break;
    do_some_work();
  }
  return 0;
}
  \end{lstlisting}~ &
  ~\begin{lstlisting}
static int my_init(void)
{
  struct task_struct *t;
  t = kthread_run(my_proc, my_data, "my_process");
  if (IS_ERR(t))
    return PTR_ERR(t);
  ...
  kthread_stop(t); /* some time later */
  ...
  \end{lstlisting}
  \end{tabular}
  \end{center}
  \end{block}

\end{frame}
