\begin{frame}
  \frametitle{Síťové rozhraní ovladače}

  \heading{\texttt{netdevice}}
  \begin{itemize}
  \item Mezivrstva mezi HW a Linuxem (síťovou vrstvou)
  \item Popisuje každé \texttt{eth*} zařízení
  \item Obsahuje háčky
  \begin{itemize}
  \item Zařízení je UP, DOWN, změna MTU, pošli paket, \ldots
  \end{itemize}
  \end{itemize}

  \pause
  \medskip

  \heading{API}
  \begin{itemize}
  \item \header{linux/netdevice.h}, \texttt{struct net\_device}
  \item Alokace: \texttt{alloc\_netdev}$_D$, \texttt{free\_netdev}$_D$
  \item Registrace: \texttt{register\_netdev}, \texttt{unregister\_netdev}
  \item Háčky
  \begin{itemize}
  \item Staré jádro: v \texttt{netdevice} přímo \texttt{open}, \texttt{stop},
    \ldots
  \item Nové jádro: \texttt{net\_device\_ops} (podobně jako znaková zařízení)
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{\texttt{net\_device\_ops}}

  \begin{lstlisting}[language=C]
struct net_device_ops {
  int (*ndo_open)(struct net_device *dev); /* ifup */
  int (*ndo_stop)(struct net_device *dev); /* ifdown */
  netdev_tx_t (*ndo_start_xmit)(struct sk_buff *skb,
       struct net_device *dev); /* send packet */
  int (*ndo_change_mtu)(struct net_device *dev,
       int new_mtu); /* change MTU */
  ... /* no receive packet */
};
  \end{lstlisting}

\end{frame}

\begin{frame}
  \frametitle{Konkrétnější rozhraní}

  \begin{itemize}
  \item Místo \texttt{alloc\_netdev} se použije konkrétnější
  \item Nastaví vnitřní položky \texttt{netdevice}
  \begin{itemize}
  \item \emph{Ethernet}: \texttt{alloc\_etherdev}
  \item IRDA: \texttt{alloc\_irdadev}
  \item CAN: \texttt{alloc\_candev}
  \item HDLC: \texttt{alloc\_hdlcdev}
  \item \ldots
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Vytvoření \texttt{eth*}}

  \heading{Postup vytvoření ethernetového zařízení v systému}
  \begin{enumerate}
  \item Alokace (\texttt{alloc\_etherdev(priv\_size)})
  \item Vyplnění informací
  \begin{itemize}
  \item Háčky (\texttt{net\_device\_ops})
  \item MAC adresa (\texttt{dev->dev\_addr})
  \item Získání privátního ukazatele o velikosti \texttt{priv\_size}
    (\texttt{netdev\_priv})
  \end{itemize}
  \item Registrace (\texttt{register\_netdev})
  \item \ldots
  \item Deregistrace (\texttt{unregister\_netdev})
  \item Uvolnění (\texttt{free\_netdev})
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Úkol}

  \heading{Vytvořit \texttt{eth*}}
  \begin{enumerate}
  \item Dle postupu z předchozího slidu
  \begin{itemize}
  \item Registrace v \texttt{module\_init}
  \item Deregistrace v \texttt{module\_exit}
  \end{itemize}
  \item Vytvořit \texttt{open}, \texttt{stop}, \texttt{start\_xmit}
  \begin{itemize}
  \item Těla vypíšou jen název funkce
  \item \texttt{start\_xmit} vrátí \texttt{NETDEV\_TX\_OK}
  \end{itemize}
  \end{enumerate}

\end{frame}

\begin{frame}
  \frametitle{Pakety}

  \heading{Tzv. socket buffery}
  \begin{itemize}
  \item \header{linux/skbuff.h}, \texttt{struct sk\_buff}
  \item \texttt{dev\_alloc\_skb}$_D$ (obecně), \texttt{netdev\_alloc\_skb}$_D$
    (pro \texttt{netdevice}), \texttt{dev\_kfree\_skb}$_D$
  \item Obsah: \texttt{skb->data}
  \item Délka obsahu: \texttt{skb->len}
  \item Příjem z HW (např. z přerušení)
  \begin{itemize}
  \item \texttt{skb = netdev\_alloc\_skb(dev, len)}
  \item Naplnění daty z HW
  \item Nastavení protokolu (\texttt{skb->protocol = eth\_type\_trans(skb,
  dev)})
  \item \texttt{netif\_rx(skb)}
  \end{itemize}
  \item Posílání do HW (parametr ve \texttt{start\_xmit})
  \begin{itemize}
  \item Poslat paket
  \item \texttt{dev\_kfree\_skb(skb)}
  \end{itemize}
  \end{itemize}

\end{frame}
