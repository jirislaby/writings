\frame{\subsectionpage}

\begin{frame}{Sdílená paměť}

  \begin{itemize}
    \item \doc{man shm\_overview}, knihovna \cmd{\-lrt}
    \item Vytvoření: \code{int shm_open(const char *name, int oflag, mode_t
      mode)}
    \begin{itemize}
      \item Parametry jako u \code{open}
      \item Ale jméno: 1 lomítko na začátku (\file{/nazev})
    \end{itemize}

    \pause

    \item Operace: \code{ftruncate}, \code{mmap}

    \item Zrušení (nepovinné): \code{int shm_unlink(const char *name)}
    \item Zavření: \code{munmap}, \code{close}

    \pause

    \item Pracuje nad \file{/dev/shm/}
  \end{itemize}

\end{frame}

\begin{frame}{Fronty zpráv}

  \begin{itemize}
    \item \doc{man mq\_overview}, \header{mqueue.h}, knihovna \cmd{\-lrt}
    \item Vytvoření: \code{mqd_t mq_open(const char *name, int oflag)}
    \begin{itemize}
      \item Parametry jako u \code{open}
      \item Ale jméno: 1 lomítko na začátku (\file{/nazev})
    \end{itemize}

    \pause

    \item Komunikace
    \begin{itemize}
      \item Prioritizovaná: vyšší priorita předbíhá
      \item \small \code{int mq_send(mqd_t mq, const char *msg, size_t len, unsigned int prio)}
      \item \small \code{ssize_t mq_receive(mqd_t mq, char *msg, size_t len, unsigned int *prio)}
    \end{itemize}

    \pause

    \item Zrušení (nepovinné): \code{int mq_unlink(const char *name)}
    \item Zavření: \code{int mq_close(mqd_t mq)}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]{Fronty zpráv}{Vlastnosti}

  \begin{itemize}
    \item \small \code{int mq_getattr(mqd_t mq, struct mq_attr *attr)}

    \pause

    \item \small \code{int mq_setattr(mqd_t mq, const struct mq_attr *newattr,} \\
      \code{struct mq_attr *oldattr)}
    \begin{itemize}
      \item Lze změnit jen \code{O_NONBLOCK} v \code{mq_flags}
    \end{itemize}
  \end{itemize}

  \onslide<1->

  \begin{block}{\code{struct mq_attr}}
  \begin{lstlisting}
long mq_flags;       /* Flags: 0 or O_NONBLOCK */
long mq_maxmsg;      /* Max. # of messages on queue */
long mq_msgsize;     /* Max. message size (bytes) */
long mq_curmsgs;     /* # of messages currently in queue */
  \end{lstlisting}
  \end{block}

\end{frame}
