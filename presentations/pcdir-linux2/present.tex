\begin{frame}[fragile]{Blokování signálů}

  { \small
    \code{int sigprocmask(int how, const sigset_t *set, sigset_t *oldset)}
  }
  \begin{itemize}
    \item \code{how} -- co s množinou \code{set}
    \begin{itemize}
      \item Přidat: \code{SIG_BLOCK}
      \item Odebrat: \code{SIG_UNBLOCK}
      \item Nastavit: \code{SIG_SETMASK}
    \end{itemize}
    \item \code{set} -- které signály
    \item \code{oldset} -- původní blokované signály
  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}
sigset_t s;
sigfillset(&s);
sigdelset(&s, SIGTERM);
sigprocmask(SIG_SETMASK, &s, NULL); /* block all except TERM */
  \end{lstlisting}
  \end{block}

\end{frame}
