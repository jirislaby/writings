\begin{frame}[fragile]{Blokování signálů}

  \onslide<2->
  \begin{block}{Prototyp a příklad}
    \begin{lstlisting}[aboveskip=2pt,belowskip=2pt,columns=fullflexible]
int sigprocmask(int how, const sigset_t *set, sigset_t *oldset);

sigset_t s;
sigfillset(&s);
sigdelset(&s, SIGTERM);
sigprocmask(SIG_SETMASK, &s, NULL); /* block all except TERM */
    \end{lstlisting}
  \end{block}
  \onslide<1->

  \begin{itemize}
    \item Blokované signály jsou ve frontě do odblokování

    \pause

    \item \code{how} -- co s množinou \code{set}
    \begin{itemize}
      \item Přidat: \code{SIG_BLOCK}
      \item Odebrat: \code{SIG_UNBLOCK}
      \item Nastavit: \code{SIG_SETMASK}
    \end{itemize}

    \item \code{set} -- které signály
    \item \code{oldset} -- původní blokované signály (jádro naplní)
  \end{itemize}

\end{frame}

\begin{frame}{Synchronní obsluha}

  \begin{itemize}
    \item Čeká se na specifikovaný signál

    \pause

    \item Ten se odebere z fronty
    \begin{itemize}
      \item Rozdíl od \code{pause}/\code{sigsuspend}
    \end{itemize}

    \pause

    \item \pozor: je nutno signály nejprve maskovat!
    \begin{itemize}
      \item Aby nebyly doručovány
    \end{itemize}

    \pause

    \item \code{int sigwait(const sigset_t *set, int *sig)}
    \begin{itemize}
      \item \code{set} -- na co čekat
      \item \code{sig} -- co skutečně přišlo
    \end{itemize}

    \pause

    \item \code{sigwaitinfo}, \code{sigtimedwait}
    \begin{itemize}
      \item Navíc \code{siginfo_t}, popř.\ timeout
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Synchronní signály}
  \begin{enumerate}

    \end{itemize}

    \item V cyklu synchronně obsluhujte signály

    \end{itemize}

    \end{itemize}

    \item Pošlete nakonec i \code{SIGTERM}

\begin{frame}{Broadcasting}

  \end{itemize}
