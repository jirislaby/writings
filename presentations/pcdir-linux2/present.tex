\begin{frame}[fragile]{Blokování signálů}

  { \small
    \code{int sigprocmask(int how, const sigset_t *set, sigset_t *oldset)}
  }
  \begin{itemize}
    \item \code{how} -- co s množinou \code{set}
    \begin{itemize}
      \item Přidat: \code{SIG_BLOCK}
      \item Odebrat: \code{SIG_UNBLOCK}
      \item Nastavit: \code{SIG_SETMASK}
    \end{itemize}
    \item \code{set} -- které signály
    \item \code{oldset} -- původní blokované signály
  \end{itemize}

  \begin{block}{Příklad}
  \begin{lstlisting}
sigset_t s;
sigfillset(&s);
sigdelset(&s, SIGTERM);
sigprocmask(SIG_SETMASK, &s, NULL); /* block all except TERM */
  \end{lstlisting}
  \end{block}

\end{frame}

    \end{itemize}

    \pause

    \item Producent
    \begin{enumerate}
      \item Má data

      \pause

      \item Čeká na místo v bufferu
      \begin{itemize}
	\item Na semafor \code{space}
      \end{itemize}

      \pause

      \item Zapíše data
      \begin{itemize}
	\item A zvýší semafor \code{data}
      \end{itemize}
    \end{enumerate}

    \pause

    \item Konzument
    \begin{enumerate}
      \item Čeká na data v bufferu
      \begin{itemize}
	\item Na semafor \code{data}
      \end{itemize}

      \pause

      \item Přečte a zpracuje data

      \pause

      \item Uvolní místo
      \begin{itemize}
	\item Zvýší semafor \code{space}
      \end{itemize}
    \end{enumerate}

    \item Na počátku
    \begin{itemize}
      \item \code{space} je velikost bufferu
      \item \code{data} je \code{0}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Úkol}

  \heading{Producent/konzument}
  \begin{enumerate}
    \item Napište si producenta a konzumenta
    \begin{itemize}
      \item Proces, který se rozdvojí
      \item Potomek a rodič si budou jednosměrně předávat bajty
    \end{itemize}

    \item Alokujte kruhový buffer 1\,KiB
    \item Mějte 2 společné semafory
    \begin{itemize}
      \item \code{space} a \code{data}
    \end{itemize}

    \item V jednom procesu plňte buffer ze standardního vstupu
    \begin{itemize}
      \item Počkejte na \code{space}, zapište bajt a inkrementujte \code{data}
    \end{itemize}

    \item V druhém čtěte z bufferu a vypisujte na standardní výstup
    \begin{itemize}
      \item Počkejte na \code{data}, přečtěte bajt a inkrementujte \code{space}
    \end{itemize}

    \item Oba musí mít vlastní ukazovátka na buffer
    \begin{itemize}
      \item \code{buf[off] = 'X';} \\
	\code{off = (off + 1) \% 1024;}
    \end{itemize}
  \end{enumerate}

\end{frame}
