\begin{frame}
  \frametitle{Úvod}
  \begin{itemize}
  \item Co je to jádro OS?
  \begin{itemize}
  \item Prostředník mezi procesy a HW
  \item Ovládá a zpřístupňuje jej
  \end{itemize}
  \item Běží ve speciálním režimu
  \item Programování vyžaduje specifické znalosti (jako každé jiné)
  \end{itemize}

  \pause
  \medskip

  \heading{Proč umět programovat jádro?}
  \begin{itemize}
  \item Nový hardware (žádné ovladače)
  \item Starý hardware (špatné ovladače)
  \item Nová funkcionalita (lepší plánovač, síťový stack, atd.)
  \item Oprava chyb
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Rozdíly program vs. jádro}
  \heading{Program (v C)}
  \begin{itemize}
  \item Knihovny
  \begin{itemize}
  \item \emph{libc} (standardní funkce typu \texttt{printf}, \texttt{strlen},
    \texttt{malloc}, \ldots)
  \item Ostatní -- přidávající funckionalitu (\emph{boost}, \emph{pthread},
    \ldots)
  \end{itemize}
  \item Oddělený paměťový prostor
  \begin{itemize}
  \item Procesy se neovlivňují, pakliže to nebylo explicitně povoleno -- zápis
    na adresu mimo alokovaných/mapovaných zabije aplikaci.
  \end{itemize}
  \item Počáteční funkce \texttt{main}
  \end{itemize}

  \pause
  \medskip

  \heading{Jádro}
  \begin{itemize}
  \item Nic z výše uvedeného
  \item \emph{Chyba v jádře často působí pád systému}
  \item Některé funkce reimplementovány
  \begin{itemize}
  \item \texttt{printf} jako \texttt{printk} v Linuxu, \texttt{printf} v
    BSD
  \end{itemize}
  \item Žádná aritmetika s pohyblivou desetinnou čárkou apod.
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Linuxové jádro -- chování I.}

  \heading{2 úrovně zabezpečení}
  \begin{itemize}
  \item \emph{Uživatelský prostor} -- kontrolované vykonávání instrukcí
  \item \emph{Prostor jádra} -- neomezené vykonávání (kód může cokoliv)
  \item Na x86 odpovídá privilege ring 0 a 3 -- podpora CPU
  \item \textbf{Větší důraz na bezpečnost}
  \end{itemize}

  \pause
  \medskip

  \heading{Několik kontextů vykonávání kódu}
  \begin{itemize}
  \item \emph{Proces} -- proces zavolá jádro
  \item \emph{Prerušení} -- kontext právě přerušeného procesu
  \item V kontextu přerušení nelze spát
  \begin{itemize}
  \item Má za sebou cizí proces
  \item Přerušení jsou zakázána -- není, kdo by proces vzbudil
  \end{itemize}
  \item Obecněji atomický (zahrnuje spinlocky) vs. neatomický
  \item \textbf{Pozor na to, co se kde (ne)může volat}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Souběžnost kódu I.}
  \heading{Multitasking a problémy s ním spojené} \\
  Příklad: 2 procesy provádějící následující
  \begin{lstlisting}[language=C]
a = 2;
a = a - a;
  \end{lstlisting}
  Může být a různé od 0?

  \pause
  \medskip

  Ano (jeden z mnoha chybných souběhů):
  \begin{center}
  \begin{tabular}{ll}
PROCES 1 & PROCES 2 \\ \hline
  \begin{lstlisting}[language=XML]
store 2, a
load a, reg1
<context switch>  ->




load a, reg2      <-
sub reg1, reg2
store reg1, a
  \end{lstlisting} &
  \begin{lstlisting}[language=XML]


store 2, a
load a, reg1
load a, reg2
sub reg1, reg2
store reg1, a
<context switch>
# 2 - 0
# a je 2
  \end{lstlisting}
  \end{tabular}
  \end{center}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Souběžnost kódu II.}
  \heading{Řešení}
  \begin{itemize}
  \item Zámky
  \begin{itemize}
  \item Vytvoří kritickou sekci
  \item 2 procesy nemohou vstoupit do kritické sekce (až na semafory)
  \end{itemize}

  \item Jádro poskytuje několik typů zámků
  \begin{itemize}
  \item \emph{mutex} -- spící, nevytváří atomický kontext, lze v něm spát
  \item \emph{semafor} -- podobný mutexu, obsahuje počítadlo, je zanořitelný
  \item \emph{spinlock} -- nespící (busy-waiting), atomický
  \item \emph{BKL} -- zastaralý, hrubozrnný, nepoužívat
  \item A další, speciální\ldots
  \end{itemize}
  \end{itemize}

  \pause
  \smallskip

  Náš kód tedy má vypadat takto:
  \begin{lstlisting}[language=C]
spin_lock(&my_lock);
a = 2;
a = a - a;
spin_unlock(&my_lock);
  \end{lstlisting}

  \pause
  \smallskip

  Co když přijde po \texttt{a\,=\,2} přerušení obsahující stejný kód?

\end{frame}
