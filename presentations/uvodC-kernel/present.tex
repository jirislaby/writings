\begin{frame}
  \frametitle{Úvod}
  \begin{itemize}
  \item Co je to jádro OS?
  \begin{itemize}
  \item Prostředník mezi procesy a HW
  \item Ovládá a zpřístupňuje jej
  \end{itemize}
  \item Běží ve speciálním režimu
  \item Programování jádra je specifické (jako každé jiné)
  \end{itemize}

  \pause
  \medskip

  \heading{Proč umět programovat jádro?}
  \begin{itemize}
  \item Nový hardware (žádné ovladače)
  \item Starý hardware (špatné ovladače)
  \item Nová funkcionalita (lepší plánovač, síťový stack, atd.)
  \item Oprava chyb
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Rozdíly program vs. jádro}
  \heading{Program (v C)}
  \begin{itemize}
  \item Knihovny
  \begin{itemize}
  \item \emph{libc} (standardní funkce typu \texttt{printf}, \texttt{strlen},
    \texttt{malloc}, \ldots)
  \item Ostatní -- přidávající funckionalitu (\emph{pthread}, \emph{gmp},
    \ldots)
  \end{itemize}
  \item Oddělený paměťový prostor
  \begin{itemize}
  \item Procesy se neovlivňují, pakliže to nebylo explicitně povoleno -- zápis
    na adresu mimo alokovaných/mapovaných zabije aplikaci.
  \end{itemize}
  \item Počáteční funkce \texttt{main}
  \end{itemize}

  \pause
  \medskip

  \heading{Jádro}
  \begin{itemize}
  \item Nic z výše uvedeného
  \item Některé funkce reimplementovány
  \begin{itemize}
  \item \texttt{printf} jako \texttt{printk} v Linuxu, \texttt{printf} v
    BSD
  \end{itemize}
  \item Žádná aritmetika s pohyblivou desetinnou čárkou apod.
  \item \emph{Chyba v jádře často působí pád systému}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Linuxové jádro -- chování I.}

  \heading{2 úrovně zabezpečení}
  \begin{itemize}
  \item \emph{Uživatelský prostor} -- kontrolované vykonávání instrukcí
  \item \emph{Prostor jádra} -- neomezené vykonávání (kód může cokoliv)
  \item Na x86 odpovídá privilege ring 0 a 3 -- podpora CPU
  \item \textbf{Větší důraz na bezpečnost}
  \item Demonstrace
  \end{itemize}

  \pause
  \medskip

  \heading{Několik kontextů vykonávání kódu}
  \begin{itemize}
  \item \emph{Proces} -- proces zavolá jádro
  \item \emph{Prerušení} -- kontext právě přerušeného procesu
  \item V kontextu přerušení nelze spát
  \begin{itemize}
  \item Má za sebou cizí proces
  \item Přerušení jsou zakázána -- není, kdo by proces vzbudil
  \end{itemize}
  \item Obecněji atomický (zahrnuje spinlocky) vs. neatomický
  \item \textbf{Pozor na to, co se kde (ne)může volat}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Souběžnost kódu I.}
  \heading{Multitasking a problémy s ním spojené} \\
  Příklad: 2 procesy provádějící následující kód
  \begin{lstlisting}[language=C]
a = 2;
a = a - a;
  \end{lstlisting}
  Může být \texttt{a} různé od 0?

  \pause
  \medskip

  Ano (jeden z mnoha chybných souběhů):
  \begin{center}
  \begin{tabular}{ll}
PROCES 1 & PROCES 2 \\ \hline
  \begin{lstlisting}[language=XML]
store 2, a
load a, reg1
<context switch>  ->




load a, reg2      <-
sub reg1, reg2
store reg1, a
  \end{lstlisting} &
  \begin{lstlisting}[language=XML]


store 2, a
load a, reg1
load a, reg2
sub reg1, reg2
store reg1, a
<context switch>
# 2 - 0
# a je 2
  \end{lstlisting}
  \end{tabular}
  \end{center}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Souběžnost kódu II.}
  \heading{Řešení}
  \begin{itemize}
  \item Zámky
  \begin{itemize}
  \item Vytvoří kritickou sekci
  \item 2 procesy nemohou vstoupit do kritické sekce (až na semafory)
  \end{itemize}

  \item Jádro poskytuje několik typů zámků
  \begin{itemize}
  \item \emph{mutex} -- spící, nevytváří atomický kontext, lze v něm spát
  \item \emph{semafor} -- podobný mutexu, obsahuje počítadlo, je zanořitelný
  \item \emph{spinlock} -- nespící (busy-waiting), atomický
  \item \emph{BKL} -- zastaralý, hrubozrnný, nepoužívat
  \item A další, speciální\ldots
  \end{itemize}
  \end{itemize}

  \pause
  \smallskip

  Náš kód tedy má vypadat takto:
  \begin{lstlisting}[language=C]
spin_lock(&my_lock);
a = 2;
a = a - a;
spin_unlock(&my_lock);
  \end{lstlisting}

  \pause
  \smallskip

  Co když přijde po \texttt{a\,=\,2} přerušení obsahující stejný kód?

\end{frame}

\begin{frame}
  \frametitle{Komunikace I.}
  \begin{center}
  \includegraphics[width=4cm]{proces-jadro}
  \end{center}
  \begin{itemize}
  \item Základní forma komunikace: volání funkcí
  \item Co se děje, když se z programu zavolá \texttt{fwrite}?
  \begin{itemize}
  \item \texttt{fwrite} je funkce \emph{libc}
  \item Vnitřní buffery, volání \texttt{write}
  \item \texttt{write} je (stále) funkce \emph{libc}
  \item Nyní nutno volat funkci jádra
  \end{itemize}\pause
  \item Jádro nelze volat přímo
  \begin{itemize}
  \item Důvod: dvě úrovně zabezpečení
  \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Komunikace II.}
  \begin{itemize}
  \item Nutno provést přepnutí privilegií (obsah registrů)
  \begin{itemize}
  \item Tzv. system call (syscall)
  \end{itemize}
  \end{itemize}

  \pause

  \heading{1) speciální instrukcí procesoru}
  \begin{itemize}
  \item Na x86 \emph{int} (80), \emph{sysenter} nebo \emph{syscall},
  na ia64 \emph{break} nebo \emph{epc}, na ARM \emph{swi} atd.
  \item Princip vesměs stejný
  \begin{itemize}
  \item Přeruší se vykonávání (podobné IRQ z HW)
  \item Uloží se registry
  \item Začne se vykonávat kód jádra\pause
  \item Provede se potřebné volání
  \item Obnoví se skoro všechny registry (a oprávnění)
  \item Pokračuje se ve vykonávání programu
  \end{itemize}\pause
  \item Nepatří mezi nejrychlejší operace
  \begin{itemize}
  \item Vylepšené instrukce: \emph{sysenter}, \emph{syscall}, \emph{epc}
  \end{itemize}
  \item Příklad
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{Komunikace III.}
  Do jádra jsme se dostali, co dál?
  \begin{itemize}
  \item V jednom z registrů je číslo syscallu
  \item Jádro v tabulce zjistí adresu funkce a zavolá ji
  \item Ta si zkontroluje (zdroj chyb) a zkopíruje vstupy
  \item Pro \texttt{write} se zavolá odpovídající souborový systém
  \item Do jednoho z registrů se zapíše návratová hodnota
  \end{itemize}

\end{frame}

\begin{frame}[fragile]

  \heading{2) skokem do speciálního prostoru}
  \begin{itemize}

  \item \emph{vsyscall} (zastaralé) a \emph{vdso}

\begin{frame}[fragile]
  \frametitle{Komunikace V.}
  \heading{3) ostatní formy komunikace}
  \begin{itemize}
  \item Syscally trochu jinak
  \begin{itemize}
  \item Nelze pro každou funkcionalitu/ovladač přidat jeden syscall
  \item Ovladače vytváří \texttt{/dev/*} zařízení
  \item Nad nimi se používá \texttt{open/read/write/ioctl/.../close}
  \item \texttt{ioctl} je rozšíreným způsobem komunikace
  \end{itemize}
  \begin{lstlisting}[language=C]
int ioctl(int fd, int command, unsigned long argument);
  \end{lstlisting}\pause
  \item Sdílená paměť
  \begin{itemize}
  \item Producer-consumer
  \item Uživatel vidí i na data např. od síťové karty
  \item Bez kopírování dat
  \item Pozor na koherentnost dat!
  \begin{itemize}
  \item Některé architektury používají bounce buffery
  \end{itemize}
  \end{itemize}
  \end{itemize}
