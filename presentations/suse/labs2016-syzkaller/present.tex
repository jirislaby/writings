\begin{frame}{\syz}{\code{__sanitizer_cov_trace_pc}}

  \begin{itemize}
    \item What does it do?
    \pause
    \begin{itemize}
      \item Stores its caller address
      \item All exported in \textit{debugfs}
    \end{itemize}

    \pause

    \item Who does instrument it into the kernel?
    \pause
    \begin{itemize}
      \item \textsc{Gcc} $\geq$ 6

      \pause

      \item But \textsc{Gcc} in SP2 is 4.8 and 5 only
    \end{itemize}

    \pause

    \item How to enable the instrumentation?
    \pause
    \begin{itemize}
      \item Turn on \texttt{CONFIG\_KCOV}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{\syz}{Requirements}

  \begin{itemize}
    \item \textsc{Gcc} $\geq$ 6
    \begin{itemize}
      \item Instrumentation support
    \end{itemize}

    \pause

    \item Kernel
    \begin{itemize}
      \item Enable instrumentation: \texttt{KCOV}
      \item Enable debugging (optional): \texttt{KASAN}, \texttt{KMEMLEAK},
	\texttt{LOCKDEP}
      \item Disable unwinder (too slow to boot): \texttt{STACK\_UNWIND}
    \end{itemize}

    \pause

    \item \syz package

    \pause

    \item \textsc{Qemu}
    \begin{itemize}
      \item \syz fuzzes in isolation
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{\syz}{Principle of Operation}

  \begin{enumerate}
    \item \texttt{syz-manager} starts a virtual machine

    \pause

    \item \texttt{syz-manager} starts \texttt{syz-fuzzer} inside the VM

    \pause

    \item Random data are generated

    \pause

    \item \label{item:invok} Syscalls sequence is invoked with appropriate
      parameters

    \pause

    \item Check dmesg for a WARNING/BUG
    \begin{itemize}
      \item Generate a report if there is one
    \end{itemize}

    \pause

    \item Look at the visited paths and permute the input accordingly

    \pause

    \item Check if the VM is viable
    \begin{itemize}
      \item Restart if needed
    \end{itemize}

    \pause

    \item Repeat from \ref{item:invok}
  \end{enumerate}

\end{frame}

\begin{frame}{\syz}{Reproducers}

  \begin{itemize}
    \item \syz can generate reproducers from reports
    \begin{itemize}
      \item A simple C code
      \item Triggers the reported WARNING/BUG
    \end{itemize}

    \pause

    \item Run \texttt{syz-repro}
    \begin{itemize}
      \item Give the report as a parameter
    \end{itemize}

    \pause

    \item The reproducer is emitted
    \begin{itemize}
      \item \texttt{syz-repro} may fail for rare race conditions etc.
    \end{itemize}
  \end{itemize}

\end{frame}
