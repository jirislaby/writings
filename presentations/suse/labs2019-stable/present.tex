\begin{frame}{Question 3}{Vanilla Flavour}

  \begin{center}
    \heading{Vanilla flavour used to be built using first $x$ hundred patches} \\
    (namely, \file{kernel.org} \& \file{rpmify}) \\[2ex]

    Patches now in sorted section, cannot cherry-pick them (conflicts). \\
  \end{center}

  \vspace{2ex}
  \pause

  \begin{enumerate}[<+->]
    \item Drop \file{vanilla} -- any real use?
    \item No stable patches -- use pure e.g. 5.3 (plus \file{rpmify}).
      \begin{itemize}
	\item New toolchains $\rightarrow$ broken build. Should be no issue.
      \end{itemize}
    \item Use upstream stable tree (plus \file{rpmify}) -- \file{rpmify} might
      not apply.
  \end{enumerate}

\end{frame}
