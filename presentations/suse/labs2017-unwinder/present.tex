\begin{frame}{Motivation}

  \heading{Uses of Call Traces}

  \begin{itemize}
    \item Debugging purposes
    \begin{itemize}
      \item Crashes (including BUGs, WARNINGs, and similar)

      \pause

      \item \textsc{FTrace}, \textsc{KMemLeak}, \textsc{KMemCheck},
	\textsc{LockDep}
    \end{itemize}

    \pause

    \item \emph{Kernel Live Patching}
    \begin{itemize}
      \item Is any thread in a to-be-patched function?

      \pause

      \item We ship and support KLP $\rightarrow$ we need fast and precise call
	traces
    \end{itemize}

  \end{itemize}

\end{frame}
