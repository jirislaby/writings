  \begin{itemize}

\begin{frame}{Guess and Frame Pointer Unwinders}

  \begin{itemize}[<+->]
    \item In upstream for a long time

    \item Rewritten to a sane interface in 4.9
    \begin{itemize}
      \item Allows for new unwinders
    \end{itemize}

    \item Frame pointers disabled $\Rightarrow$ Guess
    \begin{itemize}[<+->]
      \item Dump everything from stack which looks sensible
      \item Imprecise, but fast
    \end{itemize}

    \item Frame pointers enabled $\Rightarrow$ Use them
    \begin{itemize}[<+->]
      \item Use \code{BP} register to look at the start of a frame
      \item Repeat until bottom of the stack
      \item Errors out when assembly has no frame pointers
      \begin{itemize}[<+->]
	\item Attempts to fix this by Josh Poimboeuf \& \objt
      \end{itemize}
      \item Mostly precise and fast
    \end{itemize}

    \item In SUSE kernels, used only as a fallback
  \end{itemize}

\end{frame}

\begin{frame}{DWARF Unwinder}

  \begin{itemize}[<+->]
    \item Jan Beulich \& Andi Kleen from SUSE
    \item In upstream for two releases (2.6.18--19)
    \begin{itemize}[<+->]
      \item Then dropped by Linus
    \end{itemize}

    \item Out-of-band (debug) data
    \begin{itemize}[<+->]
      \item How much was stored on the stack by a function at a given point
      \item Subtract that number from the current \code{SP}
      \item Extract the return address from there
      \item Repeat until bottom of the stack
    \end{itemize}

    \item A big state machine and DWARF instruction set
    \begin{itemize}[<+->]
      \item Example: \code{DW\_CFA\_def\_cfa} (\code{.cfi\_def\_cfa} in .S
	files)
      \item Slow, but precise
    \end{itemize}

    \item The default in all our kernels for a decade
    \begin{itemize}[<+->]
      \item SLE11 and later
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{DWARF Unwinder \& Assembly}

  \begin{itemize}[<+->]
    \item The DWARF data are generated by a C compiler
    \item Assembler has no idea about the semantics
    \item \emph{Problem:} we have no data for hand-written assembly

    \item Previously, assembly was annotated manually
    \begin{itemize}
      \item Dropped in 4.2 due to bugs
      \item SLE12-SP2/SP3(/SP4) affected
    \end{itemize}

    \item We can no longer unwind through assembly
    \begin{itemize}[<.->]
      \item<+-> \textit{DWARF2 unwinder stuck at \dots, Leftover inexact
	backtrace: \dots}

      \item<+-> Interrupt handlers
      \item Syscall handlers
      \item Page fault handler
      \item \dots
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]{ORC}

  \begin{itemize}[<+->]
    \item Similar to DWARF
    \begin{itemize}[<+->]
      \setcounter{mybeamerpauses}{\value{beamerpauses}}
      \item Out-of-band data
      \addtocounter{beamerpauses}{2}
      \item But much simpler
    \end{itemize}

    \item In upstream
    \begin{itemize}[<+->]
      \item Preparation in 4.12
      \item Core and fixes in 4.13
    \end{itemize}

    \item We backported to 4.12
    \begin{itemize}[<+->]
      \item SLE15 uses ORC since the beginning
    \end{itemize}

  \end{itemize}

  \setcounter{beamerpauses}{\value{mybeamerpauses}}
  \onslide<+->

  \begin{block}{\code[language=]{struct orc\_entry} assigned to instructions}
    \begin{lstlisting}[aboveskip=2pt,belowskip=0pt]
s16      sp_offset; // *(sp_reg) + sp_offset points to the start of a frame
    \end{lstlisting}
    \onslide<+->
    \begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
s16      bp_offset; // similarly, but only if frame pointers are enabled
    \end{lstlisting}
    \onslide<.(-1)->
    \begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
unsigned sp_reg:4; // ORC_REG_SP, ORC_REG_DX, ORC_REG_R10, ...
    \end{lstlisting}
    \onslide<.->
    \begin{lstlisting}[aboveskip=0pt,belowskip=0pt]
unsigned bp_reg:4; // ORC_REG_BP, ORC_REG_R10, ...
    \end{lstlisting}
    \onslide<+->
    \begin{lstlisting}[aboveskip=0pt,belowskip=2pt]
unsigned type:2;   // ORC_TYPE_CALL, ORC_TYPE_REGS (pt_regs), ORC_TYPE_REGS_IRET (interrupt)
    \end{lstlisting}
  \end{block}

\end{frame}

\begin{frame}[fragile]{ORC and \objt}

  \begin{itemize}[<+->]
    \item Entries generated by \objt
    \begin{itemize}[<+->]
      \item Searches for patterns and emits entries \\
      \begin{tabular}{rcl}
	\begin{tabular}{p{7em}}
	  \begin{lstlisting}[language=assembler,morekeywords={rbp,rsp}]
push   %rbp
mov    %rsp, %rbp
	  \end{lstlisting}
	\end{tabular}
	& $\rightarrow$ &
	\begin{tabular}{p{12em}}
	  \begin{lstlisting}[language=]
sp:sp+8 bp:(und) type:call
sp:sp+16 bp:prevsp-16 type:call
	  \end{lstlisting}
	\end{tabular}
      \end{tabular}

      \item If compiler/patterns change, \objt has to be changed
    \end{itemize}

    \item Entries emitted into object files
    \begin{itemize}
      \item Linker collects them together
    \end{itemize}

    \item Assembly handled too, but hints needed
    \begin{itemize}[<+->]
      \item \code{UNWIND\_HINT\_EMPTY}, \code{UNWIND\_HINT\_REGS}, \dots
      \item Only few in comparison to DWARF
    \end{itemize}

    \item Kernel both sorts and uses the data at runtime
    \item \emph{Drawback:} two more tables in memory

  \end{itemize}

\end{frame}
