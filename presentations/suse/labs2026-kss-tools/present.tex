\section{Introduction}

\begin{frame}{Introduction}

  \begin{itemize}[<+->]
    \item Tools to handle \emph{kernel} CVEs
    \item Is CVE relevant for SUSE?
      \begin{itemize}[<+->]
	\item \cmd{CONFIG_} enabled?
	\item What \cmd{supported.conf} says?
	\item In \emph{any} branch?
      \end{itemize}
    \item If so, who maintains the vulnerable code?
      \begin{itemize}
	\item Standalone tool: \cmd{suse-get-maintainers}
      \end{itemize}
    \item Power of \cmd{check-kernel-fix}
    \item Most tools documented in Confluence: \href{https://confluence.suse.com/spaces/labs/pages/1833140703/Kernel+tools}{Kernel tools}
  \end{itemize}

\end{frame}

\begin{frame}{Prerequisites}

  \heading{Most tools assume ENV settings}
  \begin{itemize}
    \item \cmd{KSOURCE_GIT}: \url{git://kerncvs.suse.de/kernel-source.git}
    \item \cmd{LINUX_GIT}: \small\url{https://git.kernel.org/pub/scm/linux/kernel/git/torvalds/linux.git}
    \item \cmd{VULNS_GIT}: \url{https://git.kernel.org/pub/scm/linux/security/vulns.git}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]{\file{conf_file_map.sqlite}}{Content VIII.}

  \begin{center}
    \Huge
    \heading{Views!}
  \end{center}

  \pause

  \begin{block}{View Illustrated}
    \begin{lstlisting}[language=sql]
SELECT * FROM module_file_map_view /* WHERE branch='SL-16.1' AND module LIKE '\%/ext4' */;
    \end{lstlisting}
    \pause
    Executes this:
    \begin{lstlisting}[language=sql]
CREATE VIEW module_file_map_view AS
    SELECT branch.branch, module_dir.dir || '/' || module.module AS module, dir.dir || '/' || file.file AS path
    FROM module_file_map AS map
    LEFT JOIN module ON map.module = module.id
    LEFT JOIN dir AS module_dir ON module.dir = module_dir.id
    LEFT JOIN branch ON map.branch = branch.id
    LEFT JOIN file ON map.file = file.id
    LEFT JOIN dir ON file.dir = dir.id;
    \end{lstlisting}
  \end{block}

\end{frame}

\section{SlHelpers}
\frame{\sectionpage}

\begin{frame}{SlHelpers}

  \begin{itemize}[<+->]
    \item C++ helpers (wrappers) for C libraries and other tools
    \item RAII, smart pointers, \dots
    \item Easy to use
    \item Used by most of the tools above
    \item Python3 bindings (in progress)
    \item Availability
      \begin{itemize}
	\item \url{https://github.com/jirislaby/slhelpers}
	\item \href{https://build.opensuse.org/package/show/Kernel:tools/slhelpers}{\nolinkurl{obs://Kernel:tools/slhelpers}}
      \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}[fragile]{SlHelpers}{Libraries}

  \begin{tabular}{@{} p{5cm}p{5cm}p{5cm} @{}}
    \begin{itemize}
      \item SlCurl
      \item SlCVEs
    \end{itemize} &
    \begin{itemize}
      \item SlGit
      \item SlPCRE2
    \end{itemize} &
    \begin{itemize}
      \item SlSqlite
    \end{itemize} \\[-.7em]
