\begin{frame}[fragile]{\file{conf_file_map.sqlite}}{Views}

  \begin{center}
    \Huge
    \heading{Views!}
  \end{center}

  \pause

  \begin{block}{View Illustrated}
    \begin{lstlisting}[language=sql]
SELECT * FROM module_file_map_view /* WHERE branch='SL-16.1' AND module LIKE '%/ext4' */;
    \end{lstlisting}
    \pause
    Executes \textasciitilde\ this:
    \begin{lstlisting}[language=sql]
CREATE VIEW module_file_map_view AS
    SELECT branch.branch, module_dir.dir || '/' || module.module AS module, dir.dir || '/' || file.file AS path
    FROM module_file_map AS map
    LEFT JOIN module ON map.module = module.id
    LEFT JOIN dir AS module_dir ON module.dir = module_dir.id
    LEFT JOIN branch ON map.branch = branch.id
    LEFT JOIN file ON map.file = file.id
    LEFT JOIN dir ON file.dir = dir.id;
    \end{lstlisting}
  \end{block}

\end{frame}
