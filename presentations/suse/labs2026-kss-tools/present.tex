\begin{frame}{Introduction}

  \begin{itemize}[<+->]
    \item Tools to handle \emph{kernel} CVEs
    \item Is CVE relevant for SUSE?
      \begin{itemize}[<+->]
	\item \cmd{CONFIG_} enabled?
	\item What \cmd{supported.conf} says?
	\item In \emph{any} branch?
      \end{itemize}
    \item If so, who maintains the vulnerable code?
      \begin{itemize}
	\item Standalone tool: \cmd{suse-get-maintainers}
      \end{itemize}
    \item Power of \cmd{check-kernel-fix}
    \item Most tools documented in Confluence: \href{https://confluence.suse.com/spaces/labs/pages/1833140703/Kernel+tools}{Kernel tools}
  \end{itemize}

\end{frame}

\begin{frame}{Prerequisites}

  \heading{Most tools assume ENV settings}
  \begin{itemize}
    \item \cmd{KSOURCE_GIT}: \url{git://kerncvs.suse.de/kernel-source.git}
    \item \cmd{LINUX_GIT}: \small\url{https://git.kernel.org/pub/scm/linux/kernel/git/torvalds/linux.git}
    \item \cmd{VULNS_GIT}: \url{https://git.kernel.org/pub/scm/linux/security/vulns.git}
  \end{itemize}

\end{frame}

\begin{frame}{Recap: \cmd{suse-get-maintainers}}

  \begin{itemize}[<+->]
    \item M. Franc last year
      \begin{itemize}
	\item Not recorded \Sadey[][yellow]
      \end{itemize}
    \item Find a SUSE maintainer of a kernel file
    \item Examples
      \begin{itemize}[<+->]
	\item \cmd{suse-get-maintainers --path drivers/tty/serial/samsung_tty.c}
	\item \cmd{suse-get-maintainers --sha 832e231cff47}
	\item \cmd{suse-get-maintainers --cve CVE-2023-53687}
	\item \cmd{suse-get-maintainers --diff <(git -C \$KSOURCE_GIT show origin/SLE15-SP7:}\textbackslash \\
	  \cmd{patches.suse/tty-serial-samsung_tty-Fix-a-memory-leak-in-s3c.patch)}
	\item \cmd{suse-get-maintainers --grep tty --grep-names-only}
	\item All about TTY $\rightarrow$ \url{jslaby@suse.com}
	\item Try \cmd{--json} (and use \cmd{jq} to color/parse it)
      \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{\cmd{suse-get-maintainers}: Data Sources}
