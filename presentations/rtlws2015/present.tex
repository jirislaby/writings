\begin{frame}[fragile]
  \frametitle{Introduction}
  \framesubtitle{What is static analysis?}

  \begin{itemize}
    \item Analysing code similar to code inspection
    \begin{itemize}
      \item In whatever language (even machine code)
    \end{itemize}

    \pause

    \item Without directly executing code on processor
    \begin{itemize}
      \item Program state simulation
    \end{itemize}
  \end{itemize}

  \pause
  \medskip

  \begin{block}{Example}
    \begin{lstlisting}
int foo(int a) {
  int b = 0;
  if (usually)
    b = 10;
  return a / b;
}
    \end{lstlisting}
  \end{block}

\end{frame}

\begin{frame}{Introduction}{Why to use static analysis?}

  \heading{Pros}
  \begin{itemize}
    \item Can catch bugs in unlikely branches
    \begin{itemize}
      \item Higher coverage
    \end{itemize}

    \pause

    \item Does not need specific hardware
    \begin{itemize}
      \item E.g.\ when checking drivers
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Introduction}{Why not?}

  \heading{Cons}
  \begin{itemize}
    \item Produces false positives
    \begin{itemize}
      \item Imprecision in the state simulation
    \end{itemize}

    \pause

    \item May be slow
    \begin{itemize}
      \item Complex algorithms
      \item Complex code
      \begin{itemize}
	\item E.g.\ find \code{y} such that \code{MD5(y) == 1} holds
      \end{itemize}
    \end{itemize}

    \pause

    \item The problem is \emph{generally undecidable}
    \begin{itemize}
      \item Not \emph{every} program can be reasoned about
      \item We can try our best for the rest :)
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}{The History}

  \begin{itemize}
    \item Three papers laying down the foundation
    \begin{itemize}
      \item Kildall – A Unified Approach to Global Program Optimization
      \item Cousot\&Cousot – Abstract interpretation: a unified lattice model
	for static analysis of programs by construction or approximation of
	fixpoints
      \item King – Symbolic execution and program testing
    \end{itemize}

    \pause

    \item No big advances till 2000's
  \end{itemize}

\end{frame}
