\begin{frame}
  \frametitle{What is \textsc{Stanse}}
  \begin{itemize}
  \item Mainly a framework for new techniques
  \begin{itemize}
  \item No additional work, everything needed is available
  \end{itemize}
  \pause
  \item But also implements some static analyses
  \pause
  \item Developed at ITI FI, MU
  \pause
  \item Written in Java
  \begin{itemize}
  \item Ease of new techniques implementation
  \item Some drawbacks (see later)
  \end{itemize}
  \end{itemize}

  \begin{center}
  \includegraphics[height=3.5cm]{stanse}
  \end{center}

\end{frame}

\begin{frame}
  \frametitle{Current State}

  \begin{itemize}
  \item \emph{Full ANSI C99 support}, including most GNU C extensions
    \begin{itemize}
    \item I.e. analyses the Linux kernel
    \end{itemize}
  \item Modular, easily extensible
  \item Easy-to-use \emph{graphical interface} and error inspection
  \item Checker--independent system for error reporting
  \item Intra- and inter-procedural analysis
  \item Batch + build system support (\textsc{make})
  \item Web interface for errors tracking
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{AutomatonChecker}

  \begin{itemize}
  \item Finite automata for erroneous behaviour description
  \begin{itemize}
  \item Stems from \textsc{xgcc} (Stanford)
  \end{itemize}
  \end{itemize}
  \begin{center}
    \input{locking.pspdftex}
  \end{center}

  \pause

  \begin{itemize}
  \item Hlídá zámky, paměť, referenční počty a podobné chyby
  \begin{itemize}
    \item Cokoli může být popsáno KA, může být i zkontrolováno
    \item Např. chybné zamykání, vnoření zámků,
      dereference NULL, spaní uvnitř kritických sekcí, úniky zdrojů atd.
  \end{itemize}
  \end{itemize}
\end{frame}
  
\begin{frame}
  \frametitle{ThreadChecker}
  \begin{itemize}
  \item Hledá a sleduje chování vlákna
  \item Pozoruje zamykání a odemykání zámků (interprocedurálně)
  \item Tvoří tzv. Resource Allocation grafy
    \begin{itemize}
    \item K výpočtu závislostí zámků
    \end{itemize}
  \item Proloží vytvořené grafy všech vláken a hledá cykly
  \item Nachází netriviální problémy se zámky mezi vlákny
  \end{itemize}

  \begin{center}
  \begin{tabular}{ccccccc}

\begin{frame}[fragile]
  \frametitle{LockChecker}

  \begin{itemize}
  \item Statistická analýza (Engler)
  \item Sleduje přístupy do paměti
  \item Počítá statistiku, kolik z nich bylo pod zámkem
  \item Poměr alespoň 1:9 znamená pravděpodobnou chybu
  \end{itemize}

  \begin{center}
  \begin{tabular}{|c|c|} \hline
  V kódu 20$\times$ & V kódu 1$\times$ \\ \hline\hline
  \begin{lstlisting}[language=C]
lock(A);
glob_var = 1;
unlock(A);
  \end{lstlisting} &
  \begin{lstlisting}[language=C]
/* no lock */
glob_var = 5;
/* no unlock */
  \end{lstlisting} \\ \hline
  & \textcolor{red}{Potencionální chyba} \\ \hline
  \end{tabular}
  \end{center}

\end{frame}

  \end{itemize}

  Přes 120 chyb následujících druhů (nahlášeny a opraveny):
  \begin{itemize}

  \item Nesprávné zacházení se zámky

  \end{itemize}

\begin{frame}[fragile]
  \frametitle{Nalezené chyby II.}

  \heading{Malý příklad chyby v jádře}

  Zdroj: \texttt{\small linux-2.6.28/drivers/pci/hotplug/pciehp\_core.c}
  \begin{lstlisting}[language=C]
static int set_lock_status(struct hotplug_slot *hotplug_slot,
                           u8 status)
{
  struct slot *slot = hotplug_slot->private;
  ...
  mutex_lock(&slot->ctrl->crit_sect);

  /* has it been >1 sec since our last toggle? */
  if ((get_seconds() - slot->last_emi_toggle) < 1)
    return -EINVAL;
  ...
  mutex_unlock(&slot->ctrl->crit_sect);
  return 0;
}
  \end{lstlisting}

  Spoušť: \texttt{\small echo '1' > /sys/bus/pci/slots/.../lock} dvakrát za 1s

  \pause
  \medskip

  Uplný seznam na \url{http://stanse.fi.muni.cz/bugs.html}.

\end{frame}

\begin{frame}
  \frametitle{Nevýhody \textsc{Stanse}}

  \heading{Falešná hlášení}
  \begin{itemize}
  \item Oscilují mezi 30 a 70 \% -- v závislosti na checkeru, konfiguraci a
  zdrojových souborech
  \end{itemize}

  \medskip
  \pause

  \heading{Špatný parser}
  \begin{itemize}
  \item Pomalý -- zčásti Java, zčásti \textsc{antlr}
  \item Paměťově náročný -- zejména XML AST strom
  \item Málo informací -- např. chybí typy proměnných
  \item Omezený preprocesor -- nelze mapovat na původní kód
  \end{itemize}

\end{frame}
