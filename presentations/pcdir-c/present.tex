\begin{frame}{Proč C?}

  \begin{itemize}

      \item Použití na aplikace původně implementované v assembleru
      \item Zdrojový kód kompilovaný do nativního kódu HW platformy

    \item Jeden z nejpopulárnějších jazyků vůbec

    \item Vysoká rychlost kódu

\begin{frame}{Rozdíly mezi C a C++}

  \begin{itemize}
    \item Objektovost
    \begin{itemize}
      \item C++ má \code{class}, C ne
      \item Neexistují metody tříd (\code{cls->method()}) ani dědičnost
      \item Máme ale struktury (\code{struct}, viz dále)
    \end{itemize}

    \pause

    \item Šablony
    \begin{itemize}
      \item C++ má \code{template}, C ne
      \item C neumí \code{set<int>}, můžeme udělat jet obecnou \code{set}
    \end{itemize}

    \pause

    \item Rychlost překladu
    \begin{itemize}
      \item C výrazně rychlejší
      \item C++ je špatně navržený jazyk (z pohledu teorie a parsování)
    \end{itemize}

  \end{itemize}

\end{frame}
