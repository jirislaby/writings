      \item Máme ale struktury (\code{struct}, viz dále)
    \end{itemize}

    \item Šablony

      \item C neumí \code{set<int>}, můžeme udělat jet obecnou \code{set}
    \end{itemize}

    \pause

    \item Rychlost překladu

\begin{frame}{Modifikátory celočíselných typů}

  \begin{itemize}
    \item Celočíselné typy jsou znaménkové (\code{signed})
    \begin{itemize}
      \item Výchozí \code{signed} není třeba uvádět
      \item Mohou nabývat záporných hodnot
      \item Ale pozor na \code{char}: může platit \code{char} $\neq$
	\code{signed char}
    \end{itemize}

    \pause

    \item \code{unsigned}
    \begin{itemize}
      \item Změna na neznaménkový
      \item Volný bit zvyšuje rozsah hodnot (\textasciitilde\ 2$\times$)
      \item Např.\ \code{unsigned long cislo1}, \code{unsigned int cislo2}
    \end{itemize}

    \pause

    \item Detailně: \url{http://en.wikipedia.org/wiki/C_data_types}
  \end{itemize}

\end{frame}

\begin{frame}{Velikost typů}

  \begin{itemize}
    \item Velikost typů se na různých platformách liší
    \begin{itemize}
      \item Zjistíme operátorem \code{sizeof()}
    \end{itemize}

    \pause

    \item Standard předepisuje následující vztahy
    \begin{itemize}
      \item \code{sizeof(_Bool)} $\leq$ \code{sizeof(char)} $\leq$
        \code{sizeof(short)} $\leq$ \code{sizeof(int)} $\leq$
	\code{sizeof(long)} $\leq$ \code{sizeof(long long)}

      \pause

      \item \code{sizeof(float)} $\leq$ \code{sizeof(double)} $\leq$
        \code{sizeof(long double)}

      \pause

      \item \code{sizeof(char)} $=$ 1 (nemusí být ale 8 bitů, některé DSP mají
        např.\ 16b)
    \end{itemize}
  \end{itemize}

\begin{frame}{Proměnná}{Pojmenování}

  \begin{itemize}
    \item Pravidla (povinné)
    \begin{itemize}
      \item Musí začínat znakem nebo \code{_} (NE \code{int 3prom;})
      \item Může obsahovat pouze znaky, číslice a podtržítko
      \pause
      \item Záleží na velikosti znaků (\code{int prom; int Prom;})
      \pause
      \item Nesmí být klíčové slovo jazyka (NE \code{int int = 1;})
    \end{itemize}

    \pause

    \item Konvence (doporučené)
    \begin{itemize}
      \item Respektujte existující konvenci v projektu!

      \pause

      \item Volte výstižná jména
      \begin{itemize}
	\item ANO: \code{float divider = 1; int number_of_members = 0;}
	\pause
	\item NE: \code{tmp}, \code{qwerty}, \code{PROMENNA}
	\pause
	\item Ale jména \code{a}, \code{i} a \code{j} se používají často jako
	  řídící proměnné v cyklu (iterátory)
      \end{itemize}

      \pause

      \item Jména proměnných začínejte malým písmenem
      \item Nezačínejte proměnné \code{__} nebo \code{_X} (\code{X} je
        libovolné velké písmeno)
      \begin{itemize}
	\item Rezervované, typicky různé platformě závislá makra apod.
      \end{itemize}

      \pause

      \item Oddělujte slova v názvu proměnné
      \begin{itemize}
	\item \code{camelCase} nebo \code{raz_dva}
      \end{itemize}
    \end{itemize}
  \end{itemize}

\end{frame}

\begin{frame}{Operátory}

      \item Priorita určuje pořadí operátorů -- který vyhodnotit dříve

    \end{itemize}
  \end{itemize}

\begin{frame}{Operátor dereference *}

  \begin{itemize}
    \item Pracuje nad proměnnými typu ukazatel
    \begin{itemize}
      \item Podívej se na adresu v proměnné X jako na hodnotu typu Y
    \end{itemize}

    \pause

    \item Zpřístupní hodnotu, na kterou ukazatel ukazuje
    \begin{itemize}
      \item Nikoli vlastní hodnotu ukazatele (což je adresa)
      \item Ukazatel ale může ukazovat na hodnotu, která se interpretuje zase
        jako adresa (např.\ \code{int **})
    \end{itemize}

    \pause

    \item Příklady dereference typu (není reálné C)
    \begin{itemize}
      \item \code{&int} je \code{int *}
      \item \code{*(int *)} je \code{int}
      \item \code{*(int **)} je \code{int *}
      \item \code{**(int **)} je \code{int}
      \item \code{*(\&int)} je \code{int}
      \item \code{**(\&\&int)} je \code{int}
    \end{itemize}

    \pause

    \item Pokud je dereference použita jako l-hodnota, tak se mění odkazovaná
      hodnota, nikoli adresa ukazatele
    \begin{itemize}
      \item \code{int *pX; pX = 10;} (typicky špatně -- nechceme měnit ukazatel)
      \item \code{int *pX; *pX = 10;} (typicky OK -- chceme měnit hodnotu,
        která je na adrese na kterou \code{pX} ukazuje)
    \end{itemize}
  \end{itemize}

\end{frame}

  \begin{itemize}
    \item Nedostatečně velký cílový řetězec
    \begin{itemize}
      \item Např.\ při \code{strcpy}
    \end{itemize}

    \pause

    \item Chybějící koncová nula
    \begin{itemize}
      \item Následné funkce nad řetězcem nefungují korektně
      \item Vznikne např.\ nevhodným přepisem (N+1 problém)
    \end{itemize}

    \pause

    \item Nevložení koncové nuly
    \begin{itemize}
      \item Např.\ při \code{strncpy}
    \end{itemize}

    \pause

    \item Délka/velikost řetězce bez místa pro koncovou nulu
    \begin{itemize}
      \item Např.\ při \code{strlen}
    \end{itemize}

    \pause

    \item \code{if (str == "Hello") \{ ... \}}
    \begin{itemize}
      \item Špatně, porovnává hodnotu ukazatelů, nikoli obsah řetězců
      \item Nutno použít \code{if (strcmp(str, "Hello") == 0) \{ ... \}}
    \end{itemize}

    \pause

    \item Spájení řetězců pomocí \code{+}
    \begin{itemize}
      \item Sčítá ukazatele namísto obsahu
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}{Dynamická alokace}{Vícedimenzionální pole}

    \item 3D pole je pole ukazatelů na pole ukazatelů na 1D pole
  \end{itemize}

void allocate2DArray() {
  // 100 items array with elements int*
  const int LEN = 100;
  int** array2D = malloc(LEN * sizeof(int*));
  for (int i = 0; i < LEN; i++) {
    // allocated 1D array on position array2D[i]
    // length is i+1 * 10
    array2D[i] = malloc((i+1) * 10 * sizeof(int));
  }
  array2D[10][20] = 1;
  // You need to free memory properly
  for (int i = 0; i < LEN; i++) {
    free(array2D[i]); array2D[i] = NULL;
  }
  free(array2D); array2D = NULL;
}

\begin{frame}{Alokace paměti}{Typy}

    \end{itemize}

    \pause

    \item Paměť alokovaná za běhu na zásobníku, délka známa v době překladu
    \begin{itemize}
      \item lokální proměnné, lokální pole
      \item paměť je alokována a dealokována automaticky
      \item int array[10];
    \end{itemize}

    \pause

    \item Paměť alokovaná za běhu na zásobníku, délka není známa v době překladu
    \begin{itemize}

      \item paměť je alokována a dealokována automaticky

    \end{itemize}

      \item programátor musí hlídat uvolnění, jinak memory leak
      \item int* array=malloc(userGivenLength*sizeof(int)); free(array);
    \end{itemize}

  \end{itemize}
