\begin{frame}{Rozdíly mezi C a C++}

  \begin{itemize}
    \item Objektovost
    \begin{itemize}
      \item C++ má \code[language=C++]{class}, C ne
      \item Neexistují metody tříd (\code{cls->method()}) ani dědičnost
      \item Máme ale struktury (\code{struct}, viz dále)
    \end{itemize}

    \pause

    \item Šablony
    \begin{itemize}
      \item C++ má \code[language=C++]{template}, C ne
      \item C neumí \code[language=C++]{set<int>}, můžeme udělat jen obecnou
        \code{set}
    \end{itemize}

    \pause

    \item Rychlost překladu
    \begin{itemize}
      \item C výrazně rychlejší
      \item C++ je špatně navržený jazyk (z pohledu teorie a parsování)
    \end{itemize}

  \end{itemize}

\end{frame}

      \item Ale pozor na \code{char}: může platit \code{char} $\neq$
	\code{signed char}
    \end{itemize}

    \pause

      \item Změna na neznaménkový

      \item Přesun rozsahu hodnot do kladných
      \item Např.\ \code{unsigned long cislo1}, \code{unsigned int cislo2}
    \end{itemize}

    \pause

    \item Detailně: \url{http://en.wikipedia.org/wiki/C_data_types}
  \end{itemize}

\begin{frame}{Velikost typů}

    \item Velikost typů se na různých platformách liší
    \begin{itemize}
      \item Zjistíme operátorem \code{sizeof()}
    \end{itemize}

    \pause

    \item Standard předepisuje následující vztahy
    \begin{itemize}
      \item \code{sizeof(_Bool)} $\leq$ \code{sizeof(char)} $\leq$
        \code{sizeof(short)} $\leq$ \code{sizeof(int)} $\leq$
	\code{sizeof(long)} $\leq$ \code{sizeof(long long)}

      \pause

      \item \code{sizeof(float)} $\leq$ \code{sizeof(double)} $\leq$
        \code{sizeof(long double)}

      \pause

      \item \code{sizeof(char)} $=$ 1
      \begin{itemize}
	\item Nemusí být ale 8 bitů, některé chipy mají např.\ 16 bitů
      \end{itemize}
    \end{itemize}
  \end{itemize}

\begin{frame}{Logické operátory}
  \begin{itemize}
    \item Operátory vyhodnocující logickou hodnotu výrazu

    \pause

    \item A zároveň, AND: \code{\&\&}
    \begin{itemize}
      \item Oba dva operandy musejí být pravda
      \item Př.: \code{(prom1 == 1) \&\& (prom2 \% 10 == 5)}
    \end{itemize}

    \pause

    \item Nebo, OR: \code{||}
    \begin{itemize}
      \item Alespoň jeden operand musí být pravda
      \item Př.: \code{(prom1 == 1) || (prom2 \% 2 == 1)}
    \end{itemize}

    \pause

    \item Logická negace, NOT: \code{!}
    \begin{itemize}
      \item Logická inverze operandu
      \item Př.: \code{!(prom \% 2 == 1)}
    \end{itemize}

    \pause

    \item \pozor\ na líné vyhodnocování
    \begin{itemize}
      \item Pokud je znám logický výsledek, zbytek výrazu se nevyhodnocuje
%      \item Podvýraz na FALSE pro \code{\&\&}, podvýraz na TRUE pro \code{||}
      \item Pozor na vedlejší efekty (resp.\ jejich nepřítomnost)

      \pause

      \item V \code{x = ((5 == 6) \&\& (funkce() == 1));} \\
      \code{(funkce() == 1)} nebude vůbec vyhodnoceno a \code{x = 0} (FALSE)
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}{Zvýšení a snížení o \code{1}}

  \begin{itemize}
    \item Užitečná zkratka aritmetického operátoru \code{+ 1} resp.\ \code{\- 1}

    \pause

    \item Postfixová notace
    \begin{itemize}
      \item \code{A++} je zkratka pro \code{A = A + 1}
      \item \code{A--} je zkratka pro \code{A = A - 1}

      \pause

      \item \code{B = A++;} je zkratka pro \code{B = A; A = A + 1;}
      \begin{itemize}
	\item \code{++} je vyhodnoceno a \code{A} změněno \emph{po} přiřazení
      \end{itemize}
    \end{itemize}

    \pause

    \item Prefixová notace
    \begin{itemize}
      \item \code{++A} je zkratka pro \code{A = A + 1}
      \item \code{\--A} je zkratka pro \code{A = A - 1}

      \pause

      \item \code{B = ++A;} je zkratka pro \code{A = A + 1; B = A;}
      \begin{itemize}
	\item \code{++} je vyhodnoceno a \code{A} změněno \emph{před} přiřazením
      \end{itemize}
    \end{itemize}

    \pause

    \item \pozor\ ale např.\ na \code{a = b++ + b++;}
    \begin{itemize}
      \item Není definované, zda \code{a} bude \code{2 * b}

      \pause

      \item Sekvenční body
      \begin{itemize}
	\item Místa v programu, kde jsou ukončeny dopady předchozích výrazů
	\item \tinyL \url{http://publications.gbdirect.co.uk/c_book/chapter8/sequence_points.html}
      \end{itemize}
    \end{itemize}

  \end{itemize}

\end{frame}

\begin{frame}[fragile]{Cyklus FOR}

      \end{itemize}

      \item Např.\ projití řetězce po písmenkách
    \end{itemize}

  \end{itemize}

\begin{frame}[fragile]{Cyklus DO--WHILE}

  \code{do \{ prikazy \} while (podminka);}

    \begin{itemize}

\begin{frame}[fragile]{Nejdůležitější funkce pro řetězce}

  \begin{itemize}

    \pause

    \item \code{strcpy(a, b)} -- kopíruje obsah řetězce \code{b} do \code{a}
    \begin{itemize}
      \item \code{a} musí být dostatečně dlouhý pro \code{b}
    \end{itemize}

    \item \code{strcat(a, b)} -- připojí řetězec \code{b} za \code{a}
    \begin{itemize}

    \end{itemize}

    \pause

    \begin{itemize}
      \item Vrací $0$, pokud jsou \code{a} a \code{b} shodné (znaky i délka)
      \item Vrací $>0$ pokud je \code{a} ,,větší'' než \code{b}
      \item Vrací $<0$ pokud je \code{a} ,,menší'' než \code{b}
    \end{itemize}

  \end{itemize}

      \item Automatické odstranění na konci bloku
    \end{itemize}

    \pause

    \item V kódu se tedy explicitně neuvádí

    \pause

    \item \pozor: v C++11 má jiný význam

  \end{itemize}

\begin{frame}[fragile]{Klíčové slovo \texttt{static}}

  \begin{itemize}
    \item Proměnná deklarovaná \code{static} zachová svou hodnotu i po konci
      bloku s deklarací

    \pause

    \item Statické proměnné jsou inicializovány v době překladu
    \begin{itemize}
      \item Trvalé místo pro proměnnou stejně jako pro globální proměnné
      \item Při novém ,,vzniku'' proměnné obsahuje poslední hodnotu
    \end{itemize}

    \pause

    \item Hodnota proměnné je zachována jen v rámci 1 spuštění programu

    \pause

    \item Proměnná se \code{static} je lokální v rámci \emph{souboru}

  \end{itemize}

\onslide<1->

\begin{block}{Příklad}
\begin{lstlisting}
void foo() {
  static int a = 10;
  printf("%d\n", a);
  a += 10;
}
void fun(void) {
  foo(); // => 10
  foo(); // => 20
  foo(); // => 30
}
\end{lstlisting}
\end{block}

\end{frame}

\begin{frame}[fragile]{Klíčové slovo \texttt{volatile}}

  \begin{itemize}
    \item Proměnná může být měněna i mimo náš kód
    \begin{itemize}
      \item Rutinou přerušení, hardwarem, sdílená paměť, \dots
      \item Z analýzy zdrojového kódu nelze určit místa změny proměnné
    \end{itemize}

    \pause

    \item Nahráje hodnotu proměnné do registru před každou operací
    \begin{itemize}
      \item Překladač nemůže příliš optimalizovat
    \end{itemize}

  \end{itemize}

\onslide<1->

  msleep(100);

\end{block}

\end{frame}

\begin{frame}[fragile]{Klíčové slovo \texttt{register}}

    \end{itemize}

    \end{itemize}

    \pause

    \item Proměnné mohou být umístěny v registru i bez \code{register}
    \begin{itemize}
      \item Překladač sám analyzuje a vybere často používané proměnné
      \item Tj.\ \code{register} nepoužívejte, pokud nemusíte
    \end{itemize}

  \end{itemize}

    \item Snažte se minimalizovat jejich používání

      \item \code{\#include} je OK
      \item \code{\#ifndef SOUBOR_H} je OK

    \pause

      \item \code{inline int add(int a, int b) \{ return a + b; \}} \\

        \code{\#define ADD(a, b) ((a) + (b))}
    \end{itemize}

    \pause

    \item Podmíněný překlad častý při využití platformě závislých operací

\begin{frame}{Bitové operátory}{Využití}

  \begin{itemize}
    \item Sada příznaků TRUE/FALSE (pro úsporu prostoru)
    \begin{itemize}
      \item Např.\ do jednoho \code{int} (32b) uchováme 32 hodnot
      \item \code{unsigned int flags = 0x00000000;}
      \item Pozn.: jednotlivé bity odpovídají násobkům dvojky
    \end{itemize}

    \pause

    \item Vložení hodnoty na pozici X
    \begin{itemize}
      \item Vypočteme masku jako $2^{(X-1)}$ (indexujeme od 0, proto $X - 1$)
      \begin{itemize}
	\item Např.\ třetí bit $\rightarrow$ $2^2$ $\rightarrow$ 4
	  $\rightarrow$ 0x04 hexadecimálně
      \end{itemize}

      \pause

      \item Aplikujeme operaci OR s vypočtenou maskou
      \begin{itemize}
	\item \code{flags = flags | 0x04; // set 3rd bit to 1}
      \end{itemize}
    \end{itemize}

    \pause

    \item Zjištění hodnoty z pozice X
    \begin{itemize}
      \item Vypočteme masku $2^{(X-1)}$
      \item \code{if (flags \& 0x04) \{ /*...*/ \}}
    \end{itemize}

    \pause

    \item Zjištění hodnoty dolního bajtu
    \begin{itemize}
      \item Maska pro celý bajt je \code{255}, tedy \code{0xFF}
      \item \code{unsigned char lowByte = flags \& 0xFF;}
    \end{itemize}

%    \item ! Nepoužívejte se signed hodnotami (není fixní bitová reprezentace)
  \end{itemize}

\end{frame}

\begin{frame}{Bitové operátory}{Využití}

  \begin{itemize}
    \item Operace nutné na úrovni bitů
    \begin{itemize}
      \item Např.\ převod BASE64, šifrovací algoritmy, \dots
      \item Maska pro dolních 6 bitů: $2^0+2^1+2^2+2^3+2^4+2^5 = 63_{10} =$
        \code{0x3F}
      \item Maska pro horní 2 bity: $2^6+2^7 = 192_{10} =$ \code{0xC0}
    \end{itemize}

    \pause

    \item Rychlé násobení mocninou dvojky $X \cdot 2^{posun}$
    \begin{itemize}
      \item $0011_2$ (3 dekadicky)

      \pause

      \item $0011_2$ \code{<<} $1 \rightarrow 0110_2$ (6 dekadicky)

      \pause

      \item $0011_2$ \code{<<} $2 \rightarrow 1100_2$ (12 dekadicky)
    \end{itemize}

    \pause

    \item Pozor na rozlišení \code{\&} a \code{\&\&}, resp.\ \code{|} a
      \code{II}
    \begin{itemize}
      \item Např.\ záměna \code{\&} za \code{\&\&}
      \begin{itemize}
	\item $1100_2$ \code{\&\&} $0011_2$ je TRUE
	\item $1100_2$ \code{\&} $0011_2$ je FALSE
      \end{itemize}
    \end{itemize}

  \end{itemize}

\end{frame}
